64        TMM::check_availability::<R, MP>(client, &config.tmm_config)

66
67    fn make_config(
68        input: Self::Input,
69        problem: &MatmulProblem,
70        cube_dim: &CubeDim,
71        cube_count: &CubeCount,
72        advanced_config: &AdvancedConfig,
73    ) -> Self::Config {
74        let tile = input.tile;
75        let tmm_config = TMM::make_config(tile, problem, cube_dim, cube_count, advanced_config);
76        let tmm_size = TMM::size(&tmm_config);
77        let stage_size = stage_matmul_size::<TMM>(&tmm_config, &input.num_stages);
78
79        let (tile_m, tile_n, tile_k) = (tmm_size.m, tmm_size.n, tmm_size.k);
80        let (lhs_stage_dim, rhs_stage_dim, out_stage_dim) = create_stage_dim(
81            stage_size.m,
82            stage_size.n,
83            stage_size.k,
84            tile_m,
85            tile_n,
86            tile_k,
87        );
88
89        CommonStageConfig::new(
90            tmm_config,
91            input.num_stages,
92            lhs_stage_dim,
93            rhs_stage_dim,
94            out_stage_dim,
95            cube_dim.y,
96            advanced_config.lhs_tiling_order,
97            advanced_config.rhs_tiling_order,
98        )
99    }
100}

102/// Performs matrix multiplication at the stage level, where each plane is responsible for a row of tiles:
103/// - One plane per tile in m dimension,
104/// - One accumulator per tile in n dimension
105///
106/// # Assumptions
107/// - There are as many planes as the stage size in m
108pub struct MultiBufferMatmul<I: Numeric, O: Numeric, EA: Numeric, TMM: tile::TileMatmul<I, EA>> {
109    _input_precision: PhantomData<I>,
110    _output_precision: PhantomData<O>,
111    _accumulator_precision: PhantomData<EA>,
112    _instruction: PhantomData<TMM>,
113}
114
115#[cube]
116impl<I, O, EA, TMM> StageMatmul<I, O, EA> for MultiBufferMatmul<I, O, EA, TMM>
117where
118    I: Numeric,
119    O: Numeric,
120    EA: Numeric,
121    TMM: tile::TileMatmul<I, EA>,
122{
123    type Config = CommonStageConfig<TMM::Config>;
124
125    type LhsReader = LhsReader<I>;
126    type RhsReader = RhsReader<I>;
127    type Accumulator = Sequence<TMM::Accumulator>;
128    type LhsTile = TMM::Lhs;
129    type RhsTile = TMM::Rhs;
130
131    fn execute(
132        lhs_reader: &LhsReader<I>,
133        rhs_reader: &RhsReader<I>,
134        lhs_tile: &mut Self::LhsTile,
135        rhs_tile: &mut Self::RhsTile,
136        acc: &mut Self::Accumulator,
137        #[comptime] config: Self::Config,
138    ) {
139        #[unroll]
140        for buffer_iter in 0..config.num_stage.k {
141            let tile_lhs =
142                LhsReader::read_tile::<TMM::Config>(lhs_reader, UNIT_POS_Y, buffer_iter, config);
143            TMM::fill_lhs(&tile_lhs, lhs_tile, config.to_tmm_config());
144
145            #[unroll]
146            for accumulator_iter in 0..acc.len() {
147                let tile_rhs = RhsReader::read_tile::<TMM::Config>(
148                    rhs_reader,
149                    buffer_iter,
150                    accumulator_iter,
151                    config,
152                );
153                TMM::fill_rhs(&tile_rhs, rhs_tile, config.to_tmm_config());
154
155                let accumulator = acc.index_mut(accumulator_iter);
156                TMM::execute(lhs_tile, rhs_tile, accumulator, config.to_tmm_config());
157            }
158        }
159    }
160
161    fn init_tile_inputs(#[comptime] config: Self::Config) -> (TMM::Lhs, TMM::Rhs) {
162        (
163            TMM::init_lhs(config.to_tmm_config()),
164            TMM::init_rhs(config.to_tmm_config()),
165        )
166    }
167
168    fn read_accumulator<SW: StageWriter<O>, G: global::GlobalConfig>(
169        acc: &Self::Accumulator,
170        out: &mut SW,
171        #[comptime] stage_config: Self::Config,
172        #[comptime] global_config: G,
173    ) {
174        let out_smem_line_size = global_config.stage_line_size(Ident::Out);
175        let num_tile_lines =
176            stage_config.stage_dim(Ident::Out).tile_num_elements() / out_smem_line_size;
177
178        let start = num_tile_lines * UNIT_POS_Y;
179        let mut out_smem = SharedMemory::<O>::new_lined(
180            num_tile_lines * stage_config.num_planes(),
181            out_smem_line_size,
182        );
183
184        #[unroll]
185        for accumulator_iter in 0..acc.len() {
186            let accumulator = acc.index(accumulator_iter);
187            let mut smem_slice = out_smem.slice_mut(start, start + num_tile_lines);
188            TMM::read_accumulator(accumulator, &mut smem_slice, stage_config.to_tmm_config());
189            SW::write::<O, G>(
190                out,
191                smem_slice.to_slice(),
192                UNIT_POS_Y,
193                accumulator_iter,
194                global_config,
195            );
196        }
197    }
198
199    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator {
200        let mut acc = Sequence::<TMM::Accumulator>::new();
201
202        #[unroll]
203        for _ in 0..config.num_stage.n {
204            acc.push(TMM::init_accumulator(config.to_tmm_config()));
205        }
206
207        acc
208    }
209
210    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config) {
211        #[unroll]
212        for i in 0..config.num_stage.n {
213            TMM::zero_accumulator(acc.index_mut(i), config.to_tmm_config());
214        }
215    }
216
217    fn fill_accumulator<L: AccumulatorLoader<O, EA, Self::Config>>(
218        loader: &mut L,
219        acc: &mut Self::Accumulator,
220        #[comptime] config: Self::Config,
221    ) {
222        #[unroll]
223        for i in 0..config.num_stage.n {
224            let acc = acc.index_mut(i);
225            L::load::<I, TMM>(loader, acc, i, config.to_tmm_config());
226        }
227    }
228}

230fn check_num_planes(
231    expected_num_planes: u32,
232    actual_num_planes: u32,
233) -> Result<(), InvalidConfigError> {
234    if expected_num_planes != actual_num_planes {
235        return Err(Box::new("Error: Expected {expected_num_planes} planes, but found {actual_num_planes}. 
236        The number of planes is equal to cube dimension y which should be set to {expected_num_planes}."));
237    }
238
239    Ok(())
