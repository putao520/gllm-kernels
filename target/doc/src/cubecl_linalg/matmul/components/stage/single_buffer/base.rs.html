44impl<TMM> MatmulConfigFactory for SingleBufferMatmulFamily<TMM>
45where
46    TMM: TileMatmulFamily,

48    type Input = CommonStageInput<TMM>;
49    type Config = CommonStageConfig<TMM::Config>;
50
51    fn check_config(config: &Self::Config) -> Result<(), InvalidConfigError> {
52        TMM::check_config(&config.to_tmm_config())
53    }
54
55    fn check_availability<R: Runtime, MP: MatmulPrecision>(
56        client: &ComputeClient<R::Server, R::Channel>,
57        config: &Self::Config,
58    ) -> Result<(), MatmulAvailabilityError> {
59        TMM::check_availability::<R, MP>(client, &config.tmm_config)
60    }
61
62    fn make_config(
63        input: Self::Input,
64        problem: &MatmulProblem,
65        cube_dim: &CubeDim,
66        cube_count: &CubeCount,
67        advanced_config: &AdvancedConfig,
68    ) -> Self::Config {
69        let tile = input.tile;
70
71        let tmm_config = TMM::make_config(tile, problem, cube_dim, cube_count, advanced_config);
72        let tmm_size = TMM::size(&tmm_config);
73        let stage_size = stage_matmul_size::<TMM>(&tmm_config, &input.num_stages);
74
75        let (tile_m, tile_n, tile_k) = (tmm_size.m, tmm_size.n, tmm_size.k);
76        let (lhs_stage_dim, rhs_stage_dim, out_stage_dim) = create_stage_dim(
77            stage_size.m,
78            stage_size.n,
79            stage_size.k,
80            tile_m,
81            tile_n,
82            tile_k,
83        );
84
85        CommonStageConfig::new(
86            tmm_config,
87            input.num_stages,
88            lhs_stage_dim,
89            rhs_stage_dim,
90            out_stage_dim,
91            lhs_stage_dim.num_tiles_x_dim(),
92            advanced_config.lhs_tiling_order,
93            advanced_config.rhs_tiling_order,
94        )
95    }
96}

98/// Performs matrix multiplication at the stage level, where each plane is responsible for a row of tiles:
99/// - One plane per tile in m dimension,
100/// - One accumulator per tile in n dimension
101///
102/// Very similar to multi buffer, except is unable to have more than one buffer, and takes BufferReaders for StageReaders
103///
104/// # Assumptions
105/// - There are at least as many planes as the stage size in m
106pub struct SingleBufferMatmul<I: Numeric, O: Numeric, EA: Numeric, TMM: TileMatmul<I, EA>> {
107    _input_precision: PhantomData<I>,
108    _output_precision: PhantomData<O>,
109    _accumulator_precision: PhantomData<EA>,
110    _instruction: PhantomData<TMM>,
111}
112
113#[cube]
114impl<I, O, EA, TMM> stage::StageMatmul<I, O, EA> for SingleBufferMatmul<I, O, EA, TMM>
115where
116    I: Numeric,
117    O: Numeric,
118    EA: Numeric,
119    TMM: TileMatmul<I, EA>,
120{
121    type Config = CommonStageConfig<TMM::Config>;
122    type LhsReader = LhsBufferReader<I>;
123    type RhsReader = RhsBufferReader<I>;
124    type Accumulator = Sequence<TMM::Accumulator>;
125    type LhsTile = TMM::Lhs;
126    type RhsTile = TMM::Rhs;
127
128    fn execute(
129        lhs_reader: &LhsBufferReader<I>,
130        rhs_reader: &RhsBufferReader<I>,
131        lhs_tile: &mut Self::LhsTile,
132        rhs_tile: &mut Self::RhsTile,
133        acc: &mut Self::Accumulator,
134        #[comptime] config: Self::Config,
135    ) {
136        let tile_lhs = LhsBufferReader::read_tile::<TMM::Config>(lhs_reader, UNIT_POS_Y, config);
137        TMM::fill_lhs(&tile_lhs, lhs_tile, config.to_tmm_config());
138
139        #[unroll]
140        for accumulator_iter in 0..acc.len() {
141            let tile_rhs =
142                RhsBufferReader::read_tile::<TMM::Config>(rhs_reader, accumulator_iter, config);
143            TMM::fill_rhs(&tile_rhs, rhs_tile, config.to_tmm_config());
144
145            let accumulator = acc.index_mut(accumulator_iter);
146            TMM::execute(lhs_tile, rhs_tile, accumulator, config.to_tmm_config());
147        }
148    }
149
150    fn init_tile_inputs(#[comptime] config: Self::Config) -> (TMM::Lhs, TMM::Rhs) {
151        (
152            TMM::init_lhs(config.to_tmm_config()),
153            TMM::init_rhs(config.to_tmm_config()),
154        )
155    }
156
157    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator {
158        let mut accumulators = Sequence::<TMM::Accumulator>::new();
159
160        #[unroll]
161        for _ in 0..config.num_stage.n {
162            accumulators.push(TMM::init_accumulator(config.to_tmm_config()));
163        }
164
165        accumulators
166    }
167
168    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config) {
169        #[unroll]
170        for i in 0..config.num_stage.n {
171            TMM::zero_accumulator(acc.index_mut(i), config.to_tmm_config());
172        }
173    }
174
175    fn fill_accumulator<L: AccumulatorLoader<O, EA, Self::Config>>(
176        loader: &mut L,
177        acc: &mut Self::Accumulator,
178        #[comptime] config: Self::Config,
179    ) {
180        #[unroll]
181        for i in 0..config.num_stage.n {
182            let acc = acc.index_mut(i);
183            L::load::<I, TMM>(loader, acc, i, config.to_tmm_config());
184        }
185    }
186
187    fn read_accumulator<SW: StageWriter<O>, G: global::GlobalConfig>(
188        acc: &Self::Accumulator,
189        out: &mut SW,
190        #[comptime] stage_config: Self::Config,
191        #[comptime] global_config: G,
192    ) {
193        let out_smem_line_size = global_config.stage_line_size(Ident::Out);
194        let num_tile_lines =
195            stage_config.stage_dim(Ident::Out).tile_num_elements() / out_smem_line_size;
196
197        let start = num_tile_lines * UNIT_POS_Y;
198        let mut out_smem = SharedMemory::<O>::new_lined(
199            num_tile_lines * stage_config.num_planes(),
200            out_smem_line_size,
201        );
202
203        #[unroll]
204        for accumulator_iter in 0..acc.len() {
205            let accumulator = acc.index(accumulator_iter);
206            let mut smem_slice = out_smem.slice_mut(start, start + num_tile_lines);
207            TMM::read_accumulator(accumulator, &mut smem_slice, stage_config.to_tmm_config());
208            SW::write::<O, G>(
209                out,
210                smem_slice.to_slice(),
211                UNIT_POS_Y,
212                accumulator_iter,
213                global_config,
214            );
215        }
216    }
