17pub(super) fn stage_matmul_size<TMM: TileMatmulFamily>(
18    config: &TMM::Config,
19    num_stage: &MatmulSize,
20) -> MatmulSize {
21    let size = TMM::size(config);
22
23    MatmulSize {
24        m: num_stage.m * size.m,
25        n: num_stage.n * size.n,
26        k: num_stage.k * size.k,
27    }
28}

30#[derive(CubeType, Copy, Clone, Debug, Hash, PartialEq, Eq)]
31/// Configuration for the single buffer matmul
32pub struct CommonStageConfig<T: TileConfig> {
33    pub tmm_config: T,
34    pub num_stage: MatmulSize,
35    pub lhs_stage_dim: LhsStageDim,
36    pub rhs_stage_dim: RhsStageDim,
37    pub out_stage_dim: OutStageDim,
38    pub num_planes: u32,
39    pub lhs_tiling_order: TilingOrderConfig,
40    pub rhs_tiling_order: TilingOrderConfig,
41}
42
43impl<T: TileConfig> StageConfig for CommonStageConfig<T> {
44    type TmmConfig = T;
45
46    fn to_tmm_config(self) -> Self::TmmConfig {
47        self.tmm_config
48    }
49
50    fn line_size(&self, ident: Ident) -> u32 {
51        self.tmm_config.line_size(ident)
52    }
53
54    fn stage_dim(&self, ident: Ident) -> Box<dyn StageDim> {
55        match ident {
56            Ident::Lhs => Box::new(self.lhs_stage_dim),
57            Ident::Rhs => Box::new(self.rhs_stage_dim),
58            Ident::Out => Box::new(self.out_stage_dim),
59        }
60    }
61
62    fn layout(&self, ident: Ident) -> MatrixLayout {
63        self.tmm_config.layout(ident)
64    }
65
66    fn num_planes(&self) -> u32 {
67        self.num_planes
68    }
69
70    fn plane_dim(&self) -> u32 {
71        self.tmm_config.plane_dim()
72    }
73
74    fn tiling_order(&self, ident: Ident) -> TilingOrderConfig {
75        match ident.as_input() {
76            InputIdent::Lhs => self.lhs_tiling_order,
77            InputIdent::Rhs => self.rhs_tiling_order,
78        }
79    }
80
81    fn num_stages(&self) -> &MatmulSize {
82        &self.num_stage
83    }
84}
