16#[cube]
17/// Provides matrix multiplication operations at the tile level.
18///
19/// At the tile level,
20///  - Inputs are raw slices of data, called tiles.
21///  - units within one plane can collaborate to solve the problem
22///  - dimensions M, N and K are fixed to an integer, and the
23///    matrix multiplication works only for size (M, K) Â· (K, N) = (M, N).
24///
25/// Assumptions:
26///  - Slices given as inputs must always be valid. If the actual matrix multiplication
27///    should be done on smaller sizes than M, N and K, padding with zeros must be done beforehand.
28///  - Enough units are present to perform the whole computation
29pub trait TileMatmul<I: Numeric, O: Numeric>: 'static + Send + Sync {
30    type Config: TileConfig;
31    /// Contains LHS data that can be split across the units
32    type Lhs: CubeType;
33    /// Contains RHS data that can be split across the units
34    type Rhs: CubeType;
35    /// Contains output data that can be split across the units
36    type Accumulator: CubeType;
37
38    /// Executes the matrix multiplication of LHS and RHS, adding the result to the output
39    fn execute(
40        lhs: &Self::Lhs,
41        rhs: &Self::Rhs,
42        out: &mut Self::Accumulator,
43        #[comptime] config: Self::Config,
44    );
45
46    /// Create the container for LHS data
47    ///
48    /// # Safety
49    ///
50    /// This may point towards uninitialized memory.
51    /// Make sure to call fill_lhs prior to execute.
52    fn init_lhs(#[comptime] config: Self::Config) -> Self::Lhs;
53
54    /// Create the container for RHS data
55    ///
56    /// # Safety
57    ///
58    /// This may point towards uninitialized memory.
59    /// Make sure to call fill_rhs prior to execute.
60    fn init_rhs(#[comptime] config: Self::Config) -> Self::Rhs;
61
62    /// Fill the container of LHS with data
63    fn fill_lhs(slice: &Slice<Line<I>>, lhs: &mut Self::Lhs, #[comptime] config: Self::Config);
64
65    /// Fill the container of RHS with data
66    fn fill_rhs(slice: &Slice<Line<I>>, rhs: &mut Self::Rhs, #[comptime] config: Self::Config);
67
68    /// Fill the accumulator with data
69    fn fill_accumulator(
70        slice: &Slice<Line<O>>,
71        acc: &mut Self::Accumulator,
72        stride: u32,
73        #[comptime] config: Self::Config,
74    );
75
76    /// Write the content of the output container to the given slice
77    fn read_accumulator<C: Numeric>(
78        out: &Self::Accumulator,
79        slice: &mut SliceMut<Line<C>>,
80        #[comptime] config: Self::Config,
81    );
82
83    /// Create the container to receive the execution output.
84    ///
85    /// # Safety
86    ///
87    /// The output container must be initialized to some value (typically 0),
88    /// because the execution adds to the already present value.
89    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator;
90
91    /// Set the accumulator to zeros
92    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config);
93}
