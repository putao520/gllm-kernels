7#[derive(CubeType)]
8/// Writes the contents of a tile to the tensor view using a single plane,
9/// iterating with steps determined by the plane's dimension.
10pub struct TilewiseUnloading {}
11
12#[cube]
13impl TilewiseUnloading {
14    pub fn unload_from_slice<EG: Numeric, ES: Numeric, G: GlobalConfig>(
15        write_view: &mut TensorWriter<EG>,
16        slice: Slice<Line<ES>>,
17        tile_x: u32,
18        tile_y: u32,
19        #[comptime] config: G,
20    ) {
21        let stage_dim = config.stage_dim(Ident::Out);
22        let slice_line_size = config.stage_line_size(Ident::Out);
23        let out_line_size = config.global_line_size(Ident::Out);
24
25        let unit_step = config.plane_dim() * out_line_size;
26        let num_unit_writes = stage_dim.tile_num_elements() / unit_step;
27
28        #[allow(clippy::all)]
29        let _ = comptime!(check_line_size(out_line_size, slice_line_size));
30
31        for i in 0..num_unit_writes {
32            let unit_write = UNIT_POS_X * out_line_size + i * unit_step;
33
34            let value = slice[unit_write / out_line_size];
35            write_view.write_coalesced::<ES, G>(tile_x, tile_y, unit_write, value, config);
36        }
37    }
38}
39
40fn check_line_size(out_line_size: u32, slice_line_size: u32) {
41    assert_eq!(out_line_size, slice_line_size, "Error: Expected global output and output shared memory to have equal line size, but found out_line_size = {} and slice_line_size = {}.",
42        out_line_size, slice_line_size
43    );
44}
