19/// Provides matrix multiplication operations at the global level.
20///
21/// At the global level,
22///  - Inputs are views over global memory, meaning access is given to
23///    only parts of the global memory inputs at once.
24///  - All planes within a Cube can collaborate to solve the problem
25///  - Dimensions M and N are fixed to an integer, but K is arbitrary large.
26///    The matrix multiplication works only for size (M, _) Â· (_, N) = (M, N).
27///    M and N should match the underlying Stage matmul's M and N.
28///
29/// # Assumptions
30/// - Line sizes of the inputs evenly divide the dimension they are aligned with.
31///
32/// # Safety
33///
34/// It is not assumed that the matmul's dimensions match its inputs dimensions perfectly.
35/// It is therefore important that Loaders and Unloaders perform checks to avoid out-of-bounds
36/// before loading data.
37pub trait GlobalMatmul<MP: MatmulPrecision>: 'static + Send + Sync {
38    type Config: GlobalConfig;
39    type LhsLoader: InputLoader<MP::EG, MP::ES, Self::Config>;
40    type RhsLoader: InputLoader<MP::EG, MP::ES, Self::Config>;
41    type AccumulatorLoader: CubeType;
42    type Out: OutputLoader<MP::EG>;
43    type Accumulator: CubeType;
44
45    /// Performs the matrix multiplication over data loaded by the
46    /// LHS and RHS loaders, over the range given for K, and stores with
47    /// using the output unloader.
48    ///
49    /// To compute the whole range of k values, use k_range=(0, K) where
50    /// K is the K dimension of LHS and RHS.
51    fn execute(
52        lhs_loader: Self::LhsLoader,
53        rhs_loader: Self::RhsLoader,
54        unloader: Self::Out,
55        acc: &mut Self::Accumulator,
56        k_range: (u32, u32),
57        #[comptime] config: Self::Config,
58    );
59
60    /// Initialize the loader for Lhs, starting at row m and column k
61    fn init_lhs_loader(
62        lhs: VirtualTensor<MP::EG>,
63        m_offset: u32,
64        k_offset: u32,
65        batch_offset: u32,
66        #[comptime] config: Self::Config,
67    ) -> Self::LhsLoader;
68
69    /// Initialize the loader for Rhs, starting at row k and column n
70    fn init_rhs_loader(
71        rhs: VirtualTensor<MP::EG>,
72        k_offset: u32,
73        n_offset: u32,
74        batch_offset: u32,
75        #[comptime] config: Self::Config,
76    ) -> Self::RhsLoader;
77
78    /// Initialize the unloader at row m and column n
79    fn init_unloader(
80        out: VirtualTensor<MP::EG, ReadWrite>,
81        m_offset: u32,
82        n_offset: u32,
83        batch_offset: u32,
84    ) -> Self::Out;
85
86    /// Initialize the accumulator without data
87    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator;
88
89    /// Fill the accumulator with zeros
90    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config);
91}

94/// Input to the global matmul, responsible of filling the stage and providing a reader for it.
95/// Advances along the k-dimension to fill the stage with further data.
96pub trait InputLoader<EG: Numeric, ES: Numeric, G: GlobalConfig>:
97    CubeType + 'static + Send + Sync
98{
99    /// The stage reader which matches the input of the underlying stage matmul.
100    type StageReader: CubeType;
101
102    /// Fills the stage at the current k offset.
103    fn fill_stage(this: &mut Self, #[comptime] config: G);
104
105    /// Returns a reader for the stage at the current k offset
106    fn as_stage_reader(this: &Self) -> Self::StageReader;
107
108    /// Move the k offset by k_offset
109    fn advance_view(this: &mut Self, k_offset: u32);
110}

147/// Configuration for the [global matmul](GlobalMatmul) level.
148pub trait GlobalConfig: MatmulConfig {
149    /// Underlying Stage matmul config
150    type SmmConfig: stage::StageConfig;
151
152    /// Convert itself to the underlying stage matmul config
153    fn to_smm_config(&self) -> Self::SmmConfig;
154
155    /// Returns the line size for the global memory corresponding to the given ident
156    fn global_line_size(&self, ident: Ident) -> u32;
157
158    /// Returns the line size for the stage of the given ident
159    fn stage_line_size(&self, ident: Ident) -> u32;
160
161    /// Returns the [StageDim] for the given ident
162    fn stage_dim(&self, ident: Ident) -> Box<dyn StageDim>;
163
164    /// Returns the [MatrixLayout] for the given ident
165    fn layout(&self, ident: Ident) -> MatrixLayout;
166
167    /// Returns the number of planes in the cube
168    fn num_planes(&self) -> u32;
169
170    /// Returns the size of the plane dimension
171    fn plane_dim(&self) -> u32;
172
173    /// Returns the order in which tiles should be loaded to the stage
174    fn tiling_order(&self, ident: Ident) -> TilingOrderConfig;
175
176    /// Whether it is necessary to add bound checks in the m dimension
177    fn check_m_bounds(&self) -> bool;
178
179    /// Whether it is necessary to add bound checks in the k dimension
180    fn check_k_bounds(&self) -> bool;
181
182    /// Whether it is necessary to add bound checks in the n dimension
183    fn check_n_bounds(&self) -> bool;
184
185    /// Whether we transpose data when loading to the stage
186    fn transpose_load(&self, ident: Ident) -> bool;
187}
