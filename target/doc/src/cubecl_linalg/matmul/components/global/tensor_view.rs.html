43#[cube]
44impl<EG: Numeric> TensorReader<EG> {
45    /// Instantiate a read view over the given tensor, pre-fetching needed strides and shapes
46    pub fn new(tensor: VirtualTensor<EG>, x_offset: u32, y_offset: u32, batch_offset: u32) -> Self {
47        let rank = tensor.rank();
48        let stride_x = tensor.stride(rank - 2);
49        let stride_y = tensor.stride(rank - 1);
50        let shape_x = tensor.shape(rank - 2);
51        let shape_y = tensor.shape(rank - 1);
52
53        TensorReader::<EG> {
54            tensor,
55            x_offset,
56            y_offset,
57            stride_x,
58            stride_y,
59            shape_x,
60            shape_y,
61            batch_offset,
62        }
63    }
64
65    /// Advance the view along the k dimension by a specified offset, `k_offset`.
66    pub fn update_view(&mut self, k_offset: u32, #[comptime] ident: Ident) {
67        match ident.as_input() {
68            InputIdent::Lhs => {
69                self.y_offset += k_offset;
70            }
71            InputIdent::Rhs => {
72                self.x_offset += k_offset;
73            }
74        }
75    }
76
77    /// Reads data from the tensor view at the specified tile coordinates (tile_x, tile_y).
78    ///
79    /// Each unit loads one line in a coalesced manner for improved efficiency.
80    /// For row-major tensors, subsequent units read lines horizontally within the tile,
81    /// while for column-major tensors, they read lines vertically.
82    ///
83    /// # Note
84    ///
85    /// Out-of-bounds reads will be translated to zeros.
86    pub fn load_coalesced<G: global::GlobalConfig>(
87        &self,
88        tile_x: u32,
89        tile_y: u32,
90        unit_id: u32,
91        #[comptime] ident: Ident,
92        #[comptime] config: G,
93    ) -> Line<EG> {
94        let line_size = config.global_line_size(ident);
95        let tile_size_x = config.stage_dim(ident).tile_size_x_dim();
96        let tile_size_y = config.stage_dim(ident).tile_size_y_dim();
97
98        let view_tile_x = tile_x * tile_size_x + self.x_offset;
99        let view_tile_y = tile_y * tile_size_y + self.y_offset;
100
101        let (load_x, load_y) = match config.layout(ident) {
102            MatrixLayout::RowMajor => (unit_id / tile_size_y, unit_id % tile_size_y),
103            MatrixLayout::ColMajor => (unit_id % tile_size_x, unit_id / tile_size_x),
104        };
105
106        let view_x = view_tile_x + load_x;
107        let view_y = view_tile_y + load_y;
108
109        let read_pos =
110            (view_x * self.stride_x + view_y * self.stride_y + self.batch_offset) / line_size;
111
112        let (check_x_bounds, check_y_bounds) = match ident.as_input() {
113            InputIdent::Lhs => (config.check_m_bounds(), config.check_k_bounds()),
114            InputIdent::Rhs => (config.check_k_bounds(), config.check_n_bounds()),
115        };
116
117        match comptime!((check_x_bounds, check_y_bounds)) {
118            (true, true) => select(
119                view_x < self.shape_x && view_y < self.shape_y,
120                self.read(read_pos),
121                Line::empty(line_size).fill(EG::from_int(0)),
122            ),
123            (true, false) => select(
124                view_x < self.shape_x,
125                self.read(read_pos),
126                Line::empty(line_size).fill(EG::from_int(0)),
127            ),
128            (false, true) => select(
129                view_y < self.shape_y,
130                self.read(read_pos),
131                Line::empty(line_size).fill(EG::from_int(0)),
132            ),
133            (false, false) => self.read(read_pos),
134        }
135    }
136
137    fn read(&self, position: u32) -> Line<EG> {
138        self.tensor.read(position)
139    }
140}

142#[cube]
143impl<EG: Numeric> TensorWriter<EG> {
144    /// Instantiate a write view over the given tensor, pre-fetching needed strides and shapes
145    pub fn new(
146        tensor: VirtualTensor<EG, ReadWrite>,
147        x_offset: u32,
148        y_offset: u32,
149        batch_offset: u32,
150    ) -> Self {
151        let rank = tensor.rank();
152        let stride_x = tensor.stride(rank - 2);
153        let stride_y = tensor.stride(rank - 1);
154        let shape_x = tensor.shape(rank - 2);
155        let shape_y = tensor.shape(rank - 1);
156
157        TensorWriter::<EG> {
158            tensor,
159            x_offset,
160            y_offset,
161            stride_x,
162            stride_y,
163            shape_x,
164            shape_y,
165            batch_offset,
166        }
167    }
168
169    /// Writes data into the tensor view at the specified coordinates (write_x, write_y).
170    ///
171    /// Each unit writes one line in a coalesced manner for improved efficiency, assuming row-major layout.
172    pub fn write_coalesced<ES: Numeric, G: global::GlobalConfig>(
173        &mut self,
174        tile_x: u32,
175        tile_y: u32,
176        unit_id: u32,
177        value: Line<ES>,
178        #[comptime] config: G,
179    ) {
180        let stage_dim = config.stage_dim(Ident::Out);
181
182        let view_x = tile_x * stage_dim.tile_size_x_dim()
183            + unit_id / stage_dim.tile_size_y_dim()
184            + self.x_offset;
185        let view_y = tile_y * stage_dim.tile_size_y_dim()
186            + unit_id % stage_dim.tile_size_y_dim()
187            + self.y_offset;
188
189        let write_position = (view_x * self.stride_x + view_y * self.stride_y + self.batch_offset)
190            / config.global_line_size(Ident::Out);
191
192        match comptime!((config.check_m_bounds(), config.check_n_bounds())) {
193            (true, true) => {
194                if view_x < self.shape_x && view_y < self.shape_y {
195                    self.write(write_position, Line::cast_from(value));
196                }
197            }
198            (true, false) => {
199                if view_x < self.shape_x {
200                    self.write(write_position, Line::cast_from(value));
201                }
202            }
203            (false, true) => {
204                if view_y < self.shape_y {
205                    self.write(write_position, Line::cast_from(value));
206                }
207            }
208            (false, false) => {
209                self.write(write_position, Line::cast_from(value));
210            }
211        }
212    }
213
214    fn write(&mut self, position: u32, value: Line<EG>) {
215        self.tensor.write(position, value)
216    }
