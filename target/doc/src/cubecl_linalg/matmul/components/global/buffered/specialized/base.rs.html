38impl<SMM> MatmulConfigFactory for SpecializedMatmulFamily<SMM>
39where
40    SMM: stage::StageMatmulFamily,

42    type Input = SMM::Input;
43    type Config = Config<SMM::Config>;

45    fn check_config(config: &Self::Config) -> Result<(), InvalidConfigError> {
46        if config.num_producers() == 0 {
47            return Err(Box::new("There are no producer planes. Make sure there are more planes than the underlying stage matmul requires."));
48        }
49        if config.stage_dim(Ident::Lhs).num_tiles_y_dim() <= 1 {
50            return Err(Box::new("Producer-consumer needs at least 2 buffers."));
51        }
52
53        SMM::check_config(&config.to_smm_config())
54    }
55
56    fn check_availability<R: Runtime, MP: MatmulPrecision>(
57        client: &ComputeClient<R::Server, R::Channel>,
58        config: &Self::Config,
59    ) -> Result<(), MatmulAvailabilityError> {
60        SMM::check_availability::<R, MP>(client, &config.smm_config)
61    }
62
63    fn make_config(
64        input: Self::Input,
65        problem: &MatmulProblem,
66        cube_dim: &CubeDim,
67        cube_count: &CubeCount,
68        advanced_config: &AdvancedConfig,
69    ) -> Self::Config {
70        let smm_config = SMM::make_config(input, problem, cube_dim, cube_count, advanced_config);
71        let size = SMM::size(&smm_config);
72
73        Config::new(
74            smm_config,
75            problem.m as u32 % size.m != 0,
76            problem.n as u32 % size.n != 0,
77            problem.k as u32 % size.k != 0,
78            problem.lhs_layout,
79            problem.rhs_layout,
80            problem.lhs_line_size as u32,
81            problem.rhs_line_size as u32,
82            problem.out_line_size as u32,
83            cube_dim.y,
84        )
85    }
86}

88/// Performs matrix multiplication at the global level, with planes split between two roles:
89/// - First n planes are used in the stage matmul computation, with n the number of planes needed by the underlying stage matmul
90/// - Remaining planes load data to the stage
91///
92/// Both roles alternate the buffer (tile index in dimension k) they are working on
93pub struct SpecializedMatmul<MP: MatmulPrecision, SMM: StageMatmul<MP::ES, MP::EG, MP::EA>> {
94    _ms: PhantomData<MP>,
95    _stage_matmul: PhantomData<SMM>,
96}
97
98#[cube]
99impl<MP: MatmulPrecision, SMM> global::GlobalMatmul<MP> for SpecializedMatmul<MP, SMM>
100where
101    SMM: StageMatmul<
102        MP::ES,
103        MP::EG,
104        MP::EA,
105        LhsReader = LhsBufferReader<MP::ES>,
106        RhsReader = RhsBufferReader<MP::ES>,
107    >,
108{
109    type Config = Config<SMM::Config>;
110    type LhsLoader = LhsBufferLoader<MP::EG, MP::ES, SMM::Config>;
111    type RhsLoader = RhsBufferLoader<MP::EG, MP::ES, SMM::Config>;
112    type AccumulatorLoader = ZeroAccumulatorLoader;
113    type Out = Unloader<MP::EG>;
114    type Accumulator = SMM::Accumulator;
115
116    fn execute(
117        mut lhs_loader: Self::LhsLoader,
118        mut rhs_loader: Self::RhsLoader,
119        mut out_unloader: Self::Out,
120        acc: &mut Self::Accumulator,
121        k_range: (u32, u32),
122        #[comptime] config: Self::Config,
123    ) {
124        let is_consumer = Self::is_consumer(config);
125
126        let num_buffers = config.stage_dim(Ident::Lhs).num_tiles_y_dim();
127        let buffer_step = config.stage_dim(Ident::Lhs).tile_size_y_dim();
128        let k_step = num_buffers * buffer_step; // equal to SMM::K
129
130        let range = k_range.1 - k_range.0;
131        let num_stages = (range + k_step - 1) / k_step;
132        let num_loops = num_stages * num_buffers;
133
134        SMM::zero_accumulator(acc, config.to_smm_config());
135
136        let (mut lhs_tile, mut rhs_tile) = SMM::init_tile_inputs(config.to_smm_config());
137
138        for _ in 0..num_loops {
139            Self::LhsLoader::fill_stage(&mut lhs_loader, config);
140            Self::RhsLoader::fill_stage(&mut rhs_loader, config);
141
142            let lhs_stage_reader = &Self::LhsLoader::as_stage_reader(&lhs_loader);
143            let rhs_stage_reader = &Self::RhsLoader::as_stage_reader(&rhs_loader);
144
145            sync_units();
146
147            if is_consumer {
148                SMM::execute(
149                    lhs_stage_reader,
150                    rhs_stage_reader,
151                    &mut lhs_tile,
152                    &mut rhs_tile,
153                    acc,
154                    config.to_smm_config(),
155                );
156            }
157
158            Self::LhsLoader::advance_view(&mut lhs_loader, buffer_step);
159            Self::RhsLoader::advance_view(&mut rhs_loader, buffer_step);
160        }
161
162        if is_consumer {
163            SMM::read_accumulator::<Self::Out, Self::Config>(
164                acc,
165                &mut out_unloader,
166                config.to_smm_config(),
167                config,
168            );
169        }
170    }
171
172    fn init_lhs_loader(
173        lhs: VirtualTensor<MP::EG>,
174        x_offset: u32,
175        y_offset: u32,
176        batch_offset: u32,
177        #[comptime] config: Self::Config,
178    ) -> Self::LhsLoader {
179        Self::LhsLoader::new(
180            lhs,
181            x_offset,
182            y_offset,
183            batch_offset,
184            !Self::is_consumer(config),
185            config,
186        )
187    }
188
189    fn init_rhs_loader(
190        rhs: VirtualTensor<MP::EG>,
191        x_offset: u32,
192        y_offset: u32,
193        batch_offset: u32,
194        #[comptime] config: Self::Config,
195    ) -> Self::RhsLoader {
196        Self::RhsLoader::new(
197            rhs,
198            x_offset,
199            y_offset,
200            batch_offset,
201            !Self::is_consumer(config),
202            config,
203        )
204    }
205
206    fn init_unloader(
207        out: VirtualTensor<MP::EG, ReadWrite>,
208        x_offset: u32,
209        y_offset: u32,
210        batch_offset: u32,
211    ) -> Self::Out {
212        Self::Out::new(out, x_offset, y_offset, batch_offset)
213    }
214
215    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator {
216        SMM::init_accumulator(config.to_smm_config())
217    }
218
219    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config) {
220        SMM::zero_accumulator(acc, config.to_smm_config());
221    }
222}
