15/// Performs matrix multiplication at the global level, with planes pipelining their work using two buffers:
16/// While they trigger a load event from global memory to shared memory on buffer A,
17/// they trigger a computation event from tensor cores on buffer B. Then buffers are switched.
18pub struct PipelinedMatmul<MP: MatmulPrecision, SMM: stage::StageMatmul<MP::ES, MP::EG, MP::EA>> {
19    _ms: PhantomData<MP>,
20    _stage_matmul: PhantomData<SMM>,
21}
22
23#[cube]
24impl<MP: MatmulPrecision, SMM> global::GlobalMatmul<MP> for PipelinedMatmul<MP, SMM>
25where
26    SMM: stage::StageMatmul<
27        MP::ES,
28        MP::EG,
29        MP::EA,
30        LhsReader = LhsBufferReader<MP::ES>,
31        RhsReader = RhsBufferReader<MP::ES>,
32    >,
33{
34    type Config = CommonGlobalConfig<SMM::Config>;
35    type LhsLoader = LhsBufferLoader<MP::EG, MP::ES, SMM::Config>;
36    type RhsLoader = RhsBufferLoader<MP::EG, MP::ES, SMM::Config>;
37    type AccumulatorLoader = ZeroAccumulatorLoader;
38    type Out = Unloader<MP::EG>;
39    type Accumulator = SMM::Accumulator;
40
41    fn execute(
42        mut lhs_loader: Self::LhsLoader,
43        mut rhs_loader: Self::RhsLoader,
44        mut out_unloader: Self::Out,
45        acc: &mut Self::Accumulator,
46        k_range: (u32, u32),
47        #[comptime] config: Self::Config,
48    ) {
49        let num_buffers = 2;
50        let buffer_step = config.stage_dim(Ident::Lhs).tile_size_y_dim();
51        let k_step = num_buffers * buffer_step; // equal to SMM::K
52
53        let range = k_range.1 - k_range.0;
54        let num_stages = (range + k_step - 1) / k_step;
55        let num_loops = num_stages;
56
57        SMM::zero_accumulator(acc, config.to_smm_config());
58
59        let (mut lhs_tile_a, mut rhs_tile_a) = SMM::init_tile_inputs(config.to_smm_config());
60        let (mut lhs_tile_b, mut rhs_tile_b) = SMM::init_tile_inputs(config.to_smm_config());
61
62        ///////////////
63        // Load A
64        Self::LhsLoader::fill_stage(&mut lhs_loader, config);
65        Self::RhsLoader::fill_stage(&mut rhs_loader, config);
66
67        let lhs_buffer_reader_a = Self::LhsLoader::as_stage_reader(&lhs_loader);
68        let rhs_buffer_reader_a = Self::RhsLoader::as_stage_reader(&rhs_loader);
69
70        ///////////////
71        // Get B
72        Self::LhsLoader::advance_view(&mut lhs_loader, buffer_step);
73        Self::RhsLoader::advance_view(&mut rhs_loader, buffer_step);
74
75        let lhs_buffer_reader_b = Self::LhsLoader::as_stage_reader(&lhs_loader);
76        let rhs_buffer_reader_b = Self::RhsLoader::as_stage_reader(&rhs_loader);
77
78        for _ in 0..num_loops {
79            sync_units();
80
81            ///////////////
82            // Load B & Advance
83            Self::LhsLoader::fill_stage(&mut lhs_loader, config);
84            Self::RhsLoader::fill_stage(&mut rhs_loader, config);
85
86            Self::LhsLoader::advance_view(&mut lhs_loader, buffer_step);
87            Self::RhsLoader::advance_view(&mut rhs_loader, buffer_step);
88
89            ///////////////
90            // Execute A
91            SMM::execute(
92                &lhs_buffer_reader_a,
93                &rhs_buffer_reader_a,
94                &mut lhs_tile_a,
95                &mut rhs_tile_a,
96                acc,
97                config.to_smm_config(),
98            );
99
100            sync_units();
101
102            ///////////////
103            // Load Next A
104            Self::LhsLoader::fill_stage(&mut lhs_loader, config);
105            Self::RhsLoader::fill_stage(&mut rhs_loader, config);
106
107            Self::LhsLoader::advance_view(&mut lhs_loader, buffer_step);
108            Self::RhsLoader::advance_view(&mut rhs_loader, buffer_step);
109
110            ///////////////
111            // Execute B
112            SMM::execute(
113                &lhs_buffer_reader_b,
114                &rhs_buffer_reader_b,
115                &mut lhs_tile_b,
116                &mut rhs_tile_b,
117                acc,
118                config.to_smm_config(),
119            );
120        }
121
122        sync_units();
123
124        SMM::read_accumulator::<Self::Out, Self::Config>(
125            acc,
126            &mut out_unloader,
127            config.to_smm_config(),
128            config,
129        );
130    }
131
132    fn init_lhs_loader(
133        lhs: VirtualTensor<MP::EG>,
134        x_offset: u32,
135        y_offset: u32,
136        batch_offset: u32,
137        #[comptime] config: Self::Config,
138    ) -> Self::LhsLoader {
139        Self::LhsLoader::new(lhs, x_offset, y_offset, batch_offset, config)
140    }
141
142    fn init_rhs_loader(
143        rhs: VirtualTensor<MP::EG>,
144        x_offset: u32,
145        y_offset: u32,
146        batch_offset: u32,
147        #[comptime] config: Self::Config,
148    ) -> Self::RhsLoader {
149        Self::RhsLoader::new(rhs, x_offset, y_offset, batch_offset, config)
150    }
151
152    fn init_unloader(
153        out: VirtualTensor<MP::EG, ReadWrite>,
154        x_offset: u32,
155        y_offset: u32,
156        batch_offset: u32,
157    ) -> Self::Out {
158        Self::Out::new(out, x_offset, y_offset, batch_offset)
159    }
160
161    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator {
162        SMM::init_accumulator(config.to_smm_config())
163    }
164
165    fn zero_accumulator(acc: &mut Self::Accumulator, #[comptime] config: Self::Config) {
166        SMM::zero_accumulator(acc, config.to_smm_config());
167    }
