80/// Executes matrix multiplication at the batch level,
81/// assigning each cube to a single global matmul.
82///
83/// Note: This algorithm requires one cube per global matmul;
84/// insufficient cubes will result in incomplete computations.
85pub struct OneToOneMatmul<MP: MatmulPrecision, GMM: GlobalMatmul<MP>, C: CubeDispatch> {
86    _mp: PhantomData<MP>,
87    _gmm: PhantomData<GMM>,
88    _c: PhantomData<C>,
89}
90
91#[cube]
92impl<MP: MatmulPrecision, GMM: GlobalMatmul<MP>, C: CubeDispatch> BatchMatmul<MP>
93    for OneToOneMatmul<MP, GMM, C>
94{
95    type Config = Config<GMM::Config, C>;
96
97    fn execute(
98        lhs: VirtualTensor<MP::EG>,
99        rhs: VirtualTensor<MP::EG>,
100        out: VirtualTensor<MP::EG, ReadWrite>,
101        #[comptime] config: Self::Config,
102    ) {
103        let (x_index, y_index) = C::x_y_indices();
104        let x_offset = x_index * config.stage_dim(Ident::Lhs).num_elements_x_dim();
105        let y_offset = y_index * config.stage_dim(Ident::Rhs).num_elements_y_dim();
106        let nth_batch = C::batch_index();
107        let rank = lhs.rank();
108        let k_range = (0, lhs.shape(rank - 1));
109
110        let gmm_config = config.to_gmm_config();
111        gmm_execute::<MP, GMM>(
112            lhs,
113            rhs,
114            out,
115            x_offset,
116            y_offset,
117            nth_batch,
118            &mut GMM::init_accumulator(gmm_config),
119            k_range,
120            gmm_config,
121        );
122    }
123}

126/// Configuration for the OneToOneBatchMatmul
127pub struct Config<G: global::GlobalConfig, C: CubeDispatch> {
128    gmm_config: G,
129    cube_count: (u32, u32, u32),
130    _c: PhantomData<C>,
131}
132
133impl<G: global::GlobalConfig, C: CubeDispatch> batch::BatchConfig for Config<G, C> {
134    type GmmConfig = G;
135
136    fn to_gmm_config(&self) -> Self::GmmConfig {
137        self.gmm_config
138    }
139
140    fn stage_dim(&self, ident: Ident) -> Box<dyn StageDim> {
141        self.gmm_config.stage_dim(ident)
142    }
143
144    fn max_m(&self) -> u32 {
145        C::max_x(self.cube_count) * self.stage_dim(Ident::Out).num_elements_x_dim()
146    }
147
148    fn max_n(&self) -> u32 {
149        C::max_y(self.cube_count) * self.stage_dim(Ident::Out).num_elements_y_dim()
150    }
151
152    fn max_batches(&self) -> u32 {
153        C::max_batches(self.cube_count)
154    }
155}
