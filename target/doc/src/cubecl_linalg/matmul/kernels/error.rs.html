1use cubecl_core::{ir::Elem, CubeCount};

6pub enum MatmulLaunchError {
7    Unavailable(MatmulAvailabilityError),
8    InvalidProblem(MatmulInvalidProblem),
9    InvalidConfig(InvalidConfigError),

12pub enum MatmulAvailabilityError {
13    PlaneDimUnknown,
14    CubeCountTooBig(CubeCount),
15    PlaneDimUnsupported {
16        plane_dim: u32,
17    },
18    PlaneOperationsUnavailable,
19    TypesUnavailable {
20        input: Elem,
21        output: Elem,
22    },
23    CmmaInstructionUnavailable {
24        input: Elem,
25        output: Elem,
26        m: u32,
27        n: u32,
28        k: u32,
29    },
30}

32pub enum MatmulInvalidProblem {
33    ExceededMSize { m: u32, max_m: u32 },
34    ExceededNSize { n: u32, max_n: u32 },
35    ExceededBatchSize { b: u32, max_b: u32 },
36    InvalidLineSizeLhs { size: u32, line_size: u8 },
37    InvalidLineSizeRhs { size: u32, line_size: u8 },
38    InvalidLineSizeOut { size: u32, line_size: u8 },
39}

41impl From<MatmulInvalidProblem> for MatmulLaunchError {
42    fn from(value: MatmulInvalidProblem) -> Self {
43        Self::InvalidProblem(value)
44    }
45}
46
47impl From<MatmulAvailabilityError> for MatmulLaunchError {
48    fn from(value: MatmulAvailabilityError) -> Self {
49        Self::Unavailable(value)
50    }
51}
52
53impl From<InvalidConfigError> for MatmulLaunchError {
54    fn from(value: InvalidConfigError) -> Self {

59impl Debug for MatmulLaunchError {
60    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
61        match self {
62            MatmulLaunchError::Unavailable(err) => {
63                writeln!(
64                    f,
65                    "Unable to launch matmul because a required feature is unavailable: {:?}",
66                    err
67                )
68            }
69            MatmulLaunchError::InvalidProblem(err) => {
70                writeln!(
71                    f,
72                    "Unable to launch matmul because the problem isn't correctly defined: {:?}",
73                    err
74                )
75            }
76            MatmulLaunchError::InvalidConfig(err) => {
77                writeln!(
78                    f,
79                    "Unable to launch matmul because the config is invalid: {:?}",
80                    err.to_string()
81                )
82            }
83        }
84    }
85}

87impl Debug for MatmulInvalidProblem {
88    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
89        match self {
90            MatmulInvalidProblem::ExceededMSize { m, max_m } => write!(
91                f,
92                "Problem has m={} but these configs can only have m<={}",
93                m, max_m
94            ),
95            MatmulInvalidProblem::ExceededNSize { n, max_n } => write!(
96                f,
97                "Problem has n={} but these configs can only have n<={}",
98                n, max_n,
99            ),
100            MatmulInvalidProblem::ExceededBatchSize { b, max_b } => write!(
101                f,
102                "Problem has {} batches but these configs can only have batches<={}",
103                b, max_b,
104            ),
105            MatmulInvalidProblem::InvalidLineSizeLhs { size, line_size } => write!(
106                f,
107                "the lhs tensor can't be read with line size={line_size} and dimension={size}"
108            ),
109            MatmulInvalidProblem::InvalidLineSizeRhs { size, line_size } => write!(
110                f,
111                "The rhs tensor can't be read with line size={line_size} and dimension={size}"
112            ),
113            MatmulInvalidProblem::InvalidLineSizeOut { size, line_size } => write!(
114                f,
115                "The out tensor can't be written with line size={line_size} and dimension={size}"
116            ),
117        }
118    }
119}

121impl Debug for MatmulAvailabilityError {
122    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
123        match self {
124            MatmulAvailabilityError::PlaneOperationsUnavailable => {
125                writeln!(f, "Plane operations not supported.")
126            }
127            MatmulAvailabilityError::CubeCountTooBig(count) => {
128                writeln!(f, "Cube count too big {count:?}")
129            }
130            MatmulAvailabilityError::PlaneDimUnknown => {
131                writeln!(f, "Plane dimension unknown.")
132            },
133            MatmulAvailabilityError::PlaneDimUnsupported{plane_dim} => {
134                writeln!(f, "Plane dimension unsupported: {plane_dim}. Only 32 & 64 are supported.")
135            },
136            MatmulAvailabilityError::TypesUnavailable { input, output } => {
137                writeln!(
138                    f,
139                    "Types input={:?} and/or output={:?} not supported.",
140                    input, output,
141                )
142            }
143            MatmulAvailabilityError::CmmaInstructionUnavailable {
144                input,
145                output,
146                m,
147                n,
148                k,
149            } => writeln!(
150                f,
151                "Cmma on inputs {:?} and outputs {:?} with shape m={:?}, n={:?}, k={:?} not supported.",
152                input,
153                output, m, n, k
154            ),
155        }
156    }
