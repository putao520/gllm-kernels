19#[derive(Debug, Clone, Default)]
20pub enum Strategy {
21    Standard,
22    Pipelined,
23    Specialized,
24    #[cfg(any(test, feature = "export_tests"))]
25    // Very slow, only use for testing.
26    PlaneMma,
27    Simple,
28    Tiling2D(Tiling2dConfig),
29    #[default]

33pub fn launch<R: Runtime, EG: Float>(

49pub fn launch_ref<R: Runtime, EG: Float>(
50    strategy: &Strategy,
51    client: &ComputeClient<R::Server, R::Channel>,
52    lhs: &TensorHandleRef<R>,
53    rhs: &TensorHandleRef<R>,
54    out: &TensorHandleRef<R>,
55) -> Result<(), MatmulLaunchError> {
56    match strategy {
57        Strategy::Standard => {
58            matmul::launch_ref::<R, EG, StandardSelector<Accelerated>>(client, lhs, rhs, out)
59        }
60        Strategy::Pipelined => {
61            matmul::launch_ref::<R, EG, PipelinedSelector<Accelerated>>(client, lhs, rhs, out)
62        }
63        Strategy::Specialized => {
64            matmul::launch_ref::<R, EG, SpecializedSelector<Accelerated>>(client, lhs, rhs, out)
65        }
66        #[cfg(any(test, feature = "export_tests"))]
67        Strategy::PlaneMma => {
68            matmul::launch_ref::<R, EG, StandardSelector<super::components::tile::plane::PlaneMma>>(
69                client, lhs, rhs, out,
70            )
71        }
72        Strategy::Tiling2D(config) => {
73            tiling2d::launch_ref::<R, EG>(client, lhs, rhs, out, config.clone());
74            Ok(())
75        }
76        Strategy::Simple => {
77            simple::launch_ref::<R, EG>(client, lhs, rhs, out)?;
78            Ok(())
79        }
80        Strategy::Auto => {
81            if let Err(err) =
82                matmul::launch_ref::<R, EG, StandardSelector<Accelerated>>(client, lhs, rhs, out)
83            {
84                match err {
85                    super::kernels::MatmulLaunchError::Unavailable(_) => {
86                        tiling2d::launch_ref::<R, EG>(
87                            client,
88                            lhs,
89                            rhs,
90                            out,
91                            Tiling2dConfig::default(),
92                        )
93                    }
94                    _ => panic!("{err:?}"),
95                }
96            }
97
98            Ok(())
99        }
100    }
101}
