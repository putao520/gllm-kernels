8/// Tensor representation containing a [server handle](Handle) as well as basic tensor metadata.,
9pub struct TensorHandle<R, E>
10where
11    R: Runtime,
12    E: CubePrimitive,
13{
14    /// The buffer where the data are stored.

56impl<R, E> TensorHandle<R, E>
57where
58    R: Runtime,
59    E: CubePrimitive,
60{
61    /// Create a new tensor.
62    pub fn new(shape: Vec<usize>, strides: Vec<usize>, handle: Handle) -> Self {
63        Self {
64            shape,
65            strides,
66            handle,
67            elem: PhantomData,
68            runtime: PhantomData,
69        }
70    }
71
72    /// Create a new tensor with a contiguous memory layout.
73    pub fn new_contiguous(shape: Vec<usize>, handle: Handle) -> Self {
74        let strides = Self::contiguous_strides(&shape);
75
76        Self {
77            handle,
78            shape,
79            strides,
80            elem: PhantomData,
81            runtime: PhantomData,
82        }
83    }
84
85    /// Check if the tensor is safe to mutate.
86    pub fn can_mut(&self) -> bool {
87        self.handle.can_mut()
88    }
89
90    pub fn as_ref(&self) -> TensorHandleRef<'_, R> {
91        TensorHandleRef {
92            handle: &self.handle,
93            strides: &self.strides,
94            shape: &self.shape,
95            elem_size: E::size().expect("Should be a native type"),
96            runtime: PhantomData,
97        }
98    }
99
100    /// Return the reference to a tensor argument.
101    pub fn as_arg<'a>(&'a self, vectorisation: u8) -> TensorArg<'a, R> {
102        let handle: TensorHandleRef<'a, R> = self.as_ref();
103
104        unsafe {
105            TensorArg::from_raw_parts::<E>(
106                handle.handle,
107                handle.strides,
108                handle.shape,
109                vectorisation,
110            )
111        }
112    }
113
114    fn contiguous_strides(shape: &[usize]) -> Vec<usize> {
115        let mut strides = Vec::with_capacity(shape.len());
116
117        let mut current = 1;
118        shape.iter().enumerate().rev().for_each(|(_, val)| {
119            strides.push(current);
120            current *= val;
121        });
122        strides.reverse();
123        strides
124    }
125}
126impl<R, E> TensorHandle<R, E>
127where
128    R: Runtime,
129    E: Numeric,
130{
131    pub fn empty(client: &ComputeClient<R::Server, R::Channel>, shape: Vec<usize>) -> Self {
132        let num_elements: usize = shape.iter().product();
133        let size = E::size().expect("To be a native type");
134
135        let handle = client.empty(size * num_elements);
136        let strides = Self::contiguous_strides(&shape);
137
138        Self::new(shape, strides, handle)
139    }
140
141    pub fn zeros(client: &ComputeClient<R::Server, R::Channel>, shape: Vec<usize>) -> Self {
142        let num_elements: usize = shape.iter().product();
143        let rank = shape.len();
144        let output = Self::empty(client, shape);
145
146        let vectorization_factor = tensor_line_size_parallel(
147            R::supported_line_sizes().iter().cloned(),
148            &output.shape,
149            &output.strides,
150            rank - 1,
151        );
152
153        let cube_dim = CubeDim::default();
154        let cube_count =
155            calculate_cube_count_elemwise(num_elements / vectorization_factor as usize, cube_dim);
156
157        unsafe {
158            init::zeros_array::launch_unchecked::<E, R>(
159                client,
160                cube_count,
161                cube_dim,
162                ArrayArg::from_raw_parts::<E>(&output.handle, num_elements, vectorization_factor),
163            )
164        };
165
166        output
167    }
168}
