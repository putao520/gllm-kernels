5/// Returns the offset of the tensor corresponding to the layout tensor.
6#[cube]
7pub fn index_offset_with_layout<N: CubePrimitive, L: CubePrimitive>(
8    tensor: &Tensor<Line<N>>,
9    layout: &Tensor<Line<L>>,
10    offset_layout: u32,
11    dim_start: u32,
12    dim_end: u32,
13    #[comptime] unroll: bool,
14) -> u32 {
15    let offset_ref = offset_layout * tensor.line_size();
16    let mut offset = 0;
17
18    #[unroll(unroll)]
19    for i in dim_start..dim_end {
20        let ogwl = offset_ref / layout.stride(i);
21        offset += ogwl % tensor.shape(i) * tensor.stride(i);
22    }
23
24    offset / tensor.line_size()
25}

27#[cube(launch)]
28fn into_contiguous_kernel<N: CubePrimitive>(
29    input: &Tensor<Line<N>>,
30    output: &mut Tensor<Line<N>>,
31    #[comptime] rank: Option<u32>,
32    #[comptime] elems_per_thread: u32,
33) {
34    let offset_output = ABSOLUTE_POS * elems_per_thread;
35    let line_size = input.line_size();
36
37    let mut registers = Array::vectorized(elems_per_thread, line_size);
38
39    #[unroll]
40    for i in 0..elems_per_thread {
41        let offset_input = index_offset_with_layout::<N, N>(
42            input,
43            output,
44            offset_output + i,
45            0,
46            rank.unwrap_or_else(|| output.rank()),
47            rank.is_some(),
48        );
49
50        registers[i] = input[offset_input];
51    }
52
53    #[unroll]
54    for i in 0..elems_per_thread {
55        output[offset_output + i] = registers[i];
56    }
57}

59/// Make a jit tensor contiguous.
60pub fn into_contiguous<R: Runtime, E: CubePrimitive>(
61    client: &ComputeClient<R::Server, R::Channel>,
62    input: &TensorHandleRef<'_, R>,
63) -> TensorHandle<R, E> {
64    let num_elems: usize = input.shape.iter().product();
65    // Vectorization is only enabled when the last dimension is contiguous.
66    let rank = input.strides.len();
67    let vectorization_factor = tensor_line_size_parallel(
68        R::supported_line_sizes().iter().cloned(),
69        input.shape,
70        input.strides,
71        rank - 1,
72    );
73    let num_vecs = num_elems / vectorization_factor as usize;
74    let approx_sm = 64;
75    let approx_simul_vecs = approx_sm * CubeDim::default().num_elems();
76    let elems_per_unit = match num_vecs as u32 / approx_simul_vecs {
77        0..2 => 1,
78        2..4 => 2,
79        4..8 => 4,
80        8.. => 8,
81    };
82
83    // TODO: Benchmark to find good default prefetch, for now preserve existing behaviour
84    into_contiguous_prefetch(client, input, elems_per_unit)
85}

87/// Make a jit tensor contiguous.
88pub fn into_contiguous_prefetch<R: Runtime, E: CubePrimitive>(
89    client: &ComputeClient<R::Server, R::Channel>,
90    input: &TensorHandleRef<'_, R>,
91    elems_per_unit: u32,
92) -> TensorHandle<R, E> {
93    // Vectorization is only enabled when the last dimension is contiguous.
94    let rank = input.strides.len();
95    let vectorization_factor = tensor_line_size_parallel(
96        R::supported_line_sizes().iter().cloned(),
97        input.shape,
98        input.strides,
99        rank - 1,
100    );
101
102    let num_elems_per_unit = vectorization_factor as u32 * elems_per_unit;
103
104    let num_elems: usize = input.shape.iter().product();
105    let cube_dim = CubeDim::default();
106    let cube_count =
107        calculate_cube_count_elemwise(num_elems.div_ceil(num_elems_per_unit as usize), cube_dim);
108    let handle = client.empty(num_elems * E::size().expect("To be a native type"));
109    let output = TensorHandle::new_contiguous(input.shape.to_vec(), handle);
110
111    into_contiguous_kernel::launch::<Line<E>, R>(
112        client,
113        cube_count,
114        cube_dim,
115        input.as_tensor_arg(vectorization_factor),
116        output.as_ref().as_tensor_arg(vectorization_factor),
117        Some(rank as u32),
118        elems_per_unit,
119    );
120
121    output
122}
