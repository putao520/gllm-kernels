9/// Constructs a special weight tensor used for unfolding.
10///
11/// # Notes
12///
13/// The idea behind using convolution for unfolding is to leverage the sliding window mechanism of
14/// convolution. By creating a weight tensor with ones in a particular pattern, we are able to borrow
15/// the convolution operation's mechanism as it moves across the input tensor, picking up the desired
16/// values in the pattern of the unfolding operation.
17pub(crate) fn create_unfolding_weight<B: Backend>(
18    in_channels: usize,
19    kernel_size: [usize; 2],
20    device: &B::Device,
21) -> FloatTensor<B> {
22    let shape = Shape::new([
23        in_channels * kernel_size[0] * kernel_size[1],
24        in_channels,
25        kernel_size[0],
26        kernel_size[1],
27    ]);
28
29    let mut strides = [0; 4];
30    let mut current = 1;
31    shape
32        .dims
33        .iter()
34        .enumerate()
35        .rev()
36        .for_each(|(index, val)| {
37            strides[index] = current;
38            current *= val;
39        });
40
41    let num_elements = shape.num_elements();
42
43    let mut weight: Vec<B::FloatElem> = vec![0.0.elem(); num_elements];
44
45    for k in 0..in_channels {
46        for i in 0..kernel_size[0] {
47            for j in 0..kernel_size[1] {
48                let output_channel = k * kernel_size[0] * kernel_size[1] + i * kernel_size[1] + j;
49                let index =
50                    output_channel * strides[0] + k * strides[1] + i * strides[2] + j * strides[3];
51
52                weight[index] = 1.elem();
53            }
54        }
55    }
56
57    B::float_from_data(TensorData::new(weight, shape), device)
58}

60/// Compute the unfold4d operation using the conv2d operations.
61pub(crate) fn unfold4d_using_conv2d<B: Backend>(
62    x: FloatTensor<B>,
63    kernel_size: [usize; 2],
64    options: UnfoldOptions,
65) -> FloatTensor<B> {
66    let [_batch_size, in_channels, _in_height, _in_width] = x.shape().dims();
67    let weight = create_unfolding_weight::<B>(in_channels, kernel_size, &B::float_device(&x));
68    let unfolded = B::conv2d(
69        x,
70        weight,
71        None,
72        ConvOptions::new(options.stride, options.padding, options.dilation, 1),
73    );
74
75    let [batch_size, channels_out, out_height, out_width] = unfolded.shape().dims();
76
77    B::float_reshape(
78        unfolded,
79        Shape::new([batch_size, channels_out, out_height * out_width]),
80    )
