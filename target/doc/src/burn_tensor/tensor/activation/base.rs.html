5/// Applies the rectified linear unit function as described in the paper [Deep Learning using
6/// Rectified Linear Units (ReLU)](https://arxiv.org/pdf/1803.08375).
7///
8/// `y = max(0, x)`
9pub fn relu<const D: usize, B: Backend>(tensor: Tensor<B, D>) -> Tensor<B, D> {
10    tensor.relu()
11}

13/// Applies the leaky rectified linear unit function.
14///
15/// f(x) = negative_slope * x for x < 0, f(x) = x for x >= 0
16pub fn leaky_relu<const D: usize, B: Backend>(
17    tensor: Tensor<B, D>,
18    negative_slope: f64,
19) -> Tensor<B, D> {
20    Tensor::from_primitive(TensorPrimitive::Float(B::leaky_relu(
21        tensor.primitive.tensor(),
22        crate::ElementConversion::elem(negative_slope),
23    )))
24}

31/// Applies Parametric ReLu activation function as described in the paper [Delving Deep into Rectifiers: Surpassing Human-Level Performance on ImageNet Classification](https://arxiv.org/pdf/1502.01852).
32/// ` PReLu(x) = max(0,x) + \alpha * min(0,x)`
33/// tensor is assumed to be of shape \[batch_size, channels, ...\]
34/// alpha is assumed to be of shape \[channels\] or \[1\]
35pub fn prelu<const D: usize, B: Backend>(
36    tensor: Tensor<B, D>,
37    alpha: Tensor<B, 1>,
38) -> Tensor<B, D> {
39    check!(TensorCheck::check_prelu_shape::<D>(
40        &tensor.shape(),
41        &alpha.shape()
42    ));
43
44    let weight = if alpha.dims()[0] == 1 {
45        // if there is only 1 weight, then reshape it to (1,1,1... D times) so that the rank is D
46        alpha.reshape([1; D])
47    } else {
48        // D>=2 because the case where D==1 and num_weights >1 is handled by check function
49        // there is more than 1 weight and rank is more than 2
50        let num_weights = alpha.dims()[0];
51        let mut s = [1; D];
52        s[1] = num_weights;
53        // reshape the weights to (1, channels,1 ...)
54        alpha.reshape(s)
55    };
56
57    Tensor::from_primitive(TensorPrimitive::Float(B::prelu(
58        tensor.primitive.tensor(),
59        weight.primitive.tensor(),
60    )))
61}

63/// Applies the softmax function on the input tensor along the given dimension.
64///
65/// `softmax(x_i) = exp(x_i) / sum_j(exp(x_j))`
66///
67/// # Notes
68///
69/// The dimension argument `dim` specifies the dimension along which the function will be computed.
70/// It must in the range of `0` and `D-1`.
71pub fn softmax<const D: usize, B: Backend>(tensor: Tensor<B, D>, dim: usize) -> Tensor<B, D> {
72    check!(TensorCheck::dim_ops::<D>("softmax", dim));
73
74    let tensor = tensor.clone() - tensor.detach().max_dim(dim);
75    let tensor = tensor.exp();
76    let tensor_tmp = tensor.clone().sum_dim(dim);
77
78    tensor.div(tensor_tmp)
79}

81/// Applies the softmin function on the input tensor along the given dimension.
82///
83/// `softmin(x_i) = exp(-x_i) / sum_j(exp(-x_j))`
84///
85/// # Notes
86///
87/// The dimension argument `dim` specifies the dimension along which the function will be computed.
88/// It must in the range of `0` and `D-1`.
89pub fn softmin<const D: usize, B: Backend>(tensor: Tensor<B, D>, dim: usize) -> Tensor<B, D> {
90    check!(TensorCheck::dim_ops::<D>("softmin", dim));
91    softmax(tensor.neg(), dim)
92}

94/// Applies the softplus function
95///
96/// `softplus(x_i) = log(1 + exp(\beta x_i)) / \beta`
97pub fn softplus<const D: usize, B: Backend>(tensor: Tensor<B, D>, beta: f64) -> Tensor<B, D> {
98    let tensor = (tensor.mul_scalar(beta).exp() + 1).log();
99    tensor.div_scalar(beta)
100}

102/// Applies the "quiet softmax" function on the input tensor along the given dimension.
103/// This function is similar to the softmax function, but it allows for "no selection", e.g.,
104/// all outputs can tend to zero.
105///
106/// `softmax(x_i) = exp(x_i) / [ 1 + sum_j(exp(x_j)) ]`
107///
108/// # Notes
109///
110/// The dimension argument `dim` specifies the dimension along which the function will be computed.
111/// It must in the range of `0` and `D-1`.
112pub fn quiet_softmax<const D: usize, B: Backend>(tensor: Tensor<B, D>, dim: usize) -> Tensor<B, D> {
113    check!(TensorCheck::dim_ops::<D>("softmax", dim));
114
115    let tensor = tensor.clone() - tensor.detach().max_dim(dim);
116    let tensor = tensor.exp();
117    let tensor_tmp = tensor.clone().sum_dim(dim);
118
119    tensor.div(tensor_tmp + 1)
120}

122/// Applies the log softmax function on the input tensor along the given dimension.
123///
124/// `log_softmax(x_i) = log(softmax(x_i)) = log(exp(x_i) / sum_j(exp(x_j)))`
125///
126/// # Notes
127///
128/// The dimension argument `dim` specifies the dimension along which the function will be computed.
129/// It must in the range of `0` and `D-1`.
130pub fn log_softmax<const D: usize, B: Backend>(tensor: Tensor<B, D>, dim: usize) -> Tensor<B, D> {
131    check!(TensorCheck::dim_ops::<D>("log softmax", dim));
132
133    let tensor = tensor.clone() - tensor.detach().max_dim(dim);
134    let tensor_tmp = tensor.clone().exp().sum_dim(dim).log();
135
136    tensor.sub(tensor_tmp)
137}

171/// Applies the Mish function as described in the paper in [Mish: A Self Regularized Non-Monotonic Neural Activation Function](https://arxiv.org/abs/1908.08681).
172///
173/// `mish(x_i) = x_i \times tanh(softplus(x_i))`
174pub fn mish<const D: usize, B: Backend>(tensor: Tensor<B, D>) -> Tensor<B, D> {
175    tensor.clone().mul(softplus(tensor, 1.0).tanh())
176}
