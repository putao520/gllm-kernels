11impl KernelFn {
12    pub fn to_tokens_mut(&mut self) -> TokenStream {
13        let prelude_path = prelude_path();
14        let debug_source = frontend_type("debug_source_expand");
15
16        let vis = &self.vis;
17        let sig = &self.sig;
18        let body = match &self.body {
19            KernelBody::Block(block) => &block.to_tokens(&mut self.context),
20            KernelBody::Verbatim(tokens) => tokens,
21        };
22        let name = &self.full_name;
23        let debug_source = quote_spanned! {self.span=>
24            #debug_source(context, #name, file!(), line!(), column!())
25        };
26
27        let out = quote! {
28            #vis #sig {
29                #debug_source;
30                use #prelude_path::IntoRuntime as _;
31
32                #body
33            }
34        };
35
36        out
37    }
38}

40impl ToTokens for KernelSignature {
41    fn to_tokens(&self, tokens: &mut TokenStream) {
42        let cube_context = prelude_type("CubeContext");
43        let cube_type = prelude_type("CubeType");
44
45        let name = &self.name;
46        let generics = &self.generics;
47        let return_type = match &self.returns {
48            KernelReturns::ExpandType(ty) => quote![<#ty as #cube_type>::ExpandType],
49            KernelReturns::Plain(ty) => quote![#ty],
50        };
51        let out = if self
52            .parameters
53            .first()
54            .filter(|param| param.name == "self")
55            .is_some()
56        {
57            let args = self.parameters.iter().skip(1);
58
59            quote! {
60                fn #name #generics(
61                    self, // Always owned during expand.
62                    context: &mut #cube_context,
63                    #(#args),*
64                ) -> #return_type
65            }
66        } else {
67            let args = &self.parameters;
68            quote! {
69                fn #name #generics(
70                    context: &mut #cube_context,
71                    #(#args),*
72                ) -> #return_type
73            }
74        };
75
76        tokens.extend(out);
77    }
78}

88impl Launch {
89    fn kernel_phantom_data(&self) -> Option<TokenStream> {
90        let generics = self.kernel_generics.clone();
91        let declared_lifetimes = generics.declared_lifetimes();
92        let declared_type_params = generics.declared_type_params();
93
94        let used_lifetimes = self
95            .comptime_params()
96            .map(|param| &param.ty)
97            .collect_lifetimes_cloned(&Purpose::Declare.into(), &declared_lifetimes);
98        let used_type_params = self
99            .comptime_params()
100            .map(|param| &param.ty)
101            .collect_type_params_cloned(&Purpose::Declare.into(), &declared_type_params);
102        let lifetimes: Vec<_> = declared_lifetimes.difference(&used_lifetimes).collect();
103        let type_params: Vec<_> = declared_type_params.difference(&used_type_params).collect();
104
105        (!lifetimes.is_empty() || !type_params.is_empty())
106            .then(|| quote![__ty: ::core::marker::PhantomData<(#(#lifetimes,)* #(#type_params),*)>])
107    }

109    pub fn compilation_args_def(&self) -> (Vec<TokenStream>, Vec<Ident>) {
110        let mut tokens = Vec::new();
111        let mut args = Vec::new();
112        let launch_arg_expand = prelude_type("LaunchArgExpand");
113
114        self.runtime_inputs().for_each(|input| {
115            let ty = &input.ty_owned();
116            let name = &input.name;
117
118            tokens.push(quote! {
119                #name: <#ty as #launch_arg_expand>::CompilationArg
120            });
121            args.push(name.clone());
122        });
123
124        self.runtime_outputs().for_each(|output| {
125            let ty = &output.ty_owned();
126            let name = &output.name;
127
128            tokens.push(quote! {
129                #name: <#ty as #launch_arg_expand>::CompilationArg
130            });
131            args.push(name.clone());
132        });
133
134        (tokens, args)
135    }

137    pub fn compilation_args(&self) -> (TokenStream, TokenStream) {
138        let launch_arg = prelude_type("LaunchArg");
139        let mut defined = quote! {};
140        let mut args = quote! {};
141
142        self.runtime_inputs().enumerate().for_each(|(i, input)| {
143            let ty = &input.ty_owned();
144            let ident = &input.name;
145            let var = Ident::new(format!("input_arg_{i}").as_str(), ident.span());
146
147            args.extend(quote! {#var,});
148            defined.extend(quote! {
149                let #var = <#ty as #launch_arg>::compilation_arg::<__R>(&#ident);
150            });
151        });
152        self.runtime_outputs().enumerate().for_each(|(i, output)| {
153            let ty = &output.ty_owned();
154            let ident = &output.name;
155            let var = Ident::new(format!("output_arg_{i}").as_str(), ident.span());
156
157            args.extend(quote! {#var,});
158            defined.extend(quote! {
159                let #var = <#ty as #launch_arg>::compilation_arg::<__R>(&#ident);
160            });
161        });
162
163        (
164            quote! {
165                #defined
166            },
167            args,
168        )
169    }

171    pub fn io_mappings(&self) -> TokenStream {
172        let launch_arg_expand = prelude_type("LaunchArgExpand");
173        let mut define = quote! {};
174
175        let expand_fn = |ident, expand_name, ty| {
176            let ty = self.analysis.process_ty(&ty);
177
178            quote! {
179                let #ident =  <#ty as #launch_arg_expand>::#expand_name(&self.#ident.dynamic_cast(), &mut builder);
180            }
181        };
182        for input in self.runtime_inputs() {
183            define.extend(expand_fn(
184                &input.name,
185                format_ident!("expand"),
186                input.ty_owned(),
187            ));
188        }
189
190        for output in self.runtime_outputs() {
191            define.extend(expand_fn(
192                &output.name,
193                format_ident!("expand_output"),
194                output.ty_owned(),
195            ));
196        }
197
198        quote! {
199            #define
200        }
201    }

203    fn define_body(&self) -> TokenStream {
204        let kernel_builder = prelude_type("KernelBuilder");
205        let io_map = self.io_mappings();
206        let register_type = self.analysis.register_elems();
207        let runtime_args = self.runtime_params().map(|it| &it.name);
208        let comptime_args = self.comptime_params().map(|it| &it.name);
209        let generics = self.analysis.process_generics(&self.func.sig.generics);
210
211        quote! {
212            let mut builder = #kernel_builder::default();
213            #register_type
214            #io_map
215            expand #generics(&mut builder.context, #(#runtime_args.clone(),)* #(self.#comptime_args.clone()),*);
216            builder.build(self.settings.clone())
217        }
218    }

220    /// Returns the kernel entrypoint name.
221    /// Appropriate for usage in source code such as naming the CUDA or WGSL entrypoint.
222    ///
223    /// For example a kernel:
224    /// ```text
225    /// #[cube(launch)]
226    /// fn my_kernel(input: &Array<f32>, output: &mut Array<f32>) {}
227    /// ```
228    /// would produce the name `my_kernel`.
229    ///
230    /// If a generic has the `Float` or `Numeric` bound the kernel also has a suffix
231    /// with the name of that type in use:
232    /// ```text
233    /// fn my_kernel<F: Float>(input: &Array<F>, output: &mut Array<F>) {}
234    /// ```
235    /// now produces the name `my_kernel_f16` or `my_kernel_f32` etc. depending
236    /// on which variant of the kernel is launched by the user.
237    ///
238    /// If a kernel has several matching bounds they are appended as suffixes in order.
239    fn kernel_entrypoint_name(&self) -> TokenStream {
240        // This base name is always used; a suffix might be added
241        // based on generics.
242        let base_name = self.func.sig.name.to_string();
243
244        let generics = &self.kernel_generics;
245        let suffix_producing_bounds = [format_ident!("Float"), format_ident!("Numeric")];
246
247        let mut matching_generics = vec![];
248
249        // Inspect all generics for the bounds of interest in order to
250        // determine if a suffix should be added
251        for typ in generics.type_params() {
252            for bound in &typ.bounds {
253                let TypeParamBound::Trait(t) = bound else {
254                    continue;
255                };
256
257                // Using last should account for the bounds such as `Float` but also `some::prefix::Float`
258                let Some(generic_trailing) = t.path.segments.last() else {
259                    continue;
260                };
261
262                // If we find some type parameter with `Float` as a bound,
263                // add a suffix based on a shortened version of the
264                // type name
265                if suffix_producing_bounds.contains(&generic_trailing.ident) {
266                    // E.g. the `F` in `F: Float` or `N` in `N: Numeric`
267                    matching_generics.push(typ.ident.clone());
268                    continue;
269                }
270            }
271        }
272
273        if matching_generics.is_empty() {
274            quote! {
275                #base_name
276            }
277        } else {
278            quote! (
279                {
280                    // Go from type names such as `half::f16` to `f16` etc.
281                    let shorten = |p: &'static str| {
282                        if let Some((_, last)) = p.rsplit_once("::") {
283                            last
284                        } else {
285                            p
286                        }
287                    };
288
289                    let mut name = format!("{}", #base_name);
290
291                    #( {
292                        let type_name = shorten(core::any::type_name::< #matching_generics >());
293                        name.push_str(&format!("_{type_name}"));
294                    })*
295
296                    name
297                }
298            )
299        }
300    }

302    pub fn kernel_definition(&self) -> TokenStream {
303        if self.args.is_launch() {
304            let kernel = core_type("Kernel");
305            let kernel_settings = prelude_type("KernelSettings");
306            let kernel_definition: syn::Path = prelude_type("KernelDefinition");
307            let kernel_id = core_type("KernelId");
308
309            let kernel_name = self.kernel_name();
310            let define = self.define_body();
311            let kernel_doc = format!("{} Kernel", self.func.sig.name);
312
313            let (generics, generic_names, where_clause) = self.kernel_generics.split_for_impl();
314            let const_params: Vec<_> = self.comptime_params().collect();
315            let param_names = self
316                .comptime_params()
317                .map(|param| param.name.clone())
318                .collect::<Vec<_>>();
319            let phantom_data = self.kernel_phantom_data();
320            let phantom_data_init = phantom_data
321                .as_ref()
322                .map(|_| quote![__ty: ::core::marker::PhantomData]);
323            let (compilation_args, args) = self.compilation_args_def();
324            let info = param_names.clone().into_iter().chain(args.clone());
325
326            let kernel_source_name = self.kernel_entrypoint_name();
327
328            quote! {
329                #[doc = #kernel_doc]
330                pub struct #kernel_name #generics #where_clause {
331                    settings: #kernel_settings,
332                    #(#compilation_args,)*
333                    #(#const_params,)*
334                    #phantom_data
335                }
336
337                #[allow(clippy::too_many_arguments)]
338                impl #generics #kernel_name #generic_names #where_clause {
339                    pub fn new(settings: #kernel_settings, #(#compilation_args,)* #(#const_params),*) -> Self {
340                        Self {
341                            settings: settings.kernel_name(#kernel_source_name),
342                            #(#args,)*
343                            #(#param_names,)*
344                            #phantom_data_init
345                        }
346                    }
347                }
348
349                impl #generics #kernel for #kernel_name #generic_names #where_clause {
350                    fn define(&self) -> #kernel_definition {
351                        #define
352                    }
353
354                    fn id(&self) -> #kernel_id {
355                        // We don't use any other kernel settings with the macro.
356                        let cube_dim = self.settings.cube_dim.clone();
357                        #kernel_id::new::<Self>().info((cube_dim, #(self.#info.clone()),* ))
358                    }
359                }
360            }
361        } else {
362            TokenStream::new()
363        }
364    }
