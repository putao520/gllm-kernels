9/// Handle the error returned from the macro logic. This may be either a syntax
10/// error or a logic error. In either case, we want to return a [`TokenStream2`]
11/// that still provides good IDE experience. See [`Fallback`] for details.
12pub(crate) fn error_into_token_stream(err: syn::Error, item: TokenStream) -> TokenStream {
13    let compile_error = err.to_compile_error();
14
15    syn::parse2::<Fallback>(item)
16        .map(|fallback| quote!(#compile_error #fallback))
17        .unwrap_or_else(|_| compile_error)
18}

20/// This is used in error handling for better IDE experience. For example, while
21/// the developer is writing the function code they'll have a bunch of syntax
22/// errors in the process. While that happens the proc macro should output at
23/// least some representation of the input code that the developer wrote with
24/// a separate compile error entry. This keeps the syntax highlighting and IDE
25/// type analysis, completions and other hints features working even if macro
26/// fails to parse some syntax or finds some other logic errors.
27///
28/// This utility does very low-level parsing to strip helper attributes (i.e. `#[comptime]`) from
29/// the input. This is to prevent the IDE from showing errors for helper attributes that need to be
30/// processed by this macro.
31struct Fallback {
32    output: TokenStream,
33}
34
35impl Parse for Fallback {
36    fn parse(input: syn::parse::ParseStream<'_>) -> syn::Result<Self> {
37        let mut output = TokenStream::new();
38
39        loop {
40            let found_attr = input.step(|cursor| {
41                let mut cursor = *cursor;
42                while let Some((tt, next)) = cursor.token_tree() {
43                    match &tt {
44                        TokenTree::Group(group) => {
45                            let fallback: Self = syn::parse2(group.stream())?;
46                            let new_group =
47                                proc_macro2::Group::new(group.delimiter(), fallback.output);
48
49                            output.extend([TokenTree::Group(new_group)]);
50                        }
51                        TokenTree::Punct(punct) if punct.as_char() == '#' => {
52                            return Ok((true, cursor));
53                        }
54                        TokenTree::Punct(_) | TokenTree::Ident(_) | TokenTree::Literal(_) => {
55                            output.extend([tt]);
56                        }
57                    }
58
59                    cursor = next;
60                }
61
62                Ok((false, cursor))
63            })?;
64
65            if !found_attr {
66                return Ok(Self { output });
67            }
68
69            input
70                .call(syn::Attribute::parse_outer)?
71                .into_iter()
72                .filter(|attr| !is_helper(attr))
73                .for_each(|attr| attr.to_tokens(&mut output));
74        }
75    }
76}
77
78impl ToTokens for Fallback {
79    fn to_tokens(&self, tokens: &mut TokenStream) {
80        self.output.to_tokens(tokens);
81    }
82}
