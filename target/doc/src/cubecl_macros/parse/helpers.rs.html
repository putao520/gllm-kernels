10pub struct Unroll {
11    pub value: Expression,
12}
13
14impl Unroll {
15    pub fn from_attributes(
16        attrs: &[Attribute],
17        context: &mut Context,
18    ) -> syn::Result<Option<Self>> {
19        #[derive(FromMeta)]
20        struct NameVal {
21            pub value: Expr,
22        }
23
24        let attr = attrs.iter().find(|attr| attr.path().is_ident("unroll"));
25        let attr = match attr {
26            Some(attr) => attr,
27            None => return Ok(None),
28        };
29
30        let res = match &attr.meta {
31            syn::Meta::Path(_) => Self {
32                value: Expression::from_expr(parse_quote![true], context).unwrap(),
33            },
34            syn::Meta::List(list) => {
35                let expr = syn::parse2(list.tokens.clone())?;
36                let expr = Expression::from_expr(expr, context)?;
37                Self { value: expr }
38            }
39            meta => {
40                let expr = NameVal::from_meta(meta)?;
41                let expr = Expression::from_expr(expr.value, context)?;
42                Self { value: expr }
43            }
44        };
45        Ok(Some(res))
46    }
47
48    pub fn unroll_expr(attrs: &[Attribute]) -> Option<Expr> {
49        #[derive(FromMeta)]
50        struct NameVal {
51            pub value: Expr,
52        }
53
54        let attr = attrs.iter().find(|attr| attr.path().is_ident("unroll"))?;
55
56        match &attr.meta {
57            syn::Meta::Path(_) => None,
58            syn::Meta::List(list) => syn::parse2(list.tokens.clone()).ok(),
59            meta => Some(NameVal::from_meta(meta).ok()?.value),
60        }
61    }
62}

64pub struct RemoveHelpers;
65
66impl VisitMut for RemoveHelpers {
67    fn visit_fn_arg_mut(&mut self, i: &mut syn::FnArg) {
68        match i {
69            syn::FnArg::Receiver(recv) => recv.attrs.retain(|it| !is_comptime_attr(it)),
70            syn::FnArg::Typed(typed) => typed.attrs.retain(|it| !is_comptime_attr(it)),
71        }
72        visit_mut::visit_fn_arg_mut(self, i);
73    }
74
75    fn visit_expr_for_loop_mut(&mut self, i: &mut syn::ExprForLoop) {
76        let unroll = Unroll::unroll_expr(&i.attrs);
77        i.attrs.retain(|attr| !is_unroll_attr(attr));
78        if let Some(unroll) = unroll {
79            i.body
80                .stmts
81                .insert(0, parse_quote![let __unroll = #unroll;])
82        }
83        visit_mut::visit_expr_for_loop_mut(self, i);
84    }
85}

91impl VisitMut for ReplaceIndices {
92    fn visit_expr_assign_mut(&mut self, i: &mut syn::ExprAssign) {
93        ReplaceIndexMut.visit_expr_mut(&mut i.left);
94        ReplaceIndex.visit_expr_mut(&mut i.right);
95        visit_mut::visit_expr_assign_mut(self, i);
96    }
97
98    fn visit_expr_binary_mut(&mut self, i: &mut syn::ExprBinary) {
99        match i.op {
100            syn::BinOp::AddAssign(_)
101            | syn::BinOp::SubAssign(_)
102            | syn::BinOp::MulAssign(_)
103            | syn::BinOp::DivAssign(_)
104            | syn::BinOp::RemAssign(_)
105            | syn::BinOp::BitXorAssign(_)
106            | syn::BinOp::BitAndAssign(_)
107            | syn::BinOp::BitOrAssign(_)
108            | syn::BinOp::ShlAssign(_)
109            | syn::BinOp::ShrAssign(_) => {
110                ReplaceIndexMut.visit_expr_mut(&mut i.left);
111                ReplaceIndex.visit_expr_mut(&mut i.right);
112            }
113            _ => {}
114        }
115        visit_mut::visit_expr_binary_mut(self, i);
116    }
117
118    fn visit_expr_mut(&mut self, i: &mut syn::Expr) {
119        match i {
120            Expr::Reference(ExprReference {
121                mutability: Some(_),
122                expr,
123                ..
124            }) => {
125                ReplaceIndexMut.visit_expr_mut(expr);
126            }
127            Expr::Index(_) => ReplaceIndex.visit_expr_mut(i),
128            _ => {}
129        }
130        visit_mut::visit_expr_mut(self, i);
131    }
132
133    fn visit_item_fn_mut(&mut self, i: &mut syn::ItemFn) {
134        let prelude_path = prelude_path();
135        let import = parse_quote![use #prelude_path::{CubeIndex as _, CubeIndexMut as _};];
136        i.block.stmts.insert(0, import);
137        visit_mut::visit_item_fn_mut(self, i);
138    }
139
140    fn visit_impl_item_fn_mut(&mut self, i: &mut syn::ImplItemFn) {
141        let prelude_path = prelude_path();
142        let import = parse_quote![use #prelude_path::{CubeIndex as _, CubeIndexMut as _};];
143        i.block.stmts.insert(0, import);
144        visit_mut::visit_impl_item_fn_mut(self, i);
145    }
146
147    fn visit_trait_item_fn_mut(&mut self, i: &mut syn::TraitItemFn) {
148        if let Some(block) = &mut i.default {
149            let prelude_path = prelude_path();
150            let import = parse_quote![use #prelude_path::{CubeIndex as _, CubeIndexMut as _};];
151            block.stmts.insert(0, import);
152        }
153        visit_mut::visit_trait_item_fn_mut(self, i);
154    }
155}

157impl VisitMut for ReplaceIndex {
158    fn visit_expr_mut(&mut self, i: &mut Expr) {
159        match i {
160            Expr::Reference(ExprReference {
161                mutability: Some(_),
162                expr,
163                ..
164            }) => {
165                ReplaceIndexMut.visit_expr_mut(expr);
166            }
167            Expr::Index(index) => {
168                let inner = &index.expr;
169                let index = &index.index;
170                *i = parse_quote![*#inner.cube_idx(#index)]
171            }
172            _ => {}
173        }
174        visit_mut::visit_expr_mut(self, i);
175    }
176}

178impl VisitMut for ReplaceIndexMut {
179    fn visit_expr_mut(&mut self, i: &mut syn::Expr) {
180        if let Expr::Index(index) = i {
181            let inner = &index.expr;
182            let index = &index.index;
183            *i = parse_quote![*#inner.cube_idx_mut(#index)]
184        }
185        visit_mut::visit_expr_mut(self, i);
186    }

189pub fn is_comptime_attr(attr: &Attribute) -> bool {
