29impl CubeImplItem {
30    pub fn from_impl_item(struct_ty_name: &Type, item: ImplItem) -> syn::Result<Vec<Self>> {
31        let res = match item {
32            ImplItem::Fn(func) => {
33                let name = func.sig.ident.clone();
34                let full_name = quote!(#struct_ty_name::#name).to_string();
35                let mut func =
36                    KernelFn::from_sig_and_block(func.vis, func.sig, func.block, full_name)?;
37                let func_name_expand = format_ident!("__expand_{}", func.sig.name);
38
39                let is_method = func
40                    .sig
41                    .parameters
42                    .first()
43                    .map(|param| param.name == "self")
44                    .unwrap_or(false);
45
46                if is_method {
47                    let method = Self::handle_method_expand(func_name_expand, &mut func);
48                    let func_expand = Self::create_func_expand(struct_ty_name, &func, true);
49
50                    vec![
51                        CubeImplItem::Fn(func),
52                        CubeImplItem::MethodExpand(method),
53                        CubeImplItem::FnExpand(func_expand),
54                    ]
55                } else {
56                    func.sig.name = func_name_expand;
57
58                    let func_expand = Self::create_func_expand(struct_ty_name, &func, false);
59                    vec![CubeImplItem::Fn(func), CubeImplItem::FnExpand(func_expand)]
60                }
61            }
62            _ => vec![CubeImplItem::Other],
63        };
64
65        Ok(res)
66    }

89    /// Create the method from the function and update the function body to point to the method's
90    /// implementation.
91    fn handle_method_expand(func_name_expand: Ident, func: &mut KernelFn) -> KernelFn {
92        let mut method_sig = func.sig.clone();
93
94        method_sig.name = format_ident!("__expand_{}_method", func.sig.name);
95        method_sig.plain_returns_self();
96
97        // Since the function is associated to the expand type, we have to update the normalized
98        // types for the arguments.
99        for param in method_sig
100            .parameters
101            .iter_mut()
102            // We skip the self param.
103            .skip(1)
104        {
105            param.plain_normalized_self();
106        }
107
108        func.sig.name = func_name_expand;
109
110        let param = func.sig.parameters.first_mut().expect("Should be a method");
111        param.name = Ident::new("this", param.span());
112
113        let args = func.sig.parameters.iter().skip(1).map(|param| &param.name);
114        let method_name = &method_sig.name;
115        let (_, generics, _) = &method_sig.generics.split_for_impl();
116        let generics = generics.as_turbofish();
117
118        let mut body = KernelBody::Verbatim(quote! {
119            this.#method_name #generics(
120                context,
121                #(#args),*
122            )
123        });
124
125        // The function points to the method's body.
126        core::mem::swap(&mut func.body, &mut body);
127
128        KernelFn {
129            vis: func.vis.clone(),
130            sig: method_sig,
131            body,
132            full_name: func.full_name.clone(),
133            span: func.span,
134            context: Context::new(func.context.return_type.clone()),
135        }
136    }
137
138    /// Create the same function but that should be generated for the expand type.
139    ///
140    /// This is important since it allows to use the Self keyword inside methods.
141    fn create_func_expand(struct_ty_name: &Type, func: &KernelFn, is_method: bool) -> KernelFn {
142        let mut func_sig = func.sig.clone();
143
144        // Since the function is associated to the expand type, we have to update the normalized
145        // types for the arguments.
146        for param in func_sig
147            .parameters
148            .iter_mut()
149            // We skip the self param.
150            .skip(if is_method { 1 } else { 0 })
151        {
152            param.plain_normalized_self();
153        }
154
155        if let Some(param) = func_sig.parameters.first_mut() {
156            if is_method {
157                let ty = match &param.ty {
158                    Type::Reference(reference) => reference.elem.as_ref().clone(),
159                    ty => ty.clone(),
160                };
161                param.name = Ident::new("this", param.span());
162                param.normalized_ty = ty;
163            }
164        }
165        func_sig.plain_returns_self();
166
167        let args = func_sig.parameters.iter().map(|param| &param.name);
168        let struct_name = format_type_with_turbofish(struct_ty_name);
169        let fn_name = &func_sig.name;
170        let (_, generics, _) = &func_sig.generics.split_for_impl();
171        let generics = generics.as_turbofish();
172
173        let body = quote! {
174            #struct_name::#fn_name #generics(
175                context,
176                #(#args),*
177            )
178        };
179
180        KernelFn {
181            vis: func.vis.clone(),
182            sig: func_sig,
183            body: KernelBody::Verbatim(body),
184            full_name: func.full_name.clone(),
185            span: func.span,
186            context: Context::new(func.context.return_type.clone()),
187        }
188    }
189}

191impl CubeImpl {
192    pub fn from_item_impl(mut item_impl: ItemImpl) -> syn::Result<Self> {
193        let struct_name = *item_impl.self_ty.clone();
194
195        let items = item_impl
196            .items
197            .iter()
198            .cloned()
199            .map(|item| CubeImplItem::from_impl_item(&struct_name, item))
200            .flat_map(|items| {
201                let result: Vec<syn::Result<CubeImplItem>> = match items {
202                    Ok(items) => items.into_iter().map(Ok).collect(),
203                    Err(err) => vec![Err(err)],
204                };
205                result
206            })
207            .collect::<Result<_, _>>()?;
208
209        RemoveHelpers.visit_item_impl_mut(&mut item_impl);
210        ReplaceIndices.visit_item_impl_mut(&mut item_impl);
211
212        let mut attrs = item_impl.attrs;
213        attrs.retain(|attr| !attr.path().is_ident("cube"));
214
215        let unsafety = item_impl.unsafety;
216        let generics = item_impl.generics;
217
218        Ok(Self {
219            unsafety,
220            struct_name,
221            generics,
222            items,
223            original_items: item_impl.items,
224        })
225    }
226}

228/// When we use a type with generics for calling a function, we have to add more `::` between the type ident and
229/// the generic arguments.
230fn format_type_with_turbofish(ty: &Type) -> proc_macro2::TokenStream {
231    match ty {
232        Type::Path(TypePath { path, .. }) => {
233            let segments = &path.segments;
234            let last_segment = segments.last().unwrap();
235            let ident = &last_segment.ident;
236
237            match &last_segment.arguments {
238                PathArguments::AngleBracketed(args) => {
239                    let generic_args = args.args.iter().map(|arg| match arg {
240                        GenericArgument::Type(t) => t.to_token_stream(),
241                        _ => quote! { #arg },
242                    });
243
244                    quote! { #ident::<#(#generic_args),*> }
245                }
246                _ => quote! { #ident },
247            }
248        }
249        _ => ty.to_token_stream(),
250    }
