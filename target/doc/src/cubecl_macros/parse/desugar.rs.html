12pub struct Desugar;
13impl VisitMut for Desugar {
14    fn visit_expr_mut(&mut self, i: &mut syn::Expr) {
15        if let Expr::While(inner) = i {
16            *i = Expr::Loop(desugar_while(inner))
17        }
18        visit_mut::visit_expr_mut(self, i);
19    }
20
21    fn visit_block_mut(&mut self, i: &mut syn::Block) {
22        let stmts = desugar_pats(take(&mut i.stmts));
23
24        i.stmts = stmts;
25        visit_mut::visit_block_mut(self, i)
26    }
27}

29fn desugar_pats(stmts: Vec<Stmt>) -> Vec<Stmt> {
30    stmts.into_iter().flat_map(|stmt| {
31        match stmt {
32            Stmt::Local(Local {
33                pat: Pat::Struct(pat),
34                init: Some(init),
35                ..
36            }) => desugar_struct_destructure(pat, init),
37            Stmt::Local(Local {
38                pat:
39                    Pat::Tuple(PatTuple { elems, .. }) | Pat::TupleStruct(PatTupleStruct { elems, .. }),
40                init: Some(init),
41                ..
42            }) => desugar_tuple_destructure(elems, init),
43            stmt => vec![stmt],
44        }
45    }).collect()
46}
47
48fn desugar_while(inner: &ExprWhile) -> ExprLoop {
49    let cond = &inner.cond;
50    let attrs = &inner.attrs;
51    let label = &inner.label;
52    let body = &inner.body;
53    parse_quote! {
54        #(#attrs)*
55        #label loop {
56            if !(#cond) {
57                break;
58            }
59            #body
60        }
61    }
62}

64fn desugar_struct_destructure(pat: PatStruct, init: LocalInit) -> Vec<Stmt> {
65    let fields = pat.fields.into_iter().map(|field| {
66        let attrs = field.attrs;
67        let pat = field.pat;
68        let member = field.member;
69        quote_spanned! {pat.span()=>
70            #(#attrs)* let #pat = __struct_destructure_init.#member;
71        }
72    });
73    let init = init.expr;
74    let init = quote_spanned![init.span()=> let __struct_destructure_init = #init;];
75    parse_quote! {
76        #init
77        #(#fields)*
78    }
79}

81fn desugar_tuple_destructure(fields: impl IntoIterator<Item = Pat>, init: LocalInit) -> Vec<Stmt> {
82    let fields = fields.into_iter().enumerate().map(|(i, pat)| {
83        let member = Index::from(i);
84        quote_spanned! {pat.span()=>
85            let #pat = __tuple_destructure_init.#member;
86        }
87    });
88    let init = init.expr;
89    let init = quote_spanned![init.span()=> let __tuple_destructure_init = #init;];
90    parse_quote! {
91        #init
92        #(#fields)*
93    }
94}
