20    pub fn from_expr(expr: Expr, context: &mut Context) -> syn::Result<Self> {
21        let result = match expr.clone() {
22            Expr::Assign(assign) => {
23                let right = Self::from_expr(*assign.right, context)?;
24                Expression::Assignment {
25                    ty: right.ty(),
26                    left: Box::new(Self::from_expr(*assign.left, context)?),
27                    right: Box::new(right),
28                }
29            }
30            Expr::Binary(binary) => {
31                let span = binary.span();
32                let left = Self::from_expr(*binary.left, context)?;
33                let right = Self::from_expr(*binary.right, context)?;
34                if left.is_const() && right.is_const() {
35                    Expression::Verbatim {
36                        tokens: quote![#expr],
37                    }
38                } else {
39                    let ty = left.ty().or(right.ty());
40                    Expression::Binary {
41                        left: Box::new(left),
42                        operator: parse_binop(&binary.op)?,
43                        right: Box::new(right),
44                        ty,
45                        span,
46                    }
47                }
48            }
49            Expr::Lit(literal) => {
50                let ty = lit_ty(&literal.lit)?;
51                Expression::Literal {
52                    value: literal.lit,
53                    ty,
54                }
55            }
56            Expr::Unary(ExprUnary {
57                op: UnOp::Neg(_),
58                expr,
59                ..
60            }) if matches!(*expr, Expr::Lit(_)) => Expression::Verbatim {
61                tokens: quote![-{#expr}],
62            },
63            Expr::Path(path) => {
64                let name = path.path.get_ident();
65                if let Some(var) = name.and_then(|ident| context.variable(ident)) {
66                    if var.is_keyword {
67                        Expression::Keyword { name: var.name }
68                    } else {
69                        Expression::Variable(var)
70                    }
71                } else {
72                    // If it's not in the scope, it's not a managed local variable. Treat it as an
73                    // external value like a Rust `const`.
74                    Expression::Path { path: path.path }
75                }
76            }
77            Expr::Unary(unary) => {
78                let span = unary.span();
79                let input = Self::from_expr(*unary.expr, context)?;
80                let ty = input.ty();
81                Expression::Unary {
82                    input: Box::new(input),
83                    operator: parse_unop(&unary.op)?,
84                    ty,
85                    span,
86                }
87            }
88            Expr::Block(block) => {
89                let (block, _) = context.in_scope(|ctx| Block::from_block(block.block, ctx))?;
90                Expression::Block(block)
91            }
92            Expr::Break(_) => Expression::Break,
93            Expr::Call(call) => {
94                let span = call.span();
95                let func = Box::new(Expression::from_expr(*call.func, context)?);
96                let args = call
97                    .args
98                    .into_iter()
99                    .map(|arg| Expression::from_expr(arg, context))
100                    .collect::<Result<Vec<_>, _>>()?;
101                match *func {
102                    Expression::Path { path } if is_intrinsic(&path) => {
103                        Expression::CompilerIntrinsic { func: path, args }
104                    }
105                    func => {
106                        let associated_type = fn_associated_type(&func);
107                        Expression::FunctionCall {
108                            func: Box::new(func),
109                            args,
110                            associated_type,
111                            span,
112                        }
113                    }
114                }
115            }
116            Expr::MethodCall(method) => {
117                let span = method.span();
118                let receiver = Expression::from_expr(*method.receiver.clone(), context)?;
119                let args = method
120                    .args
121                    .iter()
122                    .map(|arg| Expression::from_expr(arg.clone(), context))
123                    .collect::<Result<Vec<_>, _>>()?;
124                if receiver.is_const()
125                    && args.iter().all(|arg| arg.is_const())
126                    && method.method != "runtime"
127                {
128                    Expression::Verbatim {
129                        tokens: quote![#method],
130                    }
131                } else {
132                    Expression::MethodCall {
133                        receiver: Box::new(receiver),
134                        method: method.method,
135                        generics: method.turbofish,
136                        args,
137                        span,
138                    }
139                }
140            }
141            Expr::Cast(cast) => {
142                let mut from_expr = *cast.expr;
143                // Flatten multicasts because they shouldn't exist on the GPU
144                while matches!(from_expr, Expr::Cast(_)) {
145                    match from_expr {
146                        Expr::Cast(cast) => from_expr = *cast.expr,
147                        _ => unreachable!(),
148                    }
149                }
150                let from = Expression::from_expr(from_expr, context)?;
151                if let Some(as_const) = from.as_const(context) {
152                    Expression::Verbatim { tokens: as_const }
153                } else {
154                    Expression::Cast {
155                        from: Box::new(from),
156                        to: *cast.ty,
157                    }
158                }
159            }
160            Expr::Const(block) => Expression::Verbatim {
161                tokens: quote![#block],
162            },
163            Expr::Continue(cont) => Expression::Continue(cont.span()),
164            Expr::ForLoop(for_loop) => expand_for_loop(for_loop, context)?,
165            Expr::Loop(loop_expr) => expand_loop(loop_expr, context)?,
166            Expr::If(if_expr) => expand_if(if_expr, context)?,
167            Expr::Range(range) => {
168                let span = range.span();
169                let start = range
170                    .start
171                    .map(|start| Expression::from_expr(*start, context))
172                    .transpose()?
173                    .unwrap_or_else(|| {
174                        let lit = Lit::Int(LitInt::new("0", span));
175                        Expression::Literal {
176                            value: lit,
177                            ty: parse_quote![i32],
178                        }
179                    });
180                let end = range
181                    .end
182                    .map(|end| Expression::from_expr(*end, context))
183                    .transpose()?
184                    .map(Box::new);
185                Expression::Range {
186                    start: Box::new(start),
187                    end,
188                    span,
189                    inclusive: matches!(range.limits, RangeLimits::Closed(..)),
190                }
191            }
192            Expr::Field(field) => {
193                let base = Expression::from_expr(*field.base.clone(), context)?;
194                Expression::FieldAccess {
195                    base: Box::new(base),
196                    field: field.member,
197                }
198            }
199            Expr::Group(group) => Expression::from_expr(*group.expr, context)?,
200            Expr::Paren(paren) => Expression::from_expr(*paren.expr, context)?,
201            Expr::Return(ret) => {
202                let span = ret.expr.span();
203                Expression::Return {
204                    expr: ret
205                        .expr
206                        .map(|expr| Expression::from_expr(*expr, context))
207                        .transpose()?
208                        .map(Box::new),
209                    span,
210                    _ty: context.return_type.clone(),
211                }
212            }
213            Expr::Array(array) => {
214                let span = array.span();
215                let elements = array
216                    .elems
217                    .into_iter()
218                    .map(|elem| Expression::from_expr(elem, context))
219                    .collect::<Result<_, _>>()?;
220                Expression::Array { elements, span }
221            }
222            Expr::Tuple(tuple) => {
223                let elements = tuple
224                    .elems
225                    .into_iter()
226                    .map(|elem| Expression::from_expr(elem, context))
227                    .collect::<Result<_, _>>()?;
228                Expression::Tuple { elements }
229            }
230            Expr::Index(index) => {
231                let span = index.span();
232                let expr = Expression::from_expr(*index.expr, context)?;
233                let index = Expression::from_expr(*index.index, context)?;
234                if is_slice(&index) {
235                    let ranges = match index {
236                        Expression::Array { elements, .. } => elements.clone(),
237                        Expression::Tuple { elements, .. } => elements.clone(),
238                        index => vec![index],
239                    };
240                    Expression::Slice {
241                        expr: Box::new(expr),
242                        span,
243                        _ranges: ranges,
244                    }
245                } else {
246                    let index = match index {
247                        Expression::Array { elements, span } => {
248                            generate_strided_index(&expr, elements, span)?
249                        }
250                        index => index,
251                    };
252                    Expression::Index {
253                        expr: Box::new(expr),
254                        index: Box::new(index),
255                        span,
256                    }
257                }
258            }
259            Expr::Repeat(repeat) => {
260                let span = repeat.span();
261                let len = Expression::from_expr(*repeat.len, context)?;
262                if !len.is_const() {
263                    Err(syn::Error::new(
264                        span,
265                        "Array initializer length must be known at compile time",
266                    ))?
267                }
268                Expression::ArrayInit {
269                    init: Box::new(Expression::from_expr(*repeat.expr, context)?),
270                    len: Box::new(len),
271                }
272            }
273            Expr::Let(expr) => {
274                let span = expr.span();
275                let elem = Expression::from_expr(*expr.expr.clone(), context)?;
276                if elem.is_const() {
277                    Expression::Verbatim {
278                        tokens: quote![#expr],
279                    }
280                } else {
281                    Err(syn::Error::new(
282                        span,
283                        "let bindings aren't yet supported at runtime",
284                    ))?
285                }
286            }
287            Expr::Match(mat) => {
288                let span = mat.span();
289                let elem = Expression::from_expr(*mat.expr.clone(), context)?;
290
291                if elem.is_const() {
292                    let mut arms = Vec::new();
293
294                    for arm in mat.arms.iter() {
295                        arms.push(ConstMatchArm {
296                            pat: arm.pat.clone(),
297                            expr: Box::new(Self::from_expr(arm.body.as_ref().clone(), context)?),
298                        });
299                    }
300
301                    Expression::ConstMatch {
302                        const_expr: mat.expr.as_ref().clone(),
303                        arms,
304                    }
305                } else if let Some(switch) = numeric_match(mat, context) {
306                    switch
307                } else {
308                    Err(syn::Error::new(
309                        span,
310                        "Only numeric match expression is supported at runtime",
311                    ))?
312                }
313            }
314            Expr::Macro(mac) if is_comptime_macro(&mac.mac.path) => {
315                let tokens = mac.mac.tokens;
316                Expression::Verbatim {
317                    tokens: quote![{ #tokens }],
318                }
319            }
320            Expr::Macro(mac) => Expression::Verbatim {
321                tokens: quote![#mac],
322            },
323            Expr::Struct(init) => {
324                let fields = init
325                    .fields
326                    .clone()
327                    .into_iter()
328                    .map(|field| {
329                        let member = field.member;
330                        let value = Expression::from_expr(field.expr, context)?;
331                        syn::Result::Ok((member, value))
332                    })
333                    .collect::<Result<Vec<_>, _>>()?;
334                Expression::StructInit {
335                    path: init.path,
336                    fields,
337                }
338            }
339            Expr::Unsafe(unsafe_expr) => {
340                let (block, _) =
341                    context.in_scope(|ctx| Block::from_block(unsafe_expr.block, ctx))?;
342                Expression::Block(block)
343            }
344            Expr::Infer(_) => Expression::Verbatim { tokens: quote![_] },
345            Expr::Verbatim(verbatim) => Expression::Verbatim { tokens: verbatim },
346            Expr::Reference(reference) => Expression::Reference {
347                inner: Box::new(Expression::from_expr(*reference.expr, context)?),
348            },
349            Expr::Closure(expr) => {
350                let (body, scope) =
351                    context.in_scope(|ctx| Expression::from_expr(*expr.body, ctx))?;
352                let body = Box::new(body);
353                let params = expr.inputs.into_iter().collect();
354                Expression::Closure {
355                    params,
356                    body,
357                    scope,
358                }
359            }
360
361            Expr::Try(expr) => {
362                let span = expr.span();
363                let expr = Expression::from_expr(*expr.expr, context)?
364                    .as_const(context)
365                    .ok_or_else(|| syn::Error::new(span, "? Operator not supported at runtime"))?;
366                Expression::Verbatim {
367                    tokens: quote_spanned![span=> #expr?],
368                }
369            }
370            Expr::TryBlock(_) => Err(syn::Error::new_spanned(
371                expr,
372                "try_blocks is unstable and not supported in kernels",
373            ))?,
374            e => Err(syn::Error::new_spanned(
375                expr,
376                format!("Unsupported expression {e:?}"),
377            ))?,
378        };
379        Ok(result)
380    }
381}

383fn lit_ty(lit: &Lit) -> syn::Result<Type> {
384    let res = match lit {
385        Lit::Int(int) => (!int.suffix().is_empty())
386            .then(|| int.suffix())
387            .map(|suffix| format_ident!("{suffix}"))
388            .and_then(|ident| syn::parse2(quote![#ident]).ok())
389            .unwrap_or_else(|| parse_quote![i32]),
390        Lit::Float(float) => (!float.suffix().is_empty())
391            .then(|| float.suffix())
392            .map(|suffix| format_ident!("{suffix}"))
393            .and_then(|ident| syn::parse2(quote![#ident]).ok())
394            .unwrap_or_else(|| parse_quote![f32]),
395        Lit::Bool(_) => parse_quote![bool],
396        lit => Err(syn::Error::new_spanned(
397            lit,
398            format!("Unsupported literal type: {lit:?}"),
399        ))?,
400    };
401    Ok(res)
402}

404fn generate_strided_index(
405    tensor: &Expression,
406    elements: Vec<Expression>,
407    span: Span,
408) -> syn::Result<Expression> {
409    let index_ty = elements
410        .first()
411        .unwrap()
412        .ty()
413        .unwrap_or_else(|| parse_quote![u32]);
414    let strided_indices = elements.into_iter().enumerate().map(|(i, elem)| {
415        let i = Lit::Int(LitInt::new(&i.to_string(), span));
416        let stride = Expression::MethodCall {
417            receiver: Box::new(tensor.clone()),
418            method: format_ident!("stride"),
419            args: vec![Expression::Literal {
420                value: i,
421                ty: index_ty.clone(),
422            }],
423            generics: None,
424            span,
425        };
426        Expression::Binary {
427            left: Box::new(elem),
428            operator: Operator::Mul,
429            right: Box::new(stride),
430            ty: None,
431            span,
432        }
433    });
434    let sum = strided_indices
435        .reduce(|a, b| Expression::Binary {
436            left: Box::new(a),
437            operator: Operator::Add,
438            right: Box::new(b),
439            ty: None,
440            span,
441        })
442        .unwrap();
443    Ok(sum)
444}

446fn is_slice(index: &Expression) -> bool {
447    match index {
448        Expression::Range { .. } => true,
449        Expression::Array { elements, .. } => elements.iter().any(is_slice),
450        Expression::Tuple { elements, .. } => elements.iter().any(is_slice),
451        _ => false,
452    }
453}

455fn fn_associated_type(path: &Expression) -> Option<(Path, PathSegment)> {
456    // All supported primitives. Primitives don't start with an uppercase letter
457    const PRIMITIVES: &[&str] = &[
458        "bool", "i8", "i16", "i32", "i64", "u8", "u16", "u32", "u64", "f16", "bf16", "f32", "f64",
459    ];
460    if !matches!(path, Expression::Path { .. }) {
461        panic!("path: {path:?}");
462    }
463    match path {
464        Expression::Path { path, .. } => {
465            let second_last = path.segments.iter().nth_back(1)?;
466            let name = second_last.ident.to_string();
467            let ch = name.chars().next();
468            let is_assoc = ch.map(|ch| ch.is_uppercase()).unwrap_or(false);
469            let is_primitive = PRIMITIVES.contains(&name.as_str());
470            if is_assoc || is_primitive {
471                let mut path = path.clone();
472                let name = path.segments.pop().unwrap().into_value();
473                path.segments.pop_punct();
474                Some((path, name))
475            } else {
476                None
477            }
478        }
479        _ => None,
480    }
481}

483fn is_comptime_macro(path: &Path) -> bool {

485    "::cubecl::comptime".ends_with(&path)
