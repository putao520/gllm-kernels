46/// Cooperative Matrix-Multiply and Accumulate Instruction.
47#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq)]
48#[allow(missing_docs)]
49pub enum CoopMma {
50    /// Fill the matrix with the value.
51    Fill { value: Variable },
52    /// Load the value into the matrix given the stride.
53    Load {
54        value: Variable,
55        stride: Variable,
56        layout: Option<MatrixLayout>,
57    },
58    /// Executes D=A*B+C;
59    ///
60    /// For implementing a matmul, `D=C` : `C+=A*B`
61    Execute {
62        mat_a: Variable,
63        mat_b: Variable,
64        mat_c: Variable,
65    },
66    /// Store the matrix in an output variable following the stride and the layout.
67    Store {
68        mat: Variable,
69        stride: Variable,
70        layout: MatrixLayout,
71    },
72    /// Cast a fragment to another type.
73    Cast { input: Variable },
74}
75
76impl Display for CoopMma {
77    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
78        match self {
79            CoopMma::Fill { value } => write!(f, "{}", value),
80            CoopMma::Load {
81                value,
82                stride,
83                layout,
84            } => {
85                let layout = layout
86                    .map(|it| format!(", layout: {it:?}"))
87                    .unwrap_or(String::new());
88                write!(f, "matrix_load({}, stride: {}{layout})", value, stride)
89            }
90            CoopMma::Execute {
91                mat_a,
92                mat_b,
93                mat_c,
94            } => write!(f, "execute_cmma({}, {}, {})", mat_a, mat_b, mat_c),
95            CoopMma::Store {
96                mat,
97                stride,
98                layout,
99            } => write!(
100                f,
101                "matrix_store({}, stride: {}, layout: {:?})",
102                mat, stride, layout
103            ),
104            CoopMma::Cast { input } => {
105                write!(f, "matrix_cast(input: {})", input)
106            }
107        }
108    }
109}
