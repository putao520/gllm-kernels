9const DEFAULT_CUBE_DIM: CubeDim = CubeDim::new_2d(32, 8);
10const DEFAULT_PLANE_COUNT: u32 = 8;

12#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
13pub enum LineMode {
14    Parallel,
15    Perpendicular,
16}
17
18#[derive(Debug, Clone)]
19pub(crate) struct ReduceConfig {
20    pub(crate) cube_count: CubeCount,
21    pub(crate) cube_dim: CubeDim,
22    pub(crate) line_mode: LineMode,
23    pub(crate) line_size: u32,
24    pub(crate) bound_checks: bool,
25}
26
27impl ReduceConfig {
28    pub(crate) fn generate<R: Runtime, In: CubePrimitive>(
29        client: &ComputeClient<R::Server, R::Channel>,
30        input: &TensorHandleRef<R>,
31        output: &TensorHandleRef<R>,
32        axis: usize,
33        strategy: &ReduceStrategy,
34    ) -> ReduceConfig {
35        let reduce_count = output.size() as u32;
36        ReduceConfig::new()
37            .generate_line_mode(input, axis)
38            .generate_line_size::<R, In>(input, output, axis)
39            .generate_cube_dim(client, strategy.use_planes)
40            .generate_cube_count(reduce_count, strategy)
41    }
42
43    fn new() -> Self {
44        // This is only a dummy configuration to use as a starting point.
45        Self {
46            cube_count: CubeCount::new_single(),
47            cube_dim: CubeDim::new_single(),
48            line_mode: LineMode::Parallel,
49            line_size: 1,
50            bound_checks: true,
51        }
52    }
53
54    fn generate_line_mode<R: Runtime>(mut self, input: &TensorHandleRef<R>, axis: usize) -> Self {
55        let stride = input.strides[axis];
56        self.line_mode = if stride == 1 {
57            LineMode::Parallel
58        } else {
59            LineMode::Perpendicular
60        };
61        self
62    }
63
64    fn generate_line_size<R: Runtime, In: CubePrimitive>(
65        mut self,
66        input: &TensorHandleRef<R>,
67        output: &TensorHandleRef<R>,
68        axis: usize,
69    ) -> Self {
70        let elem = In::as_elem_native_unchecked();
71        let supported_line_sizes = R::line_size_elem(&elem);
72        self.line_size = match self.line_mode {
73            LineMode::Parallel => {
74                tensor_line_size_parallel(supported_line_sizes, input.shape, input.strides, axis)
75                    as u32
76            }
77            LineMode::Perpendicular => {
78                // To compute the maximum line size we can used,
79                // we first sort both the input and output axes by increasing strides.
80                // As example, consider
81                //    input shape = [2, 4, 6, 8]
82                //    input stride = [1, 16, 64, 2]
83                //    output shape = [2, 1, 6, 8]
84                //    output stride = [1, 1, 2, 12]
85                //    axis = 1
86                //
87                // then we have
88                //    input sorted axis = [0, 3, 1, 2]
89                //    output sorted axis = [0, 1, 2, 3]
90                //
91                // From that point, we look at all the axes before the target axis in the sorted input.
92                // That is [0, 3] in the example.
93                // In the output, we remove the target axis leading to [0, 2, 3] in the example.
94                //
95                // In order to use perpendicular line, we are limited by the number of entries that are both
96                // contiguous in the input and output. This is obtained by taking the head of each list until they are different.
97                // In the above example, only the 0 axis is contiguous in both tensor, but it output sorted axis were [0, 1, 3, 2] instead,
98                // both the 0 and 3 axes would be contiguous in the two tensors.
99                // The corresponding number of entries is the product of the shape for the contigous axes.
100                // In the example, it is simply 2.
101                //
102                // This gives us an upper bound on the line size we can used.
103                // Then, we use the regular method to find the best line size that match the device capacities.
104
105                let mut input_axis_and_strides =
106                    input.strides.iter().enumerate().collect::<Vec<_>>();
107                input_axis_and_strides.sort_by_key(|(_, stride)| *stride);
108                let input_sorted_axis = input_axis_and_strides
109                    .into_iter()
110                    .map(|(a, _)| a)
111                    .take_while(|a| *a != axis);
112
113                let mut output_axis_and_strides =
114                    output.strides.iter().enumerate().collect::<Vec<_>>();
115                output_axis_and_strides.sort_by_key(|(_, stride)| *stride);
116                let output_sorted_axis = output_axis_and_strides
117                    .into_iter()
118                    .filter_map(|(a, _)| (a != axis).then_some(a));
119
120                let max_line_size = input_sorted_axis
121                    .zip(output_sorted_axis)
122                    .filter_map(|(i, o)| (i == o).then_some(output.shape[i]))
123                    .product();
124
125                tensor_line_size_perpendicular(
126                    supported_line_sizes.filter(|size| {
127                        *size as usize <= max_line_size && max_line_size % *size as usize == 0
128                    }),
129                    input.shape,
130                    input.strides,
131                    axis,
132                ) as u32
133            }
134        };
135        self
136    }
137
138    fn generate_cube_dim<S: ComputeServer, C: ComputeChannel<S>>(
139        mut self,
140        client: &ComputeClient<S, C>,
141        use_planes: bool,
142    ) -> Self {
143        self.cube_dim = if use_planes {
144            let plane_dim = client.properties().hardware_properties().plane_size_min;
145            CubeDim::new_2d(plane_dim, DEFAULT_PLANE_COUNT)
146        } else {
147            DEFAULT_CUBE_DIM
148        };
149        self
150    }
151
152    fn generate_cube_count(mut self, reduce_count: u32, strategy: &ReduceStrategy) -> Self {
153        let agent_count_per_cube =  // An agent is either a unit, a plane or a whole cube depending on the strategy.
154            match strategy {
155                ReduceStrategy { shared: true, .. } => 1,
156                ReduceStrategy { use_planes: true, .. } => self.cube_dim.y,
157                ReduceStrategy { use_planes: false, .. } => self.cube_dim.num_elems(),
158            };
159        let reduce_count_per_cube = match self.line_mode {
160            LineMode::Parallel => agent_count_per_cube,
161            LineMode::Perpendicular => agent_count_per_cube * self.line_size,
162        };
163
164        let cube_count = reduce_count.div_ceil(reduce_count_per_cube);
165
166        self.do_bound_checks_if(reduce_count_per_cube * cube_count > reduce_count);
167        self.cube_count = CubeCount::new_1d(cube_count);
168
169        self
170    }
171
172    fn do_bound_checks_if(&mut self, condition: bool) {
173        self.bound_checks = self.bound_checks || condition;
174    }
175}
