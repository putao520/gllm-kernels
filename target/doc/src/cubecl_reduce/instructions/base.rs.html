8/// An instruction for a reduce algorithm that works with [`Line`].
9///
10/// See a provided implementation, such as [`Sum`] or [`ArgMax`] for an example how to implement
11/// this trait for a custom instruction.
12///
13/// A reduction works at three levels. First, it takes input data of type `In` and reduce them
14/// with their coordinate into an `AccumulatorItem`. Then, multiple `AccumulatorItem` are possibly fused
15/// together into a single accumulator that is converted to the expected output type.
16#[cube]
17pub trait ReduceInstruction<In: Numeric>: Send + Sync + 'static + std::fmt::Debug {
18    /// The intermediate state into which we accumulate new input elements.
19    /// This is most likely a `Line<T>` or a struct or tuple of lines.
20    type AccumulatorItem: CubeType;
21
22    /// When multiple agents are collaborating to reduce a single slice,
23    /// we need a share accumulator to store multiple `AccumulatorItem`.
24    /// This is most likely a `SharedMemory<Line<T>>` or a struct or tuple of lined shared memories.
25    type SharedAccumulator: SharedAccumulator<In, Item = Self::AccumulatorItem>;
26
27    /// A input such that `Self::reduce(accumulator, Self::null_input(), coordinate, use_planes)`
28    /// is guaranteed to return `accumulator` unchanged for any choice of `coordinate`.
29    fn null_input(#[comptime] line_size: u32) -> Line<In>;
30
31    /// A accumulator such that `Self::fuse_accumulators(accumulator, Self::null_accumulator()` always returns
32    /// is guaranteed to return `accumulator` unchanged.
33    fn null_accumulator(#[comptime] line_size: u32) -> Self::AccumulatorItem;
34
35    /// Assign the value of `source` into `destination`.
36    /// In spirit, this is equivalent to `destination = source;`,
37    /// but this syntax is not currently supported by CubeCL.
38    fn assign_accumulator(destination: &mut Self::AccumulatorItem, source: &Self::AccumulatorItem);
39
40    /// If `use_planes` is `true`, reduce all the `item` and `coordinate` within the `accumulator`.
41    /// Else, reduce the given `item` and `coordinate` into the accumulator.
42    fn reduce(
43        accumulator: &Self::AccumulatorItem,
44        item: Line<In>,
45        coordinate: Line<u32>,
46        #[comptime] use_planes: bool,
47    ) -> Self::AccumulatorItem;
48
49    /// Reduce two accumulators into a single accumulator.
50    fn fuse_accumulators(
51        lhs: Self::AccumulatorItem,
52        rhs: Self::AccumulatorItem,
53    ) -> Self::AccumulatorItem;
54
55    /// Reduce all elements of the accumulator into a single output element of type `Out`.
56    fn merge_line<Out: Numeric>(accumulator: Self::AccumulatorItem, shape_axis_reduce: u32) -> Out;
57
58    /// Convert each element of the accumulator into the expected output element of type `Out`.
59    fn to_output_perpendicular<Out: Numeric>(
60        accumulator: Self::AccumulatorItem,
61        shape_axis_reduce: u32,
62    ) -> Line<Out>;
63}

65/// A simple trait that abstract over a single or multiple shared memory.
66#[cube]
67pub trait SharedAccumulator<In: Numeric>: CubeType + Send + Sync + 'static {
68    type Item: CubeType;
69
70    fn allocate(#[comptime] length: u32, #[comptime] line_size: u32) -> Self;
71
72    fn read(accumulator: &Self, index: u32) -> Self::Item;
73
74    fn write(accumulator: &mut Self, index: u32, item: Self::Item);
75}
76
77#[cube]
78impl<In: Numeric> SharedAccumulator<In> for SharedMemory<Line<In>> {
79    type Item = Line<In>;
80
81    fn allocate(#[comptime] length: u32, #[comptime] line_size: u32) -> Self {
82        SharedMemory::new_lined(length, line_size)
83    }
84
85    fn read(accumulator: &Self, index: u32) -> Self::Item {
86        accumulator[index]
87    }
88
89    fn write(accumulator: &mut Self, index: u32, item: Self::Item) {
90        accumulator[index] = item;
91    }
92}

94/// A pair of shared memory used for [`ArgMax`] and [`ArgMin`].
95#[derive(CubeType)]
96pub struct ArgAccumulator<N: Numeric> {
97    pub elements: SharedMemory<Line<N>>,
98    pub args: SharedMemory<Line<u32>>,
99}
100
101#[cube]
102impl<In: Numeric> SharedAccumulator<In> for ArgAccumulator<In> {
103    type Item = (Line<In>, Line<u32>);
104
105    fn allocate(#[comptime] length: u32, #[comptime] line_size: u32) -> Self {
106        ArgAccumulator::<In> {
107            elements: SharedMemory::new_lined(length, line_size),
108            args: SharedMemory::new_lined(length, line_size),
109        }
110    }
111
112    fn read(accumulator: &Self, index: u32) -> Self::Item {
113        (accumulator.elements[index], accumulator.args[index])
114    }
115
116    fn write(accumulator: &mut Self, index: u32, item: Self::Item) {
117        accumulator.elements[index] = item.0;
118        accumulator.args[index] = item.1;
119    }
120}

122#[cube]
123pub fn reduce_inplace<In: Numeric, R: ReduceInstruction<In>>(
124    accumulator: &mut R::AccumulatorItem,
125    item: Line<In>,
126    coordinate: Line<u32>,
127    #[comptime] use_planes: bool,
128) {
129    let reduction = &R::reduce(accumulator, item, coordinate, use_planes);
130    R::assign_accumulator(accumulator, reduction);
131}

133#[cube]
134pub fn reduce_shared_inplace<In: Numeric, R: ReduceInstruction<In>>(
135    accumulator: &mut R::SharedAccumulator,
136    index: u32,
137    item: Line<In>,
138    coordinate: Line<u32>,
139    #[comptime] use_planes: bool,
140) {
141    let acc_item = R::SharedAccumulator::read(accumulator, index);
142    let reduction = R::reduce(&acc_item, item, coordinate, use_planes);
143    R::SharedAccumulator::write(accumulator, index, reduction);
144}

146#[cube]
147pub fn fuse_accumulator_inplace<In: Numeric, R: ReduceInstruction<In>>(
148    accumulator: &mut R::SharedAccumulator,
149    destination: u32,
150    origin: u32,
151) {
152    let fused = R::fuse_accumulators(
153        R::SharedAccumulator::read(accumulator, destination),
154        R::SharedAccumulator::read(accumulator, origin),
155    );
156    R::SharedAccumulator::write(accumulator, destination, fused);
157}
