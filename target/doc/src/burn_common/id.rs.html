4/// Simple ID generator.

7impl IdGenerator {

10        // Generate a random u64 (18,446,744,073,709,551,615 combinations)

12        u64::from_le_bytes(random_bytes)

69/// Unique identifier that can represent a stream based on the current thread id.
70#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash, Serialize, Deserialize, PartialOrd, Ord)]
71pub struct StreamId {
72    /// The value representing the thread id.
73    pub value: u64,
74}
75
76impl StreamId {
77    /// Get the current thread id.
78    pub fn current() -> Self {
79        Self {
80            #[cfg(feature = "std")]
81            value: Self::from_current_thread(),
82            #[cfg(not(feature = "std"))]
83            value: 0,
84        }
85    }
86
87    #[cfg(feature = "std")]
88    fn from_current_thread() -> u64 {
89        use core::hash::Hash;
90
91        std::thread_local! {
92            static ID: std::cell::OnceCell::<u64> = const { std::cell::OnceCell::new() };
93        };
94
95        // Getting the current thread is expensive, so we cache the value into a thread local
96        // variable, which is very fast.
97        ID.with(|cell| {
98            *cell.get_or_init(|| {
99                // A way to get a thread id encoded as u64.
100                let mut hasher = std::hash::DefaultHasher::default();
101                let id = std::thread::current().id();
102                id.hash(&mut hasher);
103                std::hash::Hasher::finish(&hasher)
104            })
105        })
106    }
107}
108
109impl core::fmt::Display for StreamId {
110    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
111        f.write_fmt(format_args!("StreamId({:?})", self.value))
112    }
