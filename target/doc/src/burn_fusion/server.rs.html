8pub struct FusionServer<R: FusionRuntime> {
9    streams: MultiStream<R>,
10    pub(crate) handles: HandleContainer<R::FusionHandle>,
11}
12
13impl<R> FusionServer<R>
14where
15    R: FusionRuntime,

17    pub fn new(device: R::FusionDevice) -> Self {
18        Self {
19            streams: MultiStream::new(device.clone()),
20            handles: HandleContainer::new(),
21        }
22    }
23
24    pub fn register(
25        &mut self,
26        streams: Vec<StreamId>,
27        desc: OperationDescription,
28        operation: Box<dyn Operation<R>>,
29    ) {
30        self.streams
31            .register(streams, desc, operation, &mut self.handles)
32    }
33
34    pub fn drain_stream(&mut self, id: StreamId) {
35        self.streams.drain(&mut self.handles, id)
36    }
37
38    pub fn create_empty_handle(&mut self) -> Arc<TensorId> {
39        self.handles.create_tensor_uninit()
40    }
41
42    pub fn read_float<B>(
43        &mut self,
44        tensor: TensorDescription,
45        id: StreamId,
46    ) -> impl Future<Output = burn_tensor::TensorData> + 'static
47    where
48        B: FusionBackend<FusionRuntime = R>,
49    {
50        // Make sure all registered operations are executed.
51        // The underlying backend can still be async.
52        self.drain_stream(id);
53
54        let tensor = self.handles.get_float_tensor::<B>(&tensor);
55        B::float_into_data(tensor)
56    }

106    pub fn change_server_float<B>(
107        &mut self,
108        tensor: &TensorDescription,
109        device: &R::FusionDevice,
110        server_device: &mut Self,
111    ) -> Arc<TensorId>
112    where
113        B: FusionBackend<FusionRuntime = R>,
114    {
115        let tensor = self.handles.get_float_tensor::<B>(tensor);
116        let tensor = B::float_to_device(tensor, device);
117        let id = server_device.create_empty_handle();
118
119        server_device
120            .handles
121            .register_float_tensor::<B>(&id, tensor.clone());
122
123        id
124    }
