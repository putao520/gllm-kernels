9/// A growing list of [tensor operation descriptions](OperationDescription).
10pub struct OperationQueue<R: FusionRuntime> {
11    /// List of operation descriptions. These contain the exact tensor IDs
12    /// and shapes so that kernels can be run correctly.
13    ///
14    /// The length of this list is the same as the length of the `operations` list.
15    pub(crate) global: Vec<OperationDescription>,
16    /// List of operation descriptions. The tensor IDs and shapes are relative
17    /// because we don't need to know the exact values, but they are sufficient to
18    /// determine which operations can be fused.
19    pub(crate) relative: Vec<OperationDescription>,
20    pub(crate) converter: OperationConverter,
21    pub(crate) operations: Vec<Box<dyn Operation<R>>>,
22    pub(crate) ids: BTreeSet<TensorId>,
23}
24
25impl<R: FusionRuntime> Default for OperationQueue<R> {
26    fn default() -> Self {
27        Self::new()
28    }
29}
30
31impl<R: FusionRuntime> OperationQueue<R> {
32    /// Create a new empty queue.
33    pub fn new() -> Self {
34        Self {
35            global: Vec::new(),
36            relative: Vec::new(),
37            converter: OperationConverter::default(),
38            operations: Vec::new(),
39            ids: BTreeSet::new(),
40        }
41    }
42
43    /// Add a new tensor operation to the queue.
44    ///
45    /// The new [operation description](OperationDescription) will be converted to a local
46    /// representation that can be reused when the same pattern emerge in different but similar
47    /// scenario, so that the same optimization can be used.
48    pub fn add(&mut self, global: OperationDescription, operation: Box<dyn Operation<R>>) {
49        for node in global.nodes() {
50            self.ids.insert(node.id);
51        }
52        let relative = global.to_relative(&mut self.converter);
53        self.relative.push(relative);
54        self.global.push(global);
55        self.operations.push(operation);
56    }
57
58    /// The size of the queue.
59    pub fn len(&self) -> usize {
60        self.global.len()
61    }
62
63    /// If the queue is empty.
64    pub fn is_empty(&self) -> bool {
65        self.len() == 0
66    }
67}

73    #[test]
74    fn stream_id_from_different_threads() {
75        let current = StreamId::current();
76
77        let thread1 = std::thread::spawn(|| (StreamId::current(), StreamId::current()));
78        let thread2 = std::thread::spawn(StreamId::current);
79
80        let (stream_1, stream_11) = thread1.join().unwrap();
81        let stream_2 = thread2.join().unwrap();
82
83        assert_ne!(current, stream_1, "Should be different from thread 1");
84        assert_ne!(current, stream_2, "Should be different from thread 2");
85        assert_ne!(
86            stream_1, stream_2,
87            "Should be different from different threads"
88        );
89        assert_eq!(
90            stream_1, stream_11,
91            "Should be the same, since same thread."
92        );
93    }
