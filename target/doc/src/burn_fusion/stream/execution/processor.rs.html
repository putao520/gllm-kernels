10/// Process a [stream segment](StreamSegment) following a [policy](Policy).
11pub(crate) struct Processor<O> {
12    policy: Policy<O>,
13    explorer: Explorer<O>,
14}
15
16/// A part of a stream that can be executed partially using [execution plan](ExecutionPlan).
17pub(crate) trait StreamSegment<O> {
18    /// The operations in the segment.
19    fn operations(&self) -> &[OperationDescription];
20    /// Execute part of the segment using the given plan id.
21    fn execute(&mut self, id: ExecutionPlanId, store: &mut ExecutionPlanStore<O>);
22}
23
24impl<O> Processor<O> {
25    /// Create a new stream processor.
26    pub fn new(optimizations: Vec<Box<dyn OptimizationBuilder<O>>>) -> Self {
27        Self {
28            policy: Policy::new(),
29            explorer: Explorer::new(optimizations),
30        }
31    }
32
33    /// Process the [stream segment](StreamSegment) with the provided [mode](ExecutionMode).
34    pub fn process<Segment>(
35        &mut self,
36        mut segment: Segment,
37        store: &mut ExecutionPlanStore<O>,
38        mode: ExecutionMode,
39    ) where
40        Segment: StreamSegment<O>,
41    {
42        // We assume that we always register a new operation in lazy mode.
43        if let ExecutionMode::Lazy = mode {
44            self.on_new_operation(&segment, store);
45        }
46
47        loop {
48            if segment.operations().is_empty() {
49                break;
50            }
51
52            match self.policy.action(store, segment.operations(), mode) {
53                Action::Explore => {
54                    self.explore(&mut segment, store, mode);
55
56                    if self.explorer.is_up_to_date() {
57                        break;
58                    }
59                }
60                Action::Defer => {
61                    match mode {
62                        ExecutionMode::Lazy => break,
63                        ExecutionMode::Sync => panic!("Can't defer while sync"),
64                    };
65                }
66                Action::Execute(id) => {
67                    if let ExecutionMode::Sync = mode {
68                        store.add_trigger(id, ExecutionTrigger::OnSync);
69                    }
70
71                    segment.execute(id, store);
72                    self.reset(store, segment.operations());
73                }
74            };
75        }
76    }
77
78    fn on_new_operation<Segment>(&mut self, segment: &Segment, store: &mut ExecutionPlanStore<O>)
79    where
80        Segment: StreamSegment<O>,
81    {
82        self.policy.update(
83            store,
84            segment
85                .operations()
86                .last()
87                .expect("At least one operation in the operation list."),
88        );
89        self.explorer.on_new_operation();
90    }
91
92    fn explore<Item: StreamSegment<O>>(
93        &mut self,
94        item: &mut Item,
95        store: &mut ExecutionPlanStore<O>,
96        mode: ExecutionMode,
97    ) {
98        match self.explorer.explore(item.operations(), mode) {
99            Exploration::Found(optim) => {
100                let id = Self::on_optimization_found(
101                    &self.policy,
102                    item.operations(),
103                    store,
104                    optim,
105                    mode,
106                );
107                item.execute(id, store);
108                self.reset(store, item.operations());
109            }
110            Exploration::NotFound { num_explored } => {
111                let id = Self::on_optimization_not_found(
112                    &self.policy,
113                    item.operations(),
114                    store,
115                    mode,
116                    num_explored,
117                );
118
119                item.execute(id, store);
120                self.reset(store, item.operations());
121            }
122            Exploration::Continue => {
123                if let ExecutionMode::Sync = mode {
124                    panic!("Can't continue exploring when sync.")
125                }
126            }
127        }
128    }
129
130    fn reset(&mut self, store: &mut ExecutionPlanStore<O>, operations: &[OperationDescription]) {
131        self.explorer.reset(operations);
132        self.policy.reset();
133
134        // Reset the policy state with the remaining operations
135        for operation in operations.iter() {
136            self.policy.update(store, operation);
137        }
138    }
139
140    /// We found an optimization (i.e. a new execution plan).
141    /// Cache it in the store.
142    fn on_optimization_found(
143        policy: &Policy<O>,
144        operations: &[OperationDescription],
145        store: &mut ExecutionPlanStore<O>,
146        builder: &dyn OptimizationBuilder<O>,
147        mode: ExecutionMode,
148    ) -> ExecutionPlanId {
149        let num_fused = builder.len();
150        let relative = &operations[0..num_fused];
151
152        match mode {
153            ExecutionMode::Lazy => {
154                let next_ops = &operations[num_fused..operations.len()];
155
156                let trigger = if next_ops.is_empty() {
157                    // Happens if the next ops is included in the fused operation, and there is no
158                    // way the builder can still continue fusing.
159                    ExecutionTrigger::Always
160                } else {
161                    ExecutionTrigger::OnOperations(next_ops.to_vec())
162                };
163
164                match policy.action(store, relative, ExecutionMode::Sync) {
165                    Action::Execute(id) => {
166                        store.add_trigger(id, trigger);
167                        id
168                    }
169                    _ => store.add(ExecutionPlan {
170                        operations: relative.to_vec(),
171                        triggers: vec![trigger],
172                        strategy: ExecutionStrategy::Optimization(builder.build()),
173                    }),
174                }
175            }
176            ExecutionMode::Sync => match policy.action(store, relative, ExecutionMode::Sync) {
177                Action::Execute(id) => {
178                    store.add_trigger(id, ExecutionTrigger::OnSync);
179                    id
180                }
181                _ => store.add(ExecutionPlan {
182                    operations: relative.to_vec(),
183                    triggers: vec![ExecutionTrigger::OnSync],
184                    strategy: ExecutionStrategy::Optimization(builder.build()),
185                }),
186            },
187        }
188    }
189
190    fn on_optimization_not_found(
191        policy: &Policy<O>,
192        operations: &[OperationDescription],
193        store: &mut ExecutionPlanStore<O>,
194        mode: ExecutionMode,
195        num_explored: usize,
196    ) -> ExecutionPlanId {
197        let relative = &operations[0..num_explored];
198        let trigger = match mode {
199            ExecutionMode::Lazy => ExecutionTrigger::Always,
200            ExecutionMode::Sync => ExecutionTrigger::OnSync,
201        };
202
203        match policy.action(store, relative, ExecutionMode::Sync) {
204            Action::Execute(id) => {
205                store.add_trigger(id, trigger);
206                id
207            }
208            _ => store.add(ExecutionPlan {
209                operations: relative.to_vec(),
210                triggers: vec![trigger],
211                strategy: ExecutionStrategy::Operations,
212            }),
213        }
214    }
215}
