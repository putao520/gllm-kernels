5/// The context contains the relative graph tensor mapping so that a relative tensor id can be
6/// mapped to an existing tensor that can be fetched and updated with the
7/// [handle container](HandleContainer).
8///
9/// It also contains all scalar values, which can change even for the same graph. They are sorted
10/// in the order in which they appear in the graph.
11#[allow(clippy::too_many_arguments)]
12#[derive(new)]
13pub struct Context<'a, H> {
14    /// The tensor mapping where local tensor id points to the updated tensor description.
15    pub tensors: &'a mut HashMap<TensorId, TensorDescription>,
16    /// Handle container to retrieve tensors based on their description.
17    pub handles: &'a mut HandleContainer<H>,
18    /// F32 scalars found in the graph in the order they appeared.
19    pub scalar_f32: &'a Vec<f32>,
20    /// F16 scalars found in the graph in the order they appeared.
21    pub scalar_f16: &'a Vec<f16>,
22    /// BF16 scalars found in the graph in the order they appeared.
23    pub scalar_bf16: &'a Vec<bf16>,
24    /// i64 scalars found in the graph in the order they appeared.
25    pub scalar_i64: &'a Vec<i64>,
26    /// i32 scalars found in the graph in the order they appeared.
27    pub scalar_i32: &'a Vec<i32>,
28    /// i16 scalars found in the graph in the order they appeared.
29    pub scalar_i16: &'a Vec<i16>,
30    /// i8 scalars found in the graph in the order they appeared.
31    pub scalar_i8: &'a Vec<i8>,
32    /// u64 scalars found in the graph in the order they appeared.
33    pub scalar_u64: &'a Vec<u64>,
34    /// u32 scalars found in the graph in the order they appeared.
35    pub scalar_u32: &'a Vec<u32>,
36    /// u16 scalars found in the graph in the order they appeared.
37    pub scalar_u16: &'a Vec<u16>,
38    /// u8 scalars found in the graph in the order they appeared.
39    pub scalar_u8: &'a Vec<u8>,
40}

62pub(crate) trait RelativeOps {
63    /// Convert (usually an [`OperationDescription`]) to a relative form.
64    ///
65    /// The id and the shape of tensors will be computed relative to existing
66    /// operations in the queue. We do this because we want to fuse operations
67    /// that have similar shapes, but we do not care about the exact values.
68    ///
69    /// Similar we do not care about the exact ids of the tensor, but about their
70    /// relative ids (how close they are in the operation queue)
71    fn to_relative(&self, converter: &mut OperationConverter) -> Self;
72}
73
74trait RelativeOpsScalar<E: Element> {
75    fn to_relative<F>(&self, converter: &mut OperationConverter, local_elem: F) -> Self
76    where
77        F: Fn(&mut OperationConverter, &E) -> E;
78}

119    pub(crate) fn relative_float<E: Element>(&mut self, elem: &E, dtype: &DType) -> E {
120        match dtype {
121            burn_tensor::DType::F32 => self.scalar_f32.push(elem.elem()),
122            burn_tensor::DType::F16 => self.scalar_f16.push(elem.elem()),
123            burn_tensor::DType::BF16 => self.scalar_bf16.push(elem.elem()),
124            _ => todo!("Unsupported"),
125        }
126
127        // We return 0 so that the id from a scalar operation is the same no matter its scalar
128        // value.
129        0.elem()
130    }
131
132    pub(crate) fn relative_int<E: Element>(&mut self, elem: &E, dtype: &DType) -> E {
133        match dtype {
134            DType::I64 => self.scalar_i64.push(elem.elem()),
135            DType::I32 => self.scalar_i32.push(elem.elem()),
136            DType::I16 => self.scalar_i16.push(elem.elem()),
137            DType::I8 => self.scalar_i8.push(elem.elem()),
138            DType::U64 => self.scalar_u64.push(elem.elem()),
139            DType::U32 => self.scalar_u32.push(elem.elem()),
140            DType::U16 => self.scalar_u16.push(elem.elem()),
141            DType::U8 => self.scalar_u8.push(elem.elem()),
142            _ => todo!("Unsupported"),
143        }
144        // We return 0 so that the id from a scalar operation is the same no matter its scalar
145        // value.
146        0.elem()
147    }

150impl RelativeOps for OperationDescription {
151    fn to_relative(&self, converter: &mut OperationConverter) -> Self {
152        match self {
153            OperationDescription::BaseFloat(ops) => {
154                OperationDescription::BaseFloat(ops.to_relative(converter))
155            }
156            OperationDescription::BaseInt(ops) => {
157                OperationDescription::BaseInt(ops.to_relative(converter))
158            }
159            OperationDescription::BaseBool(ops) => {
160                OperationDescription::BaseBool(ops.to_relative(converter))
161            }
162            OperationDescription::NumericFloat(dtype, ops) => OperationDescription::NumericFloat(
163                *dtype,
164                ops.to_relative(converter, |converter, e| converter.relative_float(e, dtype)),
165            ),
166            OperationDescription::NumericInt(dtype, ops) => OperationDescription::NumericInt(
167                *dtype,
168                ops.to_relative(converter, |converter, e| converter.relative_int(e, dtype)),
169            ),
170            OperationDescription::Bool(ops) => {
171                OperationDescription::Bool(ops.to_relative(converter))
172            }
173            OperationDescription::Int(ops) => OperationDescription::Int(ops.to_relative(converter)),
174            OperationDescription::Float(dtype, ops) => OperationDescription::Float(
175                *dtype,
176                RelativeOpsScalar::<f32>::to_relative(ops, converter, |converter, e| {
177                    converter.relative_float(e, dtype)
178                }),
179            ),
180            OperationDescription::Module(ops) => {
181                OperationDescription::Module(ops.to_relative(converter))
182            }
183            OperationDescription::Custom(ops) => {
184                OperationDescription::Custom(ops.to_relative(converter))
185            }
186        }
187    }
188}

1063impl RelativeOps for TensorDescription {
1064    fn to_relative(&self, converter: &mut OperationConverter) -> Self {
1065        let relative_id = if let Some(value) = converter.tensors_global2relative.get(&self.id) {
1066            // If we already have the same tensor registered, we have to update its value, but not
1067            // its id.
1068            value.id
1069        } else {
1070            // We create a new relative id since we never seen this tensor in the graph before.
1071            TensorId::new(converter.tensors_relative2global.len() as u64)
1072        };
1073
1074        // We can create relative shapes by mapping each shape found to an ID, which is a `usize`.
1075        let mut relative_shape = Vec::with_capacity(self.shape.len());
1076        for dim in self.shape.iter() {
1077            if let Some(dim_id) = converter.shapes_global2relative.get(dim) {
1078                // We already saw that dim value before, so we retrieve its ID.
1079                relative_shape.push(*dim_id);
1080            } else {
1081                // We never saw this dim value before, therefore we create a new ID.
1082                let dim_id = converter.shapes_global2relative.len();
1083                relative_shape.push(dim_id);
1084                converter.shapes_global2relative.insert(*dim, dim_id);
1085            }
1086        }
1087
1088        // We create the relative tensor.
1089        let relative_tensor = TensorDescription {
1090            id: relative_id,
1091            shape: relative_shape,
1092            status: self.status.clone(),
1093            dtype: self.dtype,
1094        };
1095
1096        // We update both mappings.
1097        converter
1098            .tensors_relative2global
1099            .insert(relative_id, self.clone());
1100        converter
1101            .tensors_global2relative
1102            .insert(self.id, relative_tensor.clone());
1103
1104        relative_tensor
1105    }
1106}
