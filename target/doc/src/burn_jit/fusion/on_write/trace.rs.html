28/// A trace runner is responsible for determining the vectorization factor as well as launching
29/// a kernel based on global [inputs](GlobalArgsLaunch) and [outputs](GlobalArgsLaunch)
30/// with a provided [element wise config](ElemwiseConfig).
31pub trait TraceRunner<R: JitRuntime> {
32    /// The error that might happen while running the trace.
33    type Error;
34
35    /// Run the trace.
36    fn run<'a>(
37        &'a self,
38        client: &'a ComputeClient<R::Server, R::Channel>,
39        inputs: GlobalArgsLaunch<'a, R>,
40        outputs: GlobalArgsLaunch<'a, R>,
41        config: &'a ElemwiseConfig,
42    ) -> Result<(), Self::Error>;
43
44    /// The vectorization factor for all inputs and outputs.
45    fn vectorization<'a>(
46        handles_inputs: impl Iterator<Item = &'a JitFusionHandle<R>>,
47        inputs: impl Iterator<Item = &'a TensorDescription>,
48        outputs: impl Iterator<Item = &'a TensorDescription>,
49    ) -> u8 {
50        // The default version uses the last dimension as vectorization axis and assumes a
51        // perpendicular contiguous line.
52
53        let vectorization_input = |handle: &JitFusionHandle<R>, desc: &TensorDescription| {
54            let rank = handle.strides.len();
55
56            // Last dimension strides should be 1, otherwise vecX won't be contiguous.
57            if handle.strides[rank - 1] != 1 {
58                return 1;
59            }
60
61            for s in R::line_size_elem(&desc.dtype.into()) {
62                // The last dimension should be a multiple of the vector size.
63                if desc.shape[rank - 1] % s as usize == 0 {
64                    return s;
65                }
66            }
67
68            1
69        };
70
71        let vectorization_output = |desc: &TensorDescription| {
72            let rank = desc.shape.len();
73
74            for s in R::line_size_elem(&desc.dtype.into()) {
75                // The last dimension should be a multiple of the vector size.
76                if desc.shape[rank - 1] % s as usize == 0 {
77                    return s;
78                }
79            }
80
81            1
82        };
83
84        let mut output = u8::MAX;
85
86        for (handle, tensor) in handles_inputs.zip(inputs) {
87            output = Ord::min(vectorization_input(handle, tensor), output);
88        }
89
90        for tensor in outputs {
91            output = Ord::min(vectorization_output(tensor), output);
92        }
93
94        output
95    }
96}

149impl FuseOnWriteTrace {
150    /// Run a trace with the given [runner](TraceRunner).
151    pub fn run<R: JitRuntime, BT: BoolElement, Runner: TraceRunner<R>>(
152        &self,
153        client: &ComputeClient<R::Server, R::Channel>,
154        device: &R::Device,
155        context: &mut Context<'_, JitFusionHandle<R>>,
156        runner: &Runner,
157    ) -> Result<(), Runner::Error> {
158        let analysis = self.analyse::<R, BT, Runner>(client, device, context);
159
160        let inputs = self.register_inputs(context, &analysis.handle_inputs, analysis.vectorization);
161        let outputs =
162            self.register_outputs::<_, BT>(&analysis.handle_outputs, analysis.vectorization);
163
164        let mut ops = Sequence::new();
165        for op in analysis.reads.into_values() {
166            ops.push(op);
167        }
168
169        for op in self.ops.iter() {
170            ops.push(op.clone());
171        }
172
173        for op in analysis.writes.into_values() {
174            ops.push(op);
175        }
176
177        let config = ElemwiseConfig {
178            rank: analysis.rank as u32,
179            ref_layout: analysis
180                .reference
181                .expect("An output should exist for the fused kernel")
182                .layout,
183            ops,
184        };
185
186        match Runner::run(runner, client, inputs, outputs, &config) {
187            Err(err) => {
188                self.rollback(context, analysis.handle_inputs, analysis.handle_outputs);
189                Err(err)
190            }
191            Ok(val) => Ok(val),
192        }
193    }
194
195    fn rollback<R: JitRuntime>(
196        &self,
197        context: &mut Context<'_, JitFusionHandle<R>>,
198        handle_inputs: Vec<HandleInput<R>>,
199        handle_outputs: Vec<HandleOutput<R>>,
200    ) {
201        for input in handle_inputs {
202            context
203                .handles
204                .register_handle(input.global_id, input.handle);
205        }
206        for output in handle_outputs {
207            if let HandleOutput::Owned {
208                global_id, handle, ..
209            } = output
210            {
211                context.handles.register_handle(global_id, handle);
212            }
213        }
214    }

216    fn analyse<'a, R: JitRuntime, BT: BoolElement, Runner: TraceRunner<R>>(
217        &'a self,
218        client: &ComputeClient<R::Server, R::Channel>,
219        device: &R::Device,
220        context: &mut Context<'_, JitFusionHandle<R>>,
221    ) -> LaunchAnalysis<'a, R> {
222        let mut analysis = LaunchAnalysis {
223            potential_inplaces: Vec::new(),
224            global_inputs: Vec::new(),
225            global_outputs: Vec::new(),
226            handle_inputs: Vec::new(),
227            handle_outputs: Vec::new(),
228            reference: None,
229            reads: self.reads.clone(),
230            writes: self.writes.clone(),
231            rank: 1,
232            vectorization: 1,
233        };
234
235        self.analyse_inputs(context, &mut analysis);
236        self.analyse_outputs::<_, BT>(client, device, context, &mut analysis);
237
238        analysis.vectorization = Runner::vectorization(
239            analysis.handle_inputs.iter().map(|item| &item.handle),
240            analysis.global_inputs.iter(),
241            analysis.global_outputs.iter(),
242        );
243
244        analysis
245    }

247    fn analyse_inputs<'a, R: JitRuntime>(
248        &'a self,
249        context: &mut Context<'_, JitFusionHandle<R>>,
250        analysis: &mut LaunchAnalysis<'a, R>,
251    ) {
252        for (i, (precision, tensor_relative)) in self.inputs.iter().enumerate() {
253            let tensor_global = context.tensors.get(&tensor_relative.id).unwrap().clone();
254            // Important to take the status of the relative graph and not
255            // the global graph, since the status of the global graph
256            // might be of a later operation on the same tensor id.
257            let status = &tensor_relative.status;
258            let handle = context.handles.get_handle(&tensor_global.id, status);
259
260            if status == &TensorStatus::ReadWrite
261                && handle.handle.can_mut()
262                && !self.inputs_unhandled.contains(&tensor_relative.id)
263            {
264                analysis.potential_inplaces.push(PotentialInplace {
265                    input_pos: i,
266                    tensor_relative,
267                    strides: handle.strides.clone(),
268                });
269            }
270
271            analysis.rank = usize::max(tensor_global.shape.len(), analysis.rank);
272            analysis.handle_inputs.push(HandleInput {
273                precision,
274                handle,
275                relative_id: tensor_relative.id,
276                global_id: tensor_global.id,
277                global_shape: tensor_global.shape.clone(),
278            });
279            analysis.global_inputs.push(tensor_global);
280        }
281    }

283    fn analyse_outputs<'a, R: JitRuntime, BT: BoolElement>(
284        &'a self,
285        client: &ComputeClient<R::Server, R::Channel>,
286        device: &R::Device,
287        context: &mut Context<'_, JitFusionHandle<R>>,
288        analysis: &mut LaunchAnalysis<'a, R>,
289    ) {
290        for (precision, tensor_relative) in self.outputs.iter() {
291            let tensor_global = context.tensors.get(&tensor_relative.id).unwrap().clone();
292            let strides = strides_dyn_rank(&tensor_global.shape);
293
294            if let Some(index) = analysis
295                .potential_inplaces
296                .iter()
297                .enumerate()
298                .find(|(_pos, pi)| {
299                    pi.tensor_relative.dtype == tensor_global.dtype
300                        && pi.tensor_relative.shape == tensor_relative.shape
301                        && pi.strides == strides
302                })
303                .map(|(pos, _)| pos)
304            {
305                let potential_inplace = analysis.potential_inplaces.remove(index);
306                let handle_input = analysis
307                    .handle_inputs
308                    .get(potential_inplace.input_pos)
309                    .unwrap();
310
311                if analysis.reference.is_none() {
312                    let index_input = self
313                        .inputs
314                        .get_index(precision, potential_inplace.tensor_relative.id)
315                        .unwrap();
316
317                    analysis.reference = Some(Reference {
318                        layout: Arg::Input(index_input as u32, precision, LayoutInfo::IsRef),
319                        shape: tensor_global.shape.clone(),
320                        strides: handle_input.handle.strides.clone(),
321                    });
322
323                    if let Some(ElemwiseOp::Assign(op)) =
324                        analysis.reads.get_mut(&handle_input.relative_id)
325                    {
326                        op.input.add_layout_info(LayoutInfo::IsRef);
327                    };
328
329                    if let Some(ElemwiseOp::Assign(op)) =
330                        analysis.writes.get_mut(&tensor_relative.id)
331                    {
332                        op.out.add_layout_info(LayoutInfo::IsRef);
333                    };
334                }
335
336                context
337                    .handles
338                    .register_handle(tensor_global.id, handle_input.handle.clone());
339                analysis.handle_outputs.push(HandleOutput::Alias {
340                    input_pos: potential_inplace.input_pos,
341                    precision,
342                });
343                analysis.global_outputs.push(tensor_global);
344            } else {
345                if analysis.reference.is_none() {
346                    analysis.reference = Some(Reference {
347                        layout: Arg::Output(0, precision, LayoutInfo::IsRef),
348                        shape: tensor_global.shape.clone(),
349                        strides: strides.clone(),
350                    });
351
352                    if let ElemwiseOp::Assign(op) =
353                        analysis.writes.get_mut(&tensor_relative.id).unwrap()
354                    {
355                        op.out.add_layout_info(LayoutInfo::IsRef);
356                    };
357                } else if let Some(reference) = analysis.reference.as_ref() {
358                    if reference.strides == strides && reference.shape == tensor_global.shape {
359                        if let ElemwiseOp::Assign(op) =
360                            analysis.writes.get_mut(&tensor_relative.id).unwrap()
361                        {
362                            op.out.add_layout_info(LayoutInfo::SameAsRef);
363                        };
364                    }
365                }
366
367                // We encode bool tensors as `B`.
368                let dtype = match tensor_global.dtype {
369                    DType::Bool => BT::dtype(),
370                    _ => tensor_global.dtype,
371                };
372                let size = tensor_global.shape.iter().product::<usize>() * Elem::from(dtype).size();
373
374                let handle = JitFusionHandle {
375                    client: client.clone(),
376                    handle: client.empty(size),
377                    device: device.clone(),
378                    strides,
379                    dtype,
380                };
381
382                analysis.rank = usize::max(tensor_global.shape.len(), analysis.rank);
383                context
384                    .handles
385                    .register_handle(tensor_global.id, handle.clone());
386
387                analysis.handle_outputs.push(HandleOutput::Owned {
388                    precision,
389                    handle,
390                    global_shape: tensor_global.shape.clone(),
391                    global_id: tensor_global.id,
392                });
393                analysis.global_outputs.push(tensor_global);
394            }
395        }
396
397        Self::add_layout_info_inputs(analysis);
398    }

400    fn add_layout_info_inputs<R: JitRuntime>(analysis: &mut LaunchAnalysis<'_, R>) {
401        for hi in analysis.handle_inputs.iter() {
402            if let Some(reference) = analysis.reference.as_ref() {
403                if reference.strides == hi.handle.strides && reference.shape == hi.global_shape {
404                    if let Some(ElemwiseOp::Assign(op)) = analysis.reads.get_mut(&hi.relative_id) {
405                        op.input.add_layout_info(LayoutInfo::SameAsRef);
406                    }
407                }
408            }
409        }
410    }

412    fn register_inputs<'h, R: JitRuntime>(
413        &self,
414        context: &mut Context<'_, JitFusionHandle<R>>,
415        handle_inputs: &'h [HandleInput<R>],
416        vectorization: u8,
417    ) -> GlobalArgsLaunch<'h, R> {
418        let mut inputs = GlobalArgsLaunch::new(
419            SequenceArg::new(),
420            SequenceArg::new(),
421            SequenceArg::new(),
422            SequenceArg::new(),
423            SequenceArg::new(),
424            SequenceArg::new(),

428            SequenceArg::new(),

435            SequenceArg::new(),
436            SequenceArg::new(),
437            SequenceArg::new(),
438            SequenceArg::new(),
439            SequenceArg::new(),
440            SequenceArg::new(),
441        );
442
443        for hi in handle_inputs.iter() {
444            let arg = hi.handle.as_tensor_arg(&hi.global_shape, vectorization);
445            match hi.precision {
446                ElemwisePrecision::F32 => inputs.t_f32.push(arg),
447                ElemwisePrecision::F16 => inputs.t_f16.push(arg),
448                ElemwisePrecision::BF16 => inputs.t_bf16.push(arg),
449                ElemwisePrecision::I64 => inputs.t_i64.push(arg),
450                ElemwisePrecision::I32 => inputs.t_i32.push(arg),
451                ElemwisePrecision::I16 => inputs.t_i16.push(arg),
452                ElemwisePrecision::I8 => inputs.t_i8.push(arg),
453                ElemwisePrecision::U64 => inputs.t_u64.push(arg),
454                ElemwisePrecision::U32 => inputs.t_u32.push(arg),
455                ElemwisePrecision::U16 => inputs.t_u16.push(arg),
456                ElemwisePrecision::U8 => inputs.t_u8.push(arg),
457                _ => panic!("Unsupported input precision {:?}", hi.precision),
458            };
459        }
460
461        for (precision, count) in self.scalars.iter() {
462            for i in 0..(*count as usize) {
463                match precision {
464                    ElemwisePrecision::F32 => {
465                        inputs.s_f32.push(ScalarArg::new(context.scalar_f32[i]))
466                    }
467                    ElemwisePrecision::F16 => {
468                        inputs.s_f16.push(ScalarArg::new(context.scalar_f16[i]))
469                    }
470                    ElemwisePrecision::BF16 => {
471                        inputs.s_bf16.push(ScalarArg::new(context.scalar_bf16[i]))
472                    }
473                    ElemwisePrecision::I64 => {
474                        inputs.s_i64.push(ScalarArg::new(context.scalar_i64[i]))
475                    }
476                    ElemwisePrecision::I32 => {
477                        inputs.s_i32.push(ScalarArg::new(context.scalar_i32[i]))
478                    }
479                    ElemwisePrecision::I16 => {
480                        inputs.s_i16.push(ScalarArg::new(context.scalar_i16[i]))
481                    }
482                    ElemwisePrecision::I8 => inputs.s_i8.push(ScalarArg::new(context.scalar_i8[i])),
483                    ElemwisePrecision::U64 => {
484                        inputs.s_u64.push(ScalarArg::new(context.scalar_u64[i]))
485                    }
486                    ElemwisePrecision::U32 => {
487                        inputs.s_u32.push(ScalarArg::new(context.scalar_u32[i]))
488                    }
489                    ElemwisePrecision::U16 => {
490                        inputs.s_u16.push(ScalarArg::new(context.scalar_u16[i]))
491                    }
492                    ElemwisePrecision::U8 => inputs.s_u8.push(ScalarArg::new(context.scalar_u8[i])),
493                    ElemwisePrecision::Bool => todo!(),
494                }
495            }
496        }
497
498        inputs
499    }

577        }

580    }

583#[derive(Default, Clone, Serialize, Deserialize, Debug)]
584pub struct RegisteredTensors {
585    tensors: BTreeMap<ElemwisePrecision, Vec<TensorDescription>>,
586}
587
588impl RegisteredTensors {
589    pub fn iter(&self) -> impl Iterator<Item = (ElemwisePrecision, &TensorDescription)> {
590        self.tensors.iter().flat_map(|(precision, descriptions)| {
591            descriptions.iter().map(|desc| (*precision, desc))
592        })
593    }
594
595    pub fn len(&self) -> usize {
596        self.tensors.values().map(|v| v.len()).sum()
597    }
598
599    pub fn get_index(&self, precision: ElemwisePrecision, tensor_id: TensorId) -> Option<usize> {
600        self.tensors.get(&precision).and_then(|items| {
601            items
602                .iter()
603                .enumerate()
604                .find(|(_pos, tensor)| tensor.id == tensor_id)
605                .map(|(pos, _)| pos)
606        })
607    }
608
609    pub fn get_all(&self, precision: ElemwisePrecision) -> &[TensorDescription] {
610        self.tensors
611            .get(&precision)
612            .map(|v| v.as_slice())
613            .unwrap_or(&[])
614    }
615
616    pub fn get(
617        &self,
618        precision: ElemwisePrecision,
619        tensor_id: TensorId,
620    ) -> Option<&TensorDescription> {
621        self.get_all(precision)
622            .iter()
623            .find(|desc| desc.id == tensor_id)
624    }
625
626    pub fn insert(&mut self, precision: ElemwisePrecision, tensor: TensorDescription) -> u32 {
627        if let Some(tensors) = self.tensors.get_mut(&precision) {
628            let position = tensors.len() as u32;
629            tensors.push(tensor);
630            position
631        } else {
632            self.tensors.insert(precision, vec![tensor]);
633            0
634        }
635    }
636
637    pub fn update(&mut self, precision: ElemwisePrecision, tensor: &TensorDescription) {
638        if let Some(tensors) = self.tensors.get_mut(&precision) {
639            if let Some(tensor_old) = tensors
640                .iter_mut()
641                .find(|tensor_old| tensor_old.id == tensor.id)
642            {
643                tensor_old.status = tensor.status.clone();
644            }
645        }
646    }
