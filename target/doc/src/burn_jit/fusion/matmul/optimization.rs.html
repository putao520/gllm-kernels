31/// Fuse matmul operation followed by elemwise operations into a single kernel.
32pub struct MatmulOptimization<R: JitRuntime> {
33    trace: FuseOnWriteTrace,
34    trace_fallback: FuseOnWriteTrace,
35    client: ComputeClient<R::Server, R::Channel>,
36    device: R::Device,
37    len: usize,
38    matmul: FusedMatmul,
39}

42/// State for the [matrix optimization](MatmulOptimizationState).

50impl<R: JitRuntime> MatmulOptimization<R> {
51    /// Execute the optimization.
52    pub fn execute<BT: BoolElement>(&mut self, context: &mut Context<'_, JitFusionHandle<R>>) {
53        if self.execute_fused::<BT>(context).is_err() {
54            self.execute_fallback::<BT>(context);
55        }
56    }
57
58    /// Number of operations fused.
59    pub fn num_ops_fused(&self) -> usize {
60        self.len
61    }
62
63    /// Create an optimization from its [state](MatmulOptimizationState).
64    pub fn from_state(device: &R::Device, state: MatmulOptimizationState) -> Self {
65        Self {
66            trace: state.trace,
67            trace_fallback: state.trace_fallback,
68            len: state.len,
69            client: R::client(device),
70            device: device.clone(),
71            matmul: state.matmul.clone(),
72        }
73    }
74
75    /// Convert the optimization to its [state](MatmulOptimizationState).
76    pub fn to_state(&self) -> MatmulOptimizationState {
77        MatmulOptimizationState {
78            trace: self.trace.clone(),
79            trace_fallback: self.trace_fallback.clone(),
80            matmul: self.matmul.clone(),
81            len: self.len,
82        }
83    }
84
85    fn execute_fused<BT: BoolElement>(
86        &mut self,
87        context: &mut Context<'_, JitFusionHandle<R>>,
88    ) -> Result<(), FusedMatmulError> {
89        self.trace
90            .run::<R, BT, FusedMatmul>(&self.client, &self.device, context, &self.matmul)
91    }
92
93    fn execute_fallback<BT: BoolElement>(&mut self, context: &mut Context<'_, JitFusionHandle<R>>) {
94        match self.matmul.lhs.precision() {
95            ElemwisePrecision::F32 => self.run_fallback::<BT, f32>(context),
96            ElemwisePrecision::F16 => self.run_fallback::<BT, f16>(context),
97            ElemwisePrecision::BF16 => self.run_fallback::<BT, bf16>(context),
98            _ => panic!("Unsupported precision"),
99        }
100    }
101
102    fn run_fallback<BT: BoolElement, EG: FloatElement>(
103        &mut self,
104        context: &mut Context<'_, JitFusionHandle<R>>,
105    ) {
106        let (out_tensor, out_desc) = {
107            let lhs = context.tensors.get(&self.matmul.op.lhs.id).unwrap().clone();
108            let rhs = context.tensors.get(&self.matmul.op.rhs.id).unwrap().clone();
109            let out = context.tensors.get(&self.matmul.op.out.id).unwrap().clone();
110
111            let lhs_handle = context.handles.get_handle(&lhs.id, &TensorStatus::ReadOnly);
112            let rhs_handle = context.handles.get_handle(&rhs.id, &TensorStatus::ReadOnly);
113
114            let lhs_tensor = lhs_handle.into_tensor(Shape {
115                dims: lhs.shape.clone(),
116            });
117            let rhs_tensor = rhs_handle.into_tensor(Shape {
118                dims: rhs.shape.clone(),
119            });
120            let out_tensor = matmul::matmul::<R, EG>(
121                lhs_tensor,
122                rhs_tensor,
123                None,
124                matmul::MatmulStrategy::default(),
125            )
126            .unwrap();
127            (out_tensor, out)
128        };
129        context
130            .handles
131            .register_handle(out_desc.id, JitFusionHandle::from(out_tensor));
132
133        self.trace_fallback
134            .run::<R, BT, ElemwiseRunner>(&self.client, &self.device, context, &ElemwiseRunner)
135            .unwrap();
136    }
137}

159impl<R: JitRuntime> TraceRunner<R> for FusedMatmul {
160    type Error = FusedMatmulError;
161
162    fn run<'a>(
163        &'a self,
164        client: &'a ComputeClient<R::Server, R::Channel>,
165        inputs: GlobalArgsLaunch<'a, R>,
166        outputs: GlobalArgsLaunch<'a, R>,
167        config: &'a ElemwiseConfig,
168    ) -> Result<(), FusedMatmulError> {
169        match self.out.precision() {
170            ElemwisePrecision::F32 => self.matmul_fused::<R, f32>(client, inputs, outputs, config),
171            ElemwisePrecision::F16 => self.matmul_fused::<R, f16>(client, inputs, outputs, config),
172            ElemwisePrecision::BF16 => {
173                self.matmul_fused::<R, bf16>(client, inputs, outputs, config)
174            }
175            _ => panic!("Unsupported precision"),
176        }
177    }
178}

180impl FusedMatmul {
181    fn matmul_fused<'a, R: JitRuntime, EG: Numeric>(
182        &'a self,
183        client: &'a ComputeClient<R::Server, R::Channel>,
184        inputs: GlobalArgsLaunch<'a, R>,
185        outputs: GlobalArgsLaunch<'a, R>,
186        config: &'a ElemwiseConfig,
187    ) -> Result<(), FusedMatmulError> {
188        let lhs_shape = inputs.shape(&self.lhs);
189        let rhs_shape = inputs.shape(&self.rhs);
190
191        let lhs_strides = inputs.strides(&self.lhs);
192        let rhs_strides = inputs.strides(&self.rhs);
193
194        let check_layout = |strides| match matrix_layout(strides) {
195            MatrixLayout::Contiguous => (false, false),
196            MatrixLayout::MildlyPermuted {
197                transposed,
198                batch_swap: _,
199            } => (false, transposed),
200            MatrixLayout::HighlyPermuted => (true, false),
201        };
202
203        let (lhs_make_contiguous, lhs_transposed) = check_layout(lhs_strides);
204        let (rhs_make_contiguous, rhs_transposed) = check_layout(rhs_strides);
205
206        if lhs_make_contiguous || rhs_make_contiguous {
207            return Err(FusedMatmulError::InvalidInput);
208        }
209
210        let rank = lhs_shape.len();
211
212        let m = lhs_shape[rank - 2] as u32;
213        let k = lhs_shape[rank - 1] as u32;
214        let n = rhs_shape[rank - 1] as u32;
215
216        let lhs_line_size = inputs.line_size(&self.lhs);
217        let rhs_line_size = inputs.line_size(&self.rhs);
218        let out_line_size = match config.ref_layout {
219            Arg::Input(..) => inputs.line_size(&config.ref_layout),
220            Arg::Output(..) => outputs.line_size(&config.ref_layout),
221            _ => panic!("Invalid ref layout"),
222        };
223
224        if out_line_size == 1 && (lhs_line_size > 1 || rhs_line_size > 1) {
225            return Err(FusedMatmulError::InvalidInput);
226        }
227
228        let problem = MatmulProblem {
229            m: m as usize,
230            n: n as usize,
231            k: k as usize,
232            batches: (
233                lhs_shape[..lhs_shape.len() - 2].to_vec(),
234                rhs_shape[..rhs_shape.len() - 2].to_vec(),
235            ),
236            lhs_layout: match lhs_transposed {
237                true => components::MatrixLayout::ColMajor,
238                false => components::MatrixLayout::RowMajor,
239            },
240            rhs_layout: match rhs_transposed {
241                true => components::MatrixLayout::ColMajor,
242                false => components::MatrixLayout::RowMajor,
243            },
244            lhs_line_size,
245            rhs_line_size,
246            out_line_size,
247        };
248
249        let plane_size = client
250            .properties()
251            .hardware_properties()
252            .defined_plane_size();
253
254        let plane_size = match plane_size {
255            Some(val) => val,
256            None => {
257                return Err(MatmulLaunchError::Unavailable(
258                    MatmulAvailabilityError::PlaneDimUnknown,
259                )
260                .into())
261            }
262        };
263
264        match matmul_launch_kernel::<R, EG, StandardSelector<Accelerated>>(
265            client,
266            FusedMatmulInputLaunch::new(inputs, config, &self.lhs, &self.rhs, &self.out),
267            outputs,
268            problem,
269            plane_size,
270        ) {
271            Ok(_) => Ok(()),
272            Err(err) => Err(FusedMatmulError::LaunchError(err)),
273        }
274    }
275}

277fn matmul_launch_kernel<'a, R: Runtime, EG: Numeric, S: MatmulSelector>(
278    client: &ComputeClient<R::Server, R::Channel>,
279    input: FusedMatmulInputLaunch<'a, R>,
280    output: GlobalArgsLaunch<'a, R>,
281    problem: MatmulProblem,
282    plane_size: u32,
283) -> Result<(), MatmulLaunchError> {
284    if TypeId::of::<EG>() == TypeId::of::<half::f16>()
285        || TypeId::of::<EG>() == TypeId::of::<flex32>()
286    {
287        S::select_kernel::<FusedMatmulSpec<EG, half::f16, f32>, R>(
288            client, input, output, problem, plane_size,
289        )
290    } else if TypeId::of::<EG>() == TypeId::of::<half::bf16>() {
291        S::select_kernel::<FusedMatmulSpec<EG, half::bf16, f32>, R>(
292            client, input, output, problem, plane_size,
293        )
294    } else if S::stage_tf32_supported() {
295        S::select_kernel::<FusedMatmulSpec<EG, tf32, f32>, R>(
296            client, input, output, problem, plane_size,
297        )
298    } else {
299        S::select_kernel::<FusedMatmulSpec<EG, EG, f32>, R>(
300            client, input, output, problem, plane_size,
301        )
302    }
