9#[cube(launch_unchecked)]
10fn max_pool2d_with_indices_backward_kernel<E: Numeric, I: Int>(
11    grad: &Tensor<E>,
12    indices: &Tensor<I>,
13    output: &mut Tensor<E>,
14    args: &PoolBackwardArgs,
15    #[comptime] kernel_size_0: i32,
16    #[comptime] kernel_size_1: i32,
17) {
18    if ABSOLUTE_POS >= output.len() {
19        return;
20    }
21
22    let batch = ABSOLUTE_POS / output.stride(0) % output.shape(0);
23    let channel = ABSOLUTE_POS / output.stride(1) % output.shape(1);
24    let ih = ABSOLUTE_POS / output.stride(2) % output.shape(2);
25    let iw = ABSOLUTE_POS / output.stride(3) % output.shape(3);
26
27    let index_current = ih * output.stride(2) + iw * output.stride(3);
28
29    let (oh_start, oh_end, ow_start, ow_end) = loop_ranges(
30        ih as i32,
31        iw as i32,
32        grad.shape(2),
33        grad.shape(3),
34        args,
35        kernel_size_0,
36        kernel_size_1,
37    );
38
39    let mut grad_acc = E::from_int(0);
40
41    let index_base = batch * grad.stride(0) + channel * grad.stride(1);
42
43    for oh in oh_start..oh_end {
44        for ow in ow_start..ow_end {
45            let index = index_base + oh * grad.stride(2) + ow * grad.stride(3);
46            let index_max = u32::cast_from(indices[index]);
47
48            grad_acc += select(index_max == index_current, grad[index], E::from_int(0));
49        }
50    }
51
52    output[ABSOLUTE_POS] = grad_acc;
53}

55#[cube]
56fn loop_ranges(
57    ih: i32,
58    iw: i32,
59    grad_h: u32,
60    grad_w: u32,
61    args: &PoolBackwardArgs,
62    #[comptime] kernel_size_0: i32,
63    #[comptime] kernel_size_1: i32,
64) -> (u32, u32, u32, u32) {
65    let kms_0 = args.dilation_0 * kernel_size_0 - args.stride_0;
66    let kms_1 = args.dilation_1 * kernel_size_1 - args.stride_1;
67
68    let oh_start = Max::max((ih + args.padding_0 - kms_0) / args.stride_0, 0) as u32;
69    let ow_start = Max::max((iw + args.padding_1 - kms_1) / args.stride_1, 0) as u32;
70    let oh_end = Min::min(Max::max(kms_0, 0) as u32 + oh_start, grad_h - 1) + 1;
71    let ow_end = Min::min(Max::max(kms_1, 0) as u32 + ow_start, grad_w - 1) + 1;
72
73    (oh_start, oh_end, ow_start, ow_end)
74}

76pub(crate) fn max_pool2d_with_indices_backward<R: JitRuntime, E: JitElement, I: IntElement>(
77    x: JitTensor<R>,
78    grad: JitTensor<R>,
79    indices: JitTensor<R>,
80    kernel_size: [usize; 2],
81    stride: [usize; 2],
82    padding: [usize; 2],
83    dilation: [usize; 2],
84) -> JitTensor<R> {
85    let grad = into_contiguous(grad);
86    let indices = into_contiguous(indices);
87
88    let output = empty_device::<R, E>(x.client.clone(), x.device.clone(), x.shape.clone());
89    let cube_dim = CubeDim::default();
90    let cube_count = calculate_cube_count_elemwise(output.shape.num_elements(), cube_dim);
91
92    unsafe {
93        max_pool2d_with_indices_backward_kernel::launch_unchecked::<E, I, R>(
94            &x.client,
95            cube_count,
96            cube_dim,
97            grad.as_tensor_arg::<E>(1),
98            indices.as_tensor_arg::<I>(1),
99            output.as_tensor_arg::<E>(1),
100            PoolBackwardArgsLaunch::new(
101                ScalarArg::new(stride[0] as i32),
102                ScalarArg::new(stride[1] as i32),
103                ScalarArg::new(dilation[0] as i32),
104                ScalarArg::new(dilation[0] as i32),
105                ScalarArg::new(padding[0] as i32),
106                ScalarArg::new(padding[1] as i32),
107            ),
108            kernel_size[0] as i32,
109            kernel_size[1] as i32,
110        )
111    };
112
113    output
