5#[cube(launch_unchecked)]
6fn interpolate_nearest_kernel<F: Float>(input: &Tensor<F>, output: &mut Tensor<F>) {
7    if ABSOLUTE_POS >= output.len() {
8        return;
9    }
10
11    let batch = ABSOLUTE_POS / output.stride(0) % output.shape(0);
12    let channel = ABSOLUTE_POS / output.stride(1) % output.shape(1);
13    let y = ABSOLUTE_POS / output.stride(2) % output.shape(2);
14    let x = ABSOLUTE_POS / output.stride(3) % output.shape(3);
15
16    let factor = F::cast_from(y);
17    let numerator = F::cast_from(input.shape(2));
18    let denominator = F::cast_from(output.shape(2));
19    let y: F = Floor::floor(factor * numerator / denominator);
20
21    let factor = F::cast_from(x);
22    let numerator = F::cast_from(input.shape(3));
23    let denominator = F::cast_from(output.shape(3));
24    let x: F = Floor::floor(factor * numerator / denominator);
25
26    let index = batch * input.stride(0)
27        + channel * input.stride(1)
28        + u32::cast_from(y) * input.stride(2)
29        + u32::cast_from(x) * input.stride(3);
30
31    output[ABSOLUTE_POS] = input[index];
32}

34pub(crate) fn interpolate_nearest_launch<R: JitRuntime, E: FloatElement>(
35    input: JitTensor<R>,
36    output: JitTensor<R>,
37) -> JitTensor<R> {
38    let cube_dim = CubeDim::default();
39    let cube_count = calculate_cube_count_elemwise(output.shape.num_elements(), cube_dim);
40
41    unsafe {
42        interpolate_nearest_kernel::launch_unchecked::<E, R>(
43            &input.client,
44            cube_count,
45            cube_dim,
46            input.as_tensor_arg::<E>(1),
47            output.as_tensor_arg::<E>(1),
48        )
49    };
50
51    output
52}
