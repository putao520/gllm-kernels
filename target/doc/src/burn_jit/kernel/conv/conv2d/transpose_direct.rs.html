16struct ConvArgs {
17    conv_stride_0: u32,
18    conv_stride_1: u32,
19    dilation_0: u32,
20    dilation_1: u32,
21    padding_0: u32,
22    padding_1: u32,
23    groups: u32,
24}

26#[cube(launch)]
27fn conv_transpose2d_direct_kernel<E: Numeric>(
28    input: &Tensor<E>,
29    weight: &Tensor<E>,
30    bias: &Tensor<E>,
31    output: &mut Tensor<E>,
32    args: ConvArgs,
33) {
34    if ABSOLUTE_POS >= output.len() {
35        return;
36    }
37
38    let in_c_per_group = weight.shape(0) / args.groups;
39    let out_c_per_group = weight.shape(1);
40    let kernel_h = weight.shape(2);
41    let kernel_w = weight.shape(3);
42
43    let batch = ABSOLUTE_POS / output.stride(0) % output.shape(0);
44    let oc_out = ABSOLUTE_POS / output.stride(1) % output.shape(1);
45    let out_y = ABSOLUTE_POS / output.stride(2) % output.shape(2);
46    let out_x = ABSOLUTE_POS / output.stride(3) % output.shape(3);
47
48    let k = oc_out / out_c_per_group;
49    let group = k % args.groups;
50    let out_c = oc_out - out_c_per_group * group;
51
52    let in_c_start = group * in_c_per_group;
53    let in_c_end = in_c_start + in_c_per_group;
54
55    let stride_0_i = args.conv_stride_0 as i32;
56    let stride_1_i = args.conv_stride_1 as i32;
57
58    let kms_h = (kernel_h * args.dilation_0) as i32 - stride_0_i;
59    let kms_w = (kernel_w * args.dilation_1) as i32 - stride_1_i;
60
61    let y_start = ((out_y + args.padding_0) as i32 - kms_h) / stride_0_i;
62    let x_start = ((out_x + args.padding_1) as i32 - kms_w) / stride_1_i;
63
64    let y_end = Min::min(Max::max(kms_h + y_start + 1, 0) as u32, input.shape(2));
65    let x_end = Min::min(Max::max(kms_w + x_start + 1, 0) as u32, input.shape(3));
66    let y_start = Max::max(y_start, 0) as u32;
67    let x_start = Max::max(x_start, 0) as u32;
68
69    let idx_input_batch = batch * input.stride(0);
70    let idx_weight_oc = out_c * weight.stride(1);
71
72    let mut sum = bias[oc_out];
73
74    let numerator_h_base = out_y + args.padding_0;
75    let numerator_w_base = out_x + args.padding_1;
76
77    for in_c in in_c_start..in_c_end {
78        let idx_input_ic = in_c * input.stride(1);
79        let idx_weight_ic = in_c * weight.stride(0);
80
81        for in_y in y_start..y_end {
82            let numerator_tmp = in_y * args.conv_stride_0;
83            let numerator_h = numerator_h_base - numerator_tmp;
84
85            if numerator_h_base >= numerator_tmp && numerator_h % args.dilation_0 == 0 {
86                let kernel_y = numerator_h / args.dilation_0;
87                let idx_input_y = in_y * input.stride(2);
88                let idx_weight_ky = kernel_y * weight.stride(2);
89
90                for in_x in x_start..x_end {
91                    let numerator_tmp = in_x * args.conv_stride_1;
92                    let numerator_w = numerator_w_base - numerator_tmp;
93
94                    if numerator_w_base >= numerator_tmp && numerator_w % args.dilation_1 == 0 {
95                        let kernel_x = numerator_w / args.dilation_1;
96                        let idx_input_x = in_x * input.stride(3);
97                        let idx_weight_kx = kernel_x * weight.stride(3);
98
99                        let index_input =
100                            idx_input_batch + idx_input_ic + idx_input_y + idx_input_x;
101                        let index_weight =
102                            idx_weight_ic + idx_weight_oc + idx_weight_ky + idx_weight_kx;
103
104                        let value = input[index_input];
105                        let weight = weight[index_weight];
106
107                        sum += value * weight;
108                    }
109                }
110            }
111        }
112    }
113
114    output[ABSOLUTE_POS] = sum;
115}

117/// Perform a 2D convolution transposition using the direct algorithm.
118///
119/// * `input` - The input feature map
120/// * `weight` - The weights (filter) applied to each kernel
121/// * `bias` - The bias added to each channel
122/// * `options` - The options to use for the convolution
123///
124pub fn conv_transpose2d_direct<R: JitRuntime, E: JitElement>(
125    input: JitTensor<R>,
126    weight: JitTensor<R>,
127    bias: Option<JitTensor<R>>,
128    options: ConvTransposeOptions<2>,
129) -> Result<JitTensor<R>, ConvLaunchError> {
130    let input = into_contiguous(input);
131    let weight = into_contiguous(weight);
132    let [batch_size, _, in_height, in_width] = input.shape.dims();
133    let [_, out_channels, kernel_0, kernel_1] = weight.shape.dims();
134
135    let out_0 = (in_height - 1) * options.stride[0]
136        + options.dilation[0] * (kernel_0 - 1)
137        + options.padding_out[0]
138        - 2 * options.padding[0]
139        + 1;
140    let out_1 = (in_width - 1) * options.stride[1]
141        + options.dilation[1] * (kernel_1 - 1)
142        + options.padding_out[1]
143        - 2 * options.padding[1]
144        + 1;
145
146    let shape_out = Shape::new([batch_size, out_channels * options.groups, out_0, out_1]);
147
148    let output = empty_device::<R, E>(
149        input.client.clone(),
150        input.device.clone(),
151        shape_out.clone(),
152    );
153
154    let bias = match bias {
155        Some(bias) => {
156            let shape = Shape::from([bias.shape.dims[0], 1, 1, 1]);
157            reshape(bias, shape)
158        }
159        None => {
160            let shape = Shape::from([output.shape.dims[0], 1, 1, 1]);
161            zeros_device::<R, E>(input.client.clone(), input.device.clone(), shape)
162        }
163    };
164
165    let cube_dim = CubeDim::default();
166    let cube_count = calculate_cube_count_elemwise(output.shape.num_elements(), cube_dim);
167
168    conv_transpose2d_direct_kernel::launch::<E, R>(
169        &input.client,
170        cube_count,
171        cube_dim,
172        input.as_tensor_arg::<E>(1),
173        weight.as_tensor_arg::<E>(1),
174        bias.as_tensor_arg::<E>(1),
175        output.as_tensor_arg::<E>(1),
176        ConvArgsLaunch::new(
177            ScalarArg::new(options.stride[0] as u32),
178            ScalarArg::new(options.stride[1] as u32),
179            ScalarArg::new(options.dilation[0] as u32),
180            ScalarArg::new(options.dilation[1] as u32),
181            ScalarArg::new(options.padding[0] as u32),
182            ScalarArg::new(options.padding[1] as u32),
183            ScalarArg::new(options.groups as u32),
184        ),
185    );
186
187    Ok(output)
188}
