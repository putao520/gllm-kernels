20/// Specifications for a convolution algorithm
21pub trait Algorithm {
22    type TileMatmul: TileMatmulFamily;
23    type StageMatmul: StageMatmulFamily;
24    type GlobalConvolution: ConvolutionFamily<Self::StageMatmul, Input = Self::Input>;
25    type Selection;
26    type Input;
27
28    fn cube_dim(selection: &Self::Selection) -> CubeDim;
29    fn cube_count(selection: &Self::Selection, problem: &ConvolutionProblem) -> CubeCount;
30
31    /// Make a convolution config from a convolution problem, and launch options
32    fn make_config(
33        input: <Self::GlobalConvolution as ConvolutionConfigFactory>::Input,
34        problem: &ConvolutionProblem,
35        cube_dim: &CubeDim,
36        cube_count: &CubeCount,
37        advanced_config: &AdvancedConfig,
38    ) -> Result<<Self::GlobalConvolution as ConvolutionConfigFactory>::Config, InvalidConfigError>
39    {
40        let config = Self::GlobalConvolution::make_config(
41            input,
42            problem,
43            cube_dim,
44            cube_count,
45            advanced_config,
46        );
47        Self::GlobalConvolution::check_config(&config)?;
48        Ok(config)
49    }
50
51    /// Check availability of the matmul algorithm
52    fn check_availability<R: Runtime, CS: ConvPrecision>(
53        client: &ComputeClient<R::Server, R::Channel>,
54        config: &<Self::GlobalConvolution as ConvolutionConfigFactory>::Config,
55    ) -> Result<(), MatmulAvailabilityError> {
56        Self::GlobalConvolution::check_availability::<R, CS>(client, config)
57    }
58
59    /// Determine whether the given convolution problem is valid to launch (within hardware limits)
60    fn can_launch<R: Runtime, CS: ConvPrecision>(
61        client: &ComputeClient<R::Server, R::Channel>,
62        problem: &ConvolutionProblem,
63        config: &<Self::GlobalConvolution as ConvolutionConfigFactory>::Config,
64        selection: &Self::Selection,
65    ) -> bool {
66        if problem.options.groups > 1 || Self::check_availability::<R, CS>(client, config).is_err()
67        {
68            return false;
69        }
70
71        let cube_count = Self::cube_count(selection, problem);
72        let (max_x, max_y, max_z) = R::max_cube_count();
73        match cube_count {
74            CubeCount::Static(x, y, z) => x <= max_x && y <= max_y && z <= max_z,
75            _ => true,
76        }
77    }
78}

80/// Cmma convolution
81pub struct ImplicitCmmaConv;
82
83impl Algorithm for ImplicitCmmaConv {
84    type TileMatmul = Accelerated;
85    type StageMatmul = stage::multi_buffer::MultiBufferMatmulFamily<Self::TileMatmul>;
86    type GlobalConvolution = ImplicitGemmConvolutionFamily<Self::StageMatmul>;
87    type Selection = ConvSelection;
88    type Input = <Self::GlobalConvolution as ConvolutionConfigFactory>::Input;
89
90    fn cube_dim(selection: &ConvSelection) -> CubeDim {
91        CubeDim::new(
92            selection.matmul.plane_dim,
93            selection.matmul.num_stagess.m,
94            1,
95        )
96    }
97
98    fn cube_count(selection: &ConvSelection, problem: &ConvolutionProblem) -> CubeCount {
99        let m_stage = selection.matmul.num_stagess.m * selection.matmul.tile.m;
100        let n_stage = selection.matmul.num_stagess.n * selection.matmul.tile.n;
101        let cubes_needed_m = (problem.m as u32).div_ceil(m_stage);
102        let cubes_needed_n = (problem.n as u32).div_ceil(n_stage);
103
104        CubeCount::Static(cubes_needed_m, cubes_needed_n, 1)
105    }
106}
