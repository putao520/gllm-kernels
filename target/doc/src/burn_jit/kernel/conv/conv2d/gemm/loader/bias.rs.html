18/// Special loader to broadcast the 1D bias to the 2D accumulator matrix
19#[derive(CubeType)]
20pub struct BiasLoader<CS: ConvPrecision, G: StageConfig> {
21    pub tensor_view: BiasReader<CS::EG>,
22    pub stage: Stage<CS::EA>,
23    pub has_bias: bool,
24    _config: PhantomData<G>,
25}
26
27#[cube]
28impl<CS: ConvPrecision, G: StageConfig> AccumulatorLoader<CS::EG, CS::EA, G> for BiasLoader<CS, G> {
29    fn fill_stage(this: &mut Self, #[comptime] config: G) {
30        if this.has_bias {
31            let stage_dim = config.stage_dim(Ident::Rhs);
32            let line_size = config.line_size(Ident::Out);
33
34            let num_stage_elements = stage_dim.num_elements_y_dim();
35
36            let unit_id = UNIT_POS_Y * config.plane_dim() + UNIT_POS_X;
37            let unit_position_base = unit_id * line_size;
38
39            let mut slice = this.stage.as_slice_mut();
40
41            if unit_position_base < num_stage_elements {
42                let read_line = this
43                    .tensor_view
44                    .load_simple::<G>(unit_position_base, config);
45                slice[unit_id] = Line::cast_from(read_line);
46            }
47        }
48    }
49
50    /// Load accumulator
51    fn load<I: Numeric, Tile: TileMatmul<I, CS::EA>>(
52        this: &mut Self,
53        acc: &mut Tile::Accumulator,
54        tile_n: u32,
55        #[comptime] config: Tile::Config,
56    ) {
57        if this.has_bias {
58            let line_size = config.line_size(Ident::Out);
59            let tile_elems = config.size().n / line_size;
60            let start = tile_n * tile_elems;
61            let slice = this.stage.as_slice_mut().slice(start, start + tile_elems);
62            Tile::fill_accumulator(&slice, acc, 0, config);
63        } else {
64            Tile::zero_accumulator(acc, config);
65        }
66    }
67}

69#[cube]
70impl<CS: ConvPrecision, G: StageConfig> BiasLoader<CS, G> {
71    pub fn new(
72        tensor: VirtualTensor<CS::EG>,
73        n_offset: u32,
74        #[comptime] config: G,
75        #[comptime] has_bias: bool,
76    ) -> Self {
77        if has_bias {
78            let stage = init_stage::<CS::EA, G>(config);
79            let shape_n = tensor.shape(0);
80            let tensor_view = BiasReader::<CS::EG>::new(tensor, n_offset, shape_n);
81
82            BiasLoader::<CS, G> {
83                tensor_view,
84                stage,
85                has_bias,
86                _config: PhantomData::<G>.runtime(),
87            }
88        } else {
89            let stage = init_empty_stage::<CS::EA>();
90            let tensor_view = BiasReader::<CS::EG>::new(tensor, 0, 0);
91            BiasLoader::<CS, G> {
92                stage,
93                tensor_view,
94                has_bias,
95                _config: PhantomData::<G>.runtime(),
96            }
97        }
98    }
99}

101#[cube]
102fn init_stage<ES: Numeric, G: StageConfig>(#[comptime] config: G) -> Stage<ES> {
103    let line_size = config.line_size(Ident::Out);
104
105    let smem = SharedMemory::new_lined(
106        comptime!(config.stage_dim(Ident::Rhs).num_elements_y_dim() / line_size),
107        line_size,
108    );
109
110    Stage::<ES> { smem }
111}
112
113#[cube]
114fn init_empty_stage<ES: Numeric>() -> Stage<ES> {
115    Stage::<ES> {
116        smem: SharedMemory::new(1),
117    }
