49/// Performs matrix multiplication at the global level, with each plane sharing the same responsibilities
50/// - All planes load data to the stage
51/// - All planes are used in the stage matmul computation
52pub struct ImplicitGemmConvolution<
53    CS: ConvPrecision,
54    SMM: stage::StageMatmul<CS::ES, CS::EG, CS::EA>,
55> {
56    _cs: PhantomData<CS>,
57    _stage_matmul: PhantomData<SMM>,
58}
59
60#[cube]
61impl<CS: ConvPrecision, SMM> Convolution<CS, SMM> for ImplicitGemmConvolution<CS, SMM>
62where
63    SMM: stage::StageMatmul<
64        CS::ES,
65        CS::EG,
66        CS::EA,
67        LhsReader = LhsReader<CS::ES>,
68        RhsReader = RhsReader<CS::ES>,
69    >,
70{
71    type LhsLoader = SimpleIm2colLoader<CS, Self::Config>;
72    type Config = HomogeneousConfig<full_load::Config<SMM::Config>>;
73    type RhsLoader = RhsLoader<CS::EG, CS::ES, SMM::Config, CyclicLoading>;
74    type AccumulatorLoader = BiasLoader<CS, SMM::Config>;
75
76    type Out = Unloader<CS::EG>;
77    type Accumulator = SMM::Accumulator;
78
79    fn execute(
80        mut lhs_loader: Self::LhsLoader,
81        mut rhs_loader: Self::RhsLoader,
82        mut acc_loader: Self::AccumulatorLoader,
83        mut out_unloader: Self::Out,
84        acc: &mut Self::Accumulator,
85        k_range: (u32, u32),
86        #[comptime] config: Self::Config,
87    ) {
88        let k_step = config.k_step;
89        let range = k_range.1 - k_range.0;
90        #[allow(unknown_lints)] // `manual_div_ceil` only appeared in 1.83
91        #[allow(clippy::manual_div_ceil)]
92        let num_loops = (range + k_step - 1) / k_step;
93
94        Self::AccumulatorLoader::fill_stage(&mut acc_loader, config.to_smm_config());
95        let (mut lhs_tile, mut rhs_tile) = SMM::init_tile_inputs(config.to_smm_config());
96
97        sync_units();
98
99        SMM::fill_accumulator::<Self::AccumulatorLoader>(
100            &mut acc_loader,
101            acc,
102            config.to_smm_config(),
103        );
104
105        for _ in 0..num_loops {
106            sync_units();
107
108            Self::LhsLoader::fill_stage(&mut lhs_loader, config);
109            Self::RhsLoader::fill_stage(&mut rhs_loader, config.to_matmul_config());
110
111            let lhs_stage_reader = &Self::LhsLoader::as_stage_reader(&lhs_loader);
112            let rhs_stage_reader = &Self::RhsLoader::as_stage_reader(&rhs_loader);
113
114            sync_units();
115
116            SMM::execute(
117                lhs_stage_reader,
118                rhs_stage_reader,
119                &mut lhs_tile,
120                &mut rhs_tile,
121                acc,
122                config.to_smm_config(),
123            );
124
125            Self::LhsLoader::advance_view(&mut lhs_loader, k_step);
126            Self::RhsLoader::advance_view(&mut rhs_loader, k_step);
127        }
128
129        sync_units();
130
131        SMM::read_accumulator::<Self::Out, Self::Config>(
132            acc,
133            &mut out_unloader,
134            config.to_smm_config(),
135            config,
136        );
137    }
138
139    fn init_lhs_loader(
140        lhs: VirtualTensor<CS::EG>,
141        x_offset: u32,
142        y_offset: u32,
143        #[comptime] config: Self::Config,
144    ) -> Self::LhsLoader {
145        Self::LhsLoader::new(
146            lhs,
147            config.out_shape(0),
148            config.out_shape(1),
149            x_offset,
150            y_offset,
151            config,
152        )
153    }
154
155    fn init_rhs_loader(
156        rhs: VirtualTensor<CS::EG>,
157        x_offset: u32,
158        y_offset: u32,
159        #[comptime] config: Self::Config,
160    ) -> Self::RhsLoader {
161        Self::RhsLoader::new::<Self::Config>(rhs, x_offset, y_offset, 0, config)
162    }
163
164    fn init_bias_loader(
165        bias: VirtualTensor<CS::EG>,
166        n_offset: u32,
167        #[comptime] config: Self::Config,
168        #[comptime] has_bias: bool,
169    ) -> Self::AccumulatorLoader {
170        Self::AccumulatorLoader::new(bias, n_offset, config.to_smm_config(), has_bias)
171    }
172
173    fn init_unloader(
174        out: VirtualTensor<CS::EG, ReadWrite>,
175        x_offset: u32,
176        y_offset: u32,
177    ) -> Self::Out {
178        Self::Out::new(out, x_offset, y_offset, 0)
179    }
180
181    fn init_accumulator(#[comptime] config: Self::Config) -> Self::Accumulator {
182        SMM::init_accumulator(config.to_smm_config())
183    }
184}

269#[cube(launch_unchecked)]
270pub(crate) fn implicit_conv<
271    EG: Numeric,
272    ES: Numeric,
273    EA: Numeric,
274    GMM: ConvolutionFamily<SMM>,
275    SMM: StageMatmulFamily,
276>(
277    lhs: &Tensor<Line<EG>>,
278    rhs: &Tensor<Line<EG>>,
279    bias: &Tensor<Line<EG>>,
280    out: &mut Tensor<Line<EG>>,
281    #[comptime] config: GMM::Config,
282    #[comptime] has_bias: bool,
283) {
284    let x_offset = CUBE_POS_X * config.stage_dim(Ident::Lhs).num_elements_x_dim();
285    let y_offset = CUBE_POS_Y * config.stage_dim(Ident::Rhs).num_elements_y_dim();
286    let k_range = (0, rhs.shape(0));
287
288    let lhs = VirtualTensor::<EG>::new::<Tensor<Line<EG>>>(lhs);
289    let rhs = VirtualTensor::<EG>::new::<Tensor<Line<EG>>>(rhs);
290    let bias = VirtualTensor::<EG>::new::<Tensor<Line<EG>>>(bias);
291    let out = VirtualTensor::<EG, ReadWrite>::new::<Tensor<Line<EG>>>(out);
292
293    GMM::Convolution::<(EG, ES, EA)>::execute(
294        GMM::Convolution::<(EG, ES, EA)>::init_lhs_loader(lhs, x_offset, k_range.0, config),
295        GMM::Convolution::<(EG, ES, EA)>::init_rhs_loader(rhs, k_range.0, y_offset, config),
296        GMM::Convolution::<(EG, ES, EA)>::init_bias_loader(bias, y_offset, config, has_bias),
297        GMM::Convolution::<(EG, ES, EA)>::init_unloader(out, x_offset, y_offset),
298        &mut GMM::Convolution::<(EG, ES, EA)>::init_accumulator(config),
299        k_range,
300        config,
301    );
302}
