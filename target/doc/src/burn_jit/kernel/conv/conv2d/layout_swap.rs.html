10/// Efficiently transpose an NCHW tensor to NHWC for use in kernels that prefer NHWC for performance.
11/// Faster than `into_contiguous`, but specialized only for this specific permutation.
12///
13/// # Arguments
14///
15/// * `input` - The input in NCHW format
16///
17/// # Output
18///
19/// The input in NHWC format
20///
21pub fn nchw_to_nhwc<R: JitRuntime, E: JitElement>(input: JitTensor<R>) -> JitTensor<R> {
22    let tiles_per_block = 8;
23    let warp_size = 32;
24    let tile_dim = 16;
25
26    let [batch_size, in_c, h, w] = input.shape.dims();
27    let hw = h * w;
28
29    let out_shape = Shape::new([batch_size, h, w, in_c]);
30    let out = empty_device::<R, E>(input.client.clone(), input.device.clone(), out_shape);
31
32    let tiles_channel = in_c.div_ceil(tile_dim) as u32;
33    let tiles_hw = hw.div_ceil(tile_dim) as u32;
34
35    let block_tiles_y = Ord::min(tiles_channel.next_power_of_two(), tiles_per_block);
36    let block_tiles_x = Ord::min(tiles_per_block / block_tiles_y, tiles_hw);
37
38    let cube_count_y = tiles_channel.div_ceil(block_tiles_y);
39    let cube_count_x = tiles_hw.div_ceil(block_tiles_x);
40    let cube_count_z = batch_size as u32;
41
42    let config = ComptimeConfig {
43        tiles_x: block_tiles_x,
44        warps_per_cube: tiles_per_block,
45        tile_dim: tile_dim as u32,
46        warp_size,
47        num_banks: 32,
48    };
49
50    let cube_dim = CubeDim {
51        x: block_tiles_x * warp_size,
52        y: block_tiles_y,
53        z: 1,
54    };
55    let cube_count = CubeCount::Static(cube_count_x, cube_count_y, cube_count_z);
56
57    let in_vec = max_vectorization(&input);
58    let out_vec = R::supported_line_sizes()
59        .iter()
60        .copied()
61        .find(|vec| in_c % *vec as usize == 0)
62        .unwrap_or(1);
63
64    unsafe {
65        nchw_to_nhwc_kernel::launch_unchecked::<E, R>(
66            &input.client,
67            cube_count,
68            cube_dim,
69            input.as_tensor_arg::<E>(in_vec),
70            out.as_tensor_arg::<E>(out_vec),
71            ScalarArg::new(hw as u32),
72            config,
73        )
74    };
75
76    out
77}

79#[derive(Clone, PartialEq, Eq, Hash, Debug)]
80struct ComptimeConfig {
81    tiles_x: u32,
82    warps_per_cube: u32,
83    tile_dim: u32,
84    warp_size: u32,
85    num_banks: i32,
86}

88#[cube(launch_unchecked)]
89fn nchw_to_nhwc_kernel<E: Numeric>(
90    input: &Tensor<Line<E>>,
91    out: &mut Tensor<Line<E>>,
92    shape_hw: u32,
93    #[comptime] config: ComptimeConfig,
94) {
95    let ComptimeConfig {
96        tiles_x,
97        warps_per_cube,
98        tile_dim,
99        warp_size,
100        num_banks,
101    } = config;
102
103    let tile_elems = tile_dim * tile_dim;
104
105    let unit_pos = UNIT_POS;
106    let intra_warp_unit_idx = unit_pos % 32;
107    let batch = CUBE_POS_Z;
108
109    if batch >= input.shape(0) {
110        return;
111    }
112
113    let batch_offset = batch * input.stride(0);
114
115    let warp_id = plane_broadcast(unit_pos / 32, 0);
116    let warp_id_x = warp_id % tiles_x;
117
118    let tile_x = CUBE_POS_X * tiles_x + warp_id_x;
119    let tile_y = ABSOLUTE_POS_Y;
120
121    let mut shared = SharedMemory::<E>::new(warps_per_cube * tile_elems);
122    let shared_start = warp_id * tile_elems;
123
124    let base_hw = tile_x * tile_dim;
125    let base_c = tile_y * tile_dim;
126
127    let elems_per_unit = tile_elems / warp_size;
128    let unit_start = intra_warp_unit_idx * elems_per_unit;
129
130    let mat_hw_start = unit_start % tile_dim;
131
132    let mat_c = unit_start / tile_dim;
133    let channel = base_c + mat_c;
134    let offset = channel * input.stride(1) + batch_offset;
135
136    let input_vec = input.line_size();
137    let out_vec = out.line_size();
138    let in_max = input.buffer_len() - 1;
139
140    let channels = input.shape(1);
141
142    let mat_offset_base = shared_start + mat_c * tile_dim;
143
144    #[unroll]
145    for hw in range_stepped(0, elems_per_unit, input_vec) {
146        let mat_hw = mat_hw_start + hw;
147        let hw = base_hw + mat_hw;
148        let offset = Min::min((offset + hw) / input_vec, in_max);
149        let value = input[offset];
150
151        let mat_idx = mat_offset_base + mat_hw;
152
153        #[unroll]
154        for v in 0..input_vec {
155            let shared_idx = swizzle(mat_idx + v, num_banks);
156            shared[shared_idx] = value[v];
157        }
158    }
159
160    sync_units();
161
162    let mat_hw = mat_c;
163    let hw = base_hw + mat_hw;
164
165    if hw >= shape_hw {
166        return;
167    }
168
169    let mat_c_start = mat_hw_start;
170    let offset = hw * out.stride(2) + batch_offset;
171    let mat_base = shared_start + mat_hw;
172
173    #[unroll]
174    for ch in range_stepped(0, elems_per_unit, out_vec) {
175        let mat_c = mat_c_start + ch;
176        let ch = base_c + mat_c;
177
178        let mat_idx = mat_base + mat_c * tile_dim;
179        let mut value = Line::empty(out_vec);
180        let offset = (offset + ch) / out_vec;
181
182        #[unroll]
183        for v in 0..out_vec {
184            let shared_idx = swizzle(mat_idx + v * tile_dim, num_banks);
185            value[v] = shared[shared_idx];
186        }
187
188        if ch < channels {
189            out[offset] = value;
190        }
191    }
192}

194#[cube]
195pub fn swizzle(offset: u32, #[comptime] bank_count: i32) -> u32 {
196    let num_bits = comptime!(i32::BITS - bank_count.leading_zeros() - 1);
197    let bit_mask = (1 << num_bits) - 1;
198    let yyy_mask = bit_mask << (num_bits);
199    let mask_shift = num_bits;
200
201    offset ^ ((offset & yyy_mask) >> mask_shift)
202}
