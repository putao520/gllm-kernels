37#[cube(launch_unchecked)]
38fn im2col_kernel<F: Float>(
39    image: &Tensor<F>,
40    columns: &mut Tensor<F>,
41    args: &Im2ColArgs,
42    #[comptime] kernel_w_unroll: Option<u32>,
43    #[comptime] has_padding: bool,
44) {
45    // position shape: [in_channels, batch_size, out_h, out_w]
46    // columns shape: [[in_channels, kernel_h, kernel_w], [batch_size, out_h, out_w]]
47
48    let batch_size = image.shape(0);
49    let height = image.shape(2);
50    let width = image.shape(3);
51
52    let out_h = args.out_h;
53    let out_w = args.out_w;
54
55    if ABSOLUTE_POS > args.num_elements {
56        return;
57    }
58
59    let out_x = ABSOLUTE_POS % out_w;
60    let out_y = ABSOLUTE_POS / out_w % out_h;
61    let batch = ABSOLUTE_POS / (out_w * out_h) % batch_size;
62    let channel = ABSOLUTE_POS / (out_w * out_h * batch_size) % image.shape(1);
63
64    let kernel_w = kernel_w_unroll.unwrap_or(args.kernel_w);
65    let unroll_w = kernel_w_unroll.is_some();
66
67    let image_idx = batch * image.stride(0) + channel * image.stride(1);
68    let col_idx = channel * args.kernel_h * kernel_w * args.col_size_1
69        + batch * out_h * out_w
70        + out_y * out_w
71        + out_x;
72
73    for kernel_y in 0..args.kernel_h {
74        #[unroll(unroll_w)]
75        for kernel_x in 0..kernel_w {
76            let kernel_pos = kernel_y * kernel_w + kernel_x;
77            let col_pos = col_idx + kernel_pos * args.col_size_1;
78
79            if has_padding {
80                let y = (out_y * args.stride_h + kernel_y * args.dilation_h) as i32
81                    - args.padding_h as i32;
82                let x = (out_x * args.stride_w + kernel_x * args.dilation_w) as i32
83                    - args.padding_w as i32;
84                if y >= 0 && x >= 0 && y < height as i32 && x < width as i32 {
85                    let image_ptr = image_idx + y as u32 * width + x as u32;
86                    columns[col_pos] = image[image_ptr];
87                } else {
88                    columns[col_pos] = F::new(0.0)
89                };
90            } else {
91                let y = out_y * args.stride_h + kernel_y * args.dilation_h;
92                let x = out_x * args.stride_w + kernel_x * args.dilation_w;
93                let image_ptr = image_idx + y * image.stride(2) + x * image.stride(3);
94                columns[col_pos] = image[image_ptr];
95            }
96        }
97    }
98}

100#[cfg(not(test))]
101pub(crate) fn batches_per_run(batch_size: usize, out_h: usize, out_w: usize) -> Option<usize> {
102    let cube_count_per_batch = (out_h * out_w).div_ceil(cubecl::PLANE_DIM_APPROX);
103    let max_cube_count = u16::MAX as usize;
104    let max_simultaneous = (max_cube_count / cube_count_per_batch).min(batch_size);
105    if max_simultaneous == 0 {
106        return None;
107    }
108    Some(
109        (0..=max_simultaneous)
110            .rev()
111            .find(|per_run| batch_size % per_run == 0)
112            .expect("Logically not possible"),
113    )
114}

122fn im2col<R: JitRuntime, E: FloatElement>(
123    input: JitTensor<R>,
124    options: ConvOptions<2>,
125    kernel_h: usize,
126    kernel_w: usize,
127    out_h: usize,
128    out_w: usize,
129) -> JitTensor<R> {
130    let input = into_contiguous(input);
131    let [batch_size, in_channels, _, _] = input.shape.dims();
132
133    let col_shape_0 = in_channels * kernel_h * kernel_w;
134    let col_shape_1 = batch_size * out_h * out_w;
135    let shape_col = Shape::new([col_shape_0, col_shape_1]);
136    let columns = empty_device::<R, E>(
137        input.client.clone(),
138        input.device.clone(),
139        shape_col.clone(),
140    );
141
142    let num_elems = in_channels * batch_size * out_h * out_w;
143    let cube_dim = CubeDim::default();
144    let cube_count = calculate_cube_count_elemwise(num_elems, cube_dim);
145
146    let kernel_w_unroll = (kernel_w <= 8).then_some(kernel_w as u32);
147
148    let vectorization = 1;
149
150    unsafe {
151        im2col_kernel::launch_unchecked::<E, R>(
152            &input.client,
153            cube_count,
154            cube_dim,
155            input.as_handle_ref().as_tensor_arg(vectorization),
156            columns.as_handle_ref().as_tensor_arg(vectorization),
157            Im2ColArgsLaunch::new(
158                ScalarArg::new(options.stride[0] as u32),
159                ScalarArg::new(options.stride[1] as u32),
160                ScalarArg::new(options.dilation[0] as u32),
161                ScalarArg::new(options.dilation[1] as u32),
162                ScalarArg::new(options.padding[0] as u32),
163                ScalarArg::new(options.padding[1] as u32),
164                ScalarArg::new(kernel_h as u32),
165                ScalarArg::new(kernel_w as u32),
166                ScalarArg::new(out_h as u32),
167                ScalarArg::new(out_w as u32),
168                ScalarArg::new(col_shape_1 as u32),
169                ScalarArg::new(num_elems as u32),
170            ),
171            kernel_w_unroll,
172            options.padding != [0, 0],
173        )
174    };
175
176    columns
177}

179/// Perform a 2D convolution using the GEMM (im2col) algorithm.
180///
181/// * `input` - The input feature map
182/// * `weight` - The weights (filter) applied to each kernel
183/// * `bias` - The bias added to each channel
184/// * `options` - The options to use for the convolution
185///
186pub fn conv2d_im2col<R: JitRuntime, E: FloatElement>(
187    input: JitTensor<R>,
188    weight: JitTensor<R>,
189    bias: Option<JitTensor<R>>,
190    options: ConvOptions<2>,
191) -> Result<JitTensor<R>, ConvLaunchError> {
192    let [batch_size, in_channels, in_height, in_width] = input.shape.dims();
193    let [out_channels, _, kernel_h, kernel_w] = weight.shape.dims();
194    let groups = options.groups;
195    let out_c_per_group = out_channels / groups;
196
197    let out_h = calculate_conv_output_size(
198        kernel_h,
199        options.stride[0],
200        options.padding[0],
201        options.dilation[0],
202        in_height,
203    );
204    let out_w = calculate_conv_output_size(
205        kernel_w,
206        options.stride[1],
207        options.padding[1],
208        options.dilation[1],
209        in_width,
210    );
211
212    if kernel_h == 1 && kernel_w == 1 && in_height == out_h && in_width == out_w {
213        // Special case for 1x1 kernels (sometimes used to scale the image by a set of weights)
214        return execute_1x1_kernel::<R, E>(input, weight, bias, options);
215    }
216
217    let batches_per_run = batches_per_run(batch_size, out_h, out_w)
218        .expect("Image too large to run even one batch at once");
219    let matmul_shape = Shape::new([groups, out_c_per_group, batches_per_run * out_h * out_w]);
220
221    let mut out = if batches_per_run != batch_size {
222        let runs = batch_size / batches_per_run;
223        let out_shape = Shape::new([runs, out_channels, batches_per_run, out_h, out_w]);
224        let out = empty_device::<R, E>(input.client.clone(), input.device.clone(), out_shape);
225        let in_shape = Shape::new([runs, batches_per_run, in_channels, in_height, in_width]);
226        let input = reshape(input, in_shape);
227        let in_shape_run = Shape::new([batches_per_run, in_channels, in_height, in_width]);
228        for run in 0..runs {
229            let input = index::<R, E>(input.clone(), run);
230            let input = reshape(input, in_shape_run.clone());
231            let out_slice = index::<R, E>(out.clone(), run);
232            let out_slice = reshape(out_slice, matmul_shape.clone());
233            execute::<R, E>(
234                input,
235                weight.clone(),
236                out_slice,
237                options.clone(),
238                out_h,
239                out_w,
240            )?;
241        }
242        let out = swap_dims(out, 1, 2);
243        reshape(out, Shape::new([batch_size, out_channels, out_h, out_w]))
244    } else {
245        let out = empty_device::<R, E>(input.client.clone(), input.device.clone(), matmul_shape);
246        execute::<R, E>(input, weight, out.clone(), options, out_h, out_w)?;
247        let out = reshape(out, Shape::new([out_channels, batch_size, out_h, out_w]));
248        swap_dims(out, 0, 1)
249    };
250
251    if let Some(bias) = bias {
252        let bias = reshape(bias, Shape::new([1, out_channels, 1, 1]));
253        out = launch_binop::<R, E, AddOp>(out, bias)
254    }
255
256    Ok(out)
257}

259fn execute_1x1_kernel<R: JitRuntime, E: FloatElement>(
260    input: JitTensor<R>,
261    weight: JitTensor<R>,
262    bias: Option<JitTensor<R>>,
263    options: ConvOptions<2>,
264) -> Result<JitTensor<R>, ConvLaunchError> {
265    let [batch_size, _, height, width] = input.shape.dims();
266    let [out_channels, in_c_per_grp, _, _] = weight.shape.dims();
267    let groups = options.groups;
268    let out_c_per_grp = out_channels / groups;
269
270    let input = swap_dims(input, 0, 1); // [CNHW]
271
272    let weight = reshape(weight, Shape::new([groups, out_c_per_grp, in_c_per_grp]));
273    let in_shape = Shape::new([groups, in_c_per_grp, batch_size * height * width]);
274    let input = reshape(input, in_shape);
275    let out = matmul::<R, E>(weight, input, None, MatmulStrategy::default())?;
276    let mut out = reshape(out, Shape::new([out_channels, batch_size, height, width]));
277
278    if let Some(bias) = bias {
279        let bias = reshape(bias, Shape::new([out_channels, 1, 1, 1]));
280        out = launch_binop::<R, E, AddOp>(out, bias)
281    }
282
283    Ok(swap_dims(out, 0, 1))
284}

286fn execute<R: JitRuntime, E: FloatElement>(
287    input: JitTensor<R>,
288    weight: JitTensor<R>,
289    out: JitTensor<R>,
290    options: ConvOptions<2>,
291    out_h: usize,
292    out_w: usize,
293) -> Result<(), ConvLaunchError> {
294    let [out_channels, _, kernel_h, kernel_w] = weight.shape.dims();
295    let groups = options.groups;
296
297    let columns = im2col::<R, E>(input, options.clone(), kernel_h, kernel_w, out_h, out_w);
298    let [col_shape_0, col_shape_1] = columns.shape.dims();
299    let col_shape_0 = col_shape_0 / groups;
300    let out_c_per_group = out_channels / groups;
301
302    let columns = reshape(columns, Shape::new([groups, col_shape_0, col_shape_1]));
303    let weight = reshape(weight, Shape::new([groups, out_c_per_group, col_shape_0]));
304
305    matmul::<R, E>(weight, columns, Some(out), Default::default())?;
306
307    Ok(())
