32#[cube(launch)]
33fn conv3d_kernel<F: Float>(
34    input: Tensor<F>,
35    weight: Tensor<F>,
36    bias: Tensor<F>,
37    mut output: Tensor<F>,
38    args: Conv3dArgs,
39    #[comptime] kernel_size_0_unroll: Option<u32>,
40    #[comptime] kernel_size_1_unroll: Option<u32>,
41    #[comptime] kernel_size_2_unroll: Option<u32>,
42) {
43    if ABSOLUTE_POS >= output.len() {
44        return;
45    }
46
47    let in_channels = weight.shape(1);
48
49    let kernel_size_0 = kernel_size_0_unroll.unwrap_or_else(|| weight.shape(2));
50    let unroll_0 = kernel_size_0_unroll.is_some();
51    let kernel_size_1 = kernel_size_1_unroll.unwrap_or_else(|| weight.shape(3));
52    let unroll_1 = kernel_size_1_unroll.is_some();
53    let kernel_size_2 = kernel_size_2_unroll.unwrap_or_else(|| weight.shape(4));
54    let unroll_2 = kernel_size_2_unroll.is_some();
55
56    let b = ABSOLUTE_POS / output.stride(0) % output.shape(0);
57    let oc = ABSOLUTE_POS / output.stride(1) % output.shape(1);
58    let od = ABSOLUTE_POS / output.stride(2) % output.shape(2);
59    let oh = ABSOLUTE_POS / output.stride(3) % output.shape(3);
60    let ow = ABSOLUTE_POS / output.stride(4) % output.shape(4);
61
62    let g = (weight.shape(0) + oc) % args.groups;
63    let ic_start = in_channels * g;
64    let ic_end = ic_start + in_channels;
65    let mut sum = bias[oc];
66
67    let id_base = od * args.conv_stride_0;
68    let ih_base = oh * args.conv_stride_1;
69    let iw_base = ow * args.conv_stride_2;
70
71    let weight_stride_1 = weight.stride(1);
72    let weight_stride_2 = weight.stride(2);
73    let weight_stride_3 = weight.stride(3);
74    let weight_stride_4 = weight.stride(4);
75
76    let input_stride_1 = input.stride(1);
77    let input_stride_2 = input.stride(2);
78    let input_stride_3 = input.stride(3);
79    let input_stride_4 = input.stride(4);
80    let input_shape_2 = input.shape(2);
81    let input_shape_3 = input.shape(3);
82    let input_shape_4 = input.shape(4);
83
84    let border_front = args.padding_0;
85    let border_top = args.padding_1;
86    let border_left = args.padding_2;
87    let border_back = input_shape_2 + args.padding_0;
88    let border_bottom = input_shape_3 + args.padding_1;
89    let border_right = input_shape_4 + args.padding_2;
90
91    let index_input_0 = b * input.stride(0);
92    let index_weight_0 = oc * weight.stride(0);
93
94    for ic in ic_start..ic_end {
95        let index_input_1 = ic * input_stride_1;
96        let index_weight_1 = (ic - ic_start) * weight_stride_1;
97
98        #[unroll(unroll_0)]
99        for kd in 0..kernel_size_0 {
100            #[unroll(unroll_1)]
101            for kh in 0..kernel_size_1 {
102                #[unroll(unroll_2)]
103                for kw in 0..kernel_size_2 {
104                    let id = kd * args.dilation_0 + id_base;
105                    let ih = kh * args.dilation_1 + ih_base;
106                    let iw = kw * args.dilation_2 + iw_base;
107
108                    let within_padding = id >= border_front
109                        && id < border_back
110                        && ih >= border_top
111                        && ih < border_bottom
112                        && iw >= border_left
113                        && iw < border_right;
114
115                    if within_padding {
116                        let id_pad = id - args.padding_0;
117                        let ih_pad = ih - args.padding_1;
118                        let iw_pad = iw - args.padding_2;
119
120                        let index_input = index_input_0
121                            + index_input_1
122                            + id_pad * input_stride_2
123                            + ih_pad * input_stride_3
124                            + iw_pad * input_stride_4;
125
126                        let index_weight = index_weight_0
127                            + index_weight_1
128                            + kd * weight_stride_2
129                            + kh * weight_stride_3
130                            + kw * weight_stride_4;
131
132                        sum += input[index_input] * weight[index_weight];
133                    }
134                }
135            }
136        }
137    }
138
139    output[ABSOLUTE_POS] = sum;
140}

142pub(crate) fn conv3d<R: JitRuntime, E: FloatElement>(
143    input: JitTensor<R>,
144    weight: JitTensor<R>,
145    bias: Option<JitTensor<R>>,
146    options: ConvOptions<3>,
147) -> JitTensor<R> {
148    let input = into_contiguous(input);
149    let weight = into_contiguous(weight);
150    let [batch_size, _, in_depth, in_height, in_width] = input.shape.dims();
151    let [out_channels, _, kernel_0, kernel_1, kernel_2] = weight.shape.dims();
152
153    let out_0 = calculate_conv_output_size(
154        kernel_0,
155        options.stride[0],
156        options.padding[0],
157        options.dilation[0],
158        in_depth,
159    );
160    let out_1 = calculate_conv_output_size(
161        kernel_1,
162        options.stride[1],
163        options.padding[1],
164        options.dilation[1],
165        in_height,
166    );
167    let out_2 = calculate_conv_output_size(
168        kernel_2,
169        options.stride[2],
170        options.padding[2],
171        options.dilation[2],
172        in_width,
173    );
174
175    let shape_out = Shape::new([batch_size, out_channels, out_0, out_1, out_2]);
176
177    let output = empty_device::<R, E>(
178        input.client.clone(),
179        input.device.clone(),
180        shape_out.clone(),
181    );
182
183    let bias = match bias {
184        Some(bias) => {
185            let shape = Shape::from([bias.shape.dims[0], 1, 1, 1, 1]);
186            reshape(bias, shape)
187        }
188        None => {
189            let shape = Shape::from([output.shape.dims[0], 1, 1, 1, 1]);
190            zeros_device::<R, E>(input.client.clone(), input.device.clone(), shape)
191        }
192    };
193
194    let cube_dim = CubeDim::default();
195    let cube_count = calculate_cube_count_elemwise(output.shape.num_elements(), cube_dim);
196
197    conv3d_kernel::launch::<E, R>(
198        &input.client,
199        cube_count,
200        cube_dim,
201        input.as_tensor_arg::<E>(1),
202        weight.as_tensor_arg::<E>(1),
203        bias.as_tensor_arg::<E>(1),
204        output.as_tensor_arg::<E>(1),
205        Conv3dArgsLaunch::new(
206            ScalarArg::new(options.stride[0] as u32),
207            ScalarArg::new(options.stride[1] as u32),
208            ScalarArg::new(options.stride[2] as u32),
209            ScalarArg::new(options.dilation[0] as u32),
210            ScalarArg::new(options.dilation[1] as u32),
211            ScalarArg::new(options.dilation[2] as u32),
212            ScalarArg::new(options.padding[0] as u32),
213            ScalarArg::new(options.padding[1] as u32),
214            ScalarArg::new(options.padding[2] as u32),
215            ScalarArg::new(options.groups as u32),
216        ),
217        Some(kernel_0 as u32),
218        Some(kernel_1 as u32),
219        Some(kernel_2 as u32),
220    );
221
222    output
