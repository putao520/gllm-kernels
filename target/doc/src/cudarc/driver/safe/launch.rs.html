8impl CudaDevice {
9    /// Whether a module and function are currently loaded into the device.
10    pub fn has_func(self: &Arc<Self>, module_name: &str, func_name: &str) -> bool {
11        let modules = self.modules.read();
12        #[cfg(not(feature = "no-std"))]
13        let modules = modules.unwrap();
14
15        modules
16            .get(module_name)
17            .map_or(false, |module| module.has_func(func_name))
18    }
19
20    /// Retrieves a [CudaFunction] that was registered under `module_name` and `func_name`.
21    pub fn get_func(self: &Arc<Self>, module_name: &str, func_name: &str) -> Option<CudaFunction> {
22        let modules = self.modules.read();
23        #[cfg(not(feature = "no-std"))]
24        let modules = modules.unwrap();
25
26        modules
27            .get(module_name)
28            .and_then(|m| m.get_func(func_name))
29            .map(|cu_function| CudaFunction {
30                cu_function,
31                device: self.clone(),
32            })
33    }
34}

103/// Configuration for [result::launch_kernel]
104///
105/// See [cuda docs](https://docs.nvidia.com/cuda/cuda-driver-api/group__CUDA__EXEC.html#group__CUDA__EXEC_1gb8f3dc3031b40da29d5f9a7139e52e15)
106/// for description of each parameter.
107#[derive(Clone, Copy, Debug)]
108pub struct LaunchConfig {
109    /// (width, height, depth) of grid in blocks
110    pub grid_dim: (u32, u32, u32),
111
112    /// (x, y, z) dimension of each thread block
113    pub block_dim: (u32, u32, u32),
114
115    /// Dynamic shared-memory size per thread block in bytes
116    pub shared_mem_bytes: u32,
117}
118
119impl LaunchConfig {
120    /// Creates a [LaunchConfig] with:
121    /// - block_dim == `1024`
122    /// - grid_dim == `(n + 1023) / 1024`
123    /// - shared_mem_bytes == `0`
124    pub fn for_num_elems(n: u32) -> Self {
125        const NUM_THREADS: u32 = 1024;
126        let num_blocks = (n + NUM_THREADS - 1) / NUM_THREADS;
127        Self {
128            grid_dim: (num_blocks, 1, 1),
129            block_dim: (NUM_THREADS, 1, 1),
130            shared_mem_bytes: 0,
131        }
132    }
133}

135/// Consumes a [CudaFunction] to execute asychronously on the device with
136/// params determined by generic parameter `Params`.
137///
138/// This is impl'd multiple times for different number and types of params. In
139/// general, `Params` should impl [DeviceRepr].
140///
141/// ```ignore
142/// # use cudarc::driver::*;
143/// # let dev = CudaDevice::new(0).unwrap();
144/// let my_kernel: CudaFunction = dev.get_func("my_module", "my_kernel").unwrap();
145/// let cfg: LaunchConfig = LaunchConfig {
146///     grid_dim: (1, 1, 1),
147///     block_dim: (1, 1, 1),
148///     shared_mem_bytes: 0,
149/// };
150/// let params = (1i32, 2u64, 3usize);
151/// unsafe { my_kernel.launch(cfg, params) }.unwrap();
152/// ```
153///
154/// # Safety
155///
156/// This is not safe really ever, because there's no garuntee that `Params`
157/// will work for any [CudaFunction] passed in. Great care should be taken
158/// to ensure that [CudaFunction] works with `Params` and that the correct
159/// parameters have `&mut` in front of them.
160///
161/// Additionally, kernels can mutate data that is marked as immutable,
162/// such as `&CudaSlice<T>`.
163///
164/// See [LaunchAsync::launch] for more details
165pub unsafe trait LaunchAsync<Params> {
166    /// Launches the [CudaFunction] with the corresponding `Params`.
167    ///
168    /// # Safety
169    ///
170    /// This method is **very** unsafe.
171    ///
172    /// See cuda documentation notes on this as well:
173    /// <https://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#functions>
174    ///
175    /// 1. `params` can be changed regardless of `&` or `&mut` usage.
176    /// 2. `params` will be changed at some later point after the
177    ///    function returns because the kernel is executed async.
178    /// 3. There are no guaruntees that the `params`
179    ///    are the correct number/types/order for `func`.
180    /// 4. Specifying the wrong values for [LaunchConfig] can result
181    ///    in accessing/modifying values past memory limits.
182    ///
183    /// ## Asynchronous mutation
184    ///
185    /// Since this library queues kernels to be launched on a single
186    /// stream, and really the only way to modify [crate::driver::CudaSlice] is through
187    /// kernels, mutating the same [crate::driver::CudaSlice] with multiple kernels
188    /// is safe. This is because each kernel is executed sequentially
189    /// on the stream.
190    ///
191    /// **Modifying a value on the host that is in used by a
192    /// kernel is undefined behavior.** But is hard to do
193    /// accidentally.
194    ///
195    /// Also for this reason, do not pass in any values to kernels
196    /// that can be modified on the host. This is the reason
197    /// [DeviceRepr] is not implemented for rust primitive
198    /// references.
199    ///
200    /// ## Use after free
201    ///
202    /// Since the drop implementation for [crate::driver::CudaSlice] also occurs
203    /// on the device's single stream, any kernels launched before
204    /// the drop will complete before the value is actually freed.
205    ///
206    /// **If you launch a kernel or drop a value on a different stream
207    /// this may not hold**
208    unsafe fn launch(self, cfg: LaunchConfig, params: Params) -> Result<(), result::DriverError>;
209
210    /// Launch the function on a stream concurrent to the device's default
211    /// work stream.
212    ///
213    /// # Safety
214    /// This method is even more unsafe than [LaunchAsync::launch], all the same rules apply,
215    /// except now things are executing in parallel to each other.
216    ///
217    /// That means that if any of the kernels modify the same memory location, you'll get race
218    /// conditions or potentially undefined behavior.
219    unsafe fn launch_on_stream(
220        self,
221        stream: &CudaStream,
222        cfg: LaunchConfig,
223        params: Params,
224    ) -> Result<(), result::DriverError>;
225
226    /// Launch the cooperative function on a stream concurrent to the device's default
227    /// work stream.
228    ///
229    /// # Safety
230    /// This method is even more unsafe than [LaunchAsync::launch_cooperative], all the same rules apply,
231    /// except now things are executing in parallel to each other.
232    ///
233    /// That means that if any of the kernels modify the same memory location, you'll get race
234    /// conditions or potentially undefined behavior.
235    unsafe fn launch_cooperative(
236        self,
237        cfg: LaunchConfig,
238        params: Params,
239    ) -> Result<(), result::DriverError>;
240}

302macro_rules! impl_launch {
303    ([$($Vars:tt),*], [$($Idx:tt),*]) => {
304unsafe impl<$($Vars: DeviceRepr),*> LaunchAsync<($($Vars, )*)> for CudaFunction {
305    #[inline(always)]
306    unsafe fn launch(
307        self,
308        cfg: LaunchConfig,
309        args: ($($Vars, )*)
310    ) -> Result<(), result::DriverError> {
311        let params = &mut [$(args.$Idx.as_kernel_param(), )*];
312        self.launch_async_impl(cfg, params)
313    }
314
315    #[inline(always)]
316    unsafe fn launch_cooperative(
317        self,
318        cfg: LaunchConfig,
319        args: ($($Vars, )*)
320    ) -> Result<(), result::DriverError> {
321        let params = &mut [$(args.$Idx.as_kernel_param(), )*];
322        self.launch_cooperative_async_impl(cfg, params)
323    }
324
325    #[inline(always)]
326    unsafe fn launch_on_stream(
327        self,
328        stream: &CudaStream,
329        cfg: LaunchConfig,
330        args: ($($Vars, )*)
331    ) -> Result<(), result::DriverError> {
332        let params = &mut [$(args.$Idx.as_kernel_param(), )*];
333        self.par_launch_async_impl(stream, cfg, params)
334    }
335}
336    };

628            f.launch(
629                LaunchConfig::for_num_elems(1),
630                (half::f16::from_f32(1.234),),
631            )
632        }
633        .unwrap();
634        dev.synchronize().unwrap();
635    }

637    const SLOW_KERNELS: &str = "
638extern \"C\" __global__ void slow_worker(const float *data, const size_t len, float *out) {
639    float tmp = 0.0;
640    for(size_t i = 0; i < 1000000; i++) {
641        tmp += data[i % len];
642    }
643    *out = tmp;
644}
645";
646
647    #[test]
648    fn test_par_launch() -> Result<(), DriverError> {
649        let ptx = compile_ptx_with_opts(SLOW_KERNELS, Default::default()).unwrap();
650        let dev = CudaDevice::new(0).unwrap();
651        dev.load_ptx(ptx, "tests", &["slow_worker"]).unwrap();
652        let slice = dev.alloc_zeros::<f32>(1000)?;
653        let mut a = dev.alloc_zeros::<f32>(1)?;
654        let mut b = dev.alloc_zeros::<f32>(1)?;
655        let cfg = LaunchConfig::for_num_elems(1);
656
657        let start = Instant::now();
658        {
659            // launch two kernels on the default stream
660            let f = dev.get_func("tests", "slow_worker").unwrap();
661            unsafe { f.launch(cfg, (&slice, slice.len(), &mut a))? };
662            let f = dev.get_func("tests", "slow_worker").unwrap();
663            unsafe { f.launch(cfg, (&slice, slice.len(), &mut b))? };
664            dev.synchronize()?;
665        }
666        let double_launch_s = start.elapsed().as_secs_f64();
667
668        let start = Instant::now();
669        {
670            // create a new stream & launch them concurrently
671            let stream = dev.fork_default_stream()?;
672            let f = dev.get_func("tests", "slow_worker").unwrap();
673            unsafe { f.launch(cfg, (&slice, slice.len(), &mut a))? };
674            let f = dev.get_func("tests", "slow_worker").unwrap();
675            unsafe { f.launch_on_stream(&stream, cfg, (&slice, slice.len(), &mut b))? };
676            dev.wait_for(&stream)?;
677            dev.synchronize()?;
678        }
679        let par_launch_s = start.elapsed().as_secs_f64();
680
681        assert!(
682            (double_launch_s - 2.0 * par_launch_s).abs() < 20.0 / 100.0,
683            "par={:?} dbl={:?}",
684            par_launch_s,
685            double_launch_s
686        );
687        Ok(())
688    }
