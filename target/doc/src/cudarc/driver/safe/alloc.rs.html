8/// Something that can be copied to device memory and
9/// turned into a parameter for [result::launch_kernel].
10///
11/// # Safety
12///
13/// This is unsafe because a struct should likely
14/// be `#[repr(C)]` to be represented in cuda memory,
15/// and not all types are valid.
16pub unsafe trait DeviceRepr {
17    #[inline(always)]
18    fn as_kernel_param(&self) -> *mut std::ffi::c_void {
19        self as *const Self as *mut _
20    }
21}

71impl<T> CudaSlice<T> {
72    /// Takes ownership of the underlying [sys::CUdeviceptr]. **It is up
73    /// to the owner to free this value**.
74    ///
75    /// Drops the underlying host_buf if there is one.
76    pub fn leak(mut self) -> sys::CUdeviceptr {
77        if let Some(host_buf) = std::mem::take(&mut self.host_buf) {
78            drop(host_buf);
79        }
80        let ptr = self.cu_device_ptr;
81        std::mem::forget(self);
82        ptr
83    }
84}

87    /// Creates a [CudaSlice] from a [sys::CUdeviceptr]. Useful in conjunction with
88    /// [`CudaSlice::leak()`].
89    ///
90    /// # Safety
91    /// - `cu_device_ptr` must be a valid allocation
92    /// - `cu_device_ptr` must space for `len * std::mem::size_of<T>()` bytes
93    /// - The memory may not be valid for type `T`, so some sort of memset operation
94    ///   should be called on the memory.
95    pub unsafe fn upgrade_device_ptr<T>(
96        self: &Arc<Self>,
97        cu_device_ptr: sys::CUdeviceptr,
98        len: usize,
99    ) -> CudaSlice<T> {
100        CudaSlice {
101            cu_device_ptr,
102            len,
103            device: self.clone(),
104            host_buf: None,
105        }
106    }
107}

110    /// Allocates an empty [CudaSlice] with 0 length.
111    pub fn null<T>(self: &Arc<Self>) -> Result<CudaSlice<T>, result::DriverError> {
112        self.bind_to_thread()?;
113        let cu_device_ptr = unsafe {
114            if self.is_async {
115                result::malloc_async(self.stream, 0)?
116            } else {
117                result::malloc_sync(0)?
118            }
119        };
120        Ok(CudaSlice {
121            cu_device_ptr,
122            len: 0,
123            device: self.clone(),
124            host_buf: None,
125        })
126    }

128    /// Allocates device memory and increments the reference counter of [CudaDevice].
129    ///
130    /// # Safety
131    /// This is unsafe because the device memory is unset after this call.
132    pub unsafe fn alloc<T: DeviceRepr>(
133        self: &Arc<Self>,
134        len: usize,
135    ) -> Result<CudaSlice<T>, result::DriverError> {
136        self.bind_to_thread()?;
137        let cu_device_ptr = if self.is_async {
138            result::malloc_async(self.stream, len * std::mem::size_of::<T>())?
139        } else {
140            result::malloc_sync(len * std::mem::size_of::<T>())?
141        };
142        Ok(CudaSlice {
143            cu_device_ptr,
144            len,
145            device: self.clone(),
146            host_buf: None,
147        })
148    }

150    /// Allocates device memory with no associated host memory, and memsets
151    /// the device memory to all 0s.
152    ///
153    /// # Safety
154    /// 1. `T` is marked as [ValidAsZeroBits], so the device memory is valid to use
155    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
156    pub fn alloc_zeros<T: ValidAsZeroBits + DeviceRepr>(
157        self: &Arc<Self>,
158        len: usize,
159    ) -> Result<CudaSlice<T>, result::DriverError> {
160        let mut dst = unsafe { self.alloc(len) }?;
161        self.memset_zeros(&mut dst)?;
162        Ok(dst)
163    }

165    /// Sets all memory to 0 asynchronously.
166    ///
167    /// # Safety
168    /// 1. `T` is marked as [ValidAsZeroBits], so the device memory is valid to use
169    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
170    pub fn memset_zeros<T: ValidAsZeroBits + DeviceRepr, Dst: DevicePtrMut<T>>(
171        self: &Arc<Self>,
172        dst: &mut Dst,
173    ) -> Result<(), result::DriverError> {
174        self.bind_to_thread()?;
175        if self.is_async {
176            unsafe {
177                result::memset_d8_async(*dst.device_ptr_mut(), 0, dst.num_bytes(), self.stream)
178            }
179        } else {
180            unsafe { result::memset_d8_sync(*dst.device_ptr_mut(), 0, dst.num_bytes()) }
181        }
182    }

184    /// Device to device copy (safe version of [result::memcpy_dtod_async]).
185    ///
186    /// # Panics
187    ///
188    /// If the length of the two values are different
189    ///
190    /// # Safety
191    /// 1. We are guarunteed that `src` and `dst` are pointers to the same underlying
192    ///     type `T`
193    /// 2. Since they are both references, they can't have been freed
194    /// 3. Self is [`Arc<Self>`], and this method increments the rc for self
195    pub fn dtod_copy<T: DeviceRepr, Src: DevicePtr<T>, Dst: DevicePtrMut<T>>(
196        self: &Arc<Self>,
197        src: &Src,
198        dst: &mut Dst,
199    ) -> Result<(), result::DriverError> {
200        assert_eq!(src.len(), dst.len());
201        self.bind_to_thread()?;
202        if self.is_async {
203            unsafe {
204                result::memcpy_dtod_async(
205                    *dst.device_ptr_mut(),
206                    *src.device_ptr(),
207                    src.len() * std::mem::size_of::<T>(),
208                    self.stream,
209                )
210            }
211        } else {
212            unsafe {
213                result::memcpy_dtod_sync(
214                    *dst.device_ptr_mut(),
215                    *src.device_ptr(),
216                    src.len() * std::mem::size_of::<T>(),
217                )
218            }
219        }
220    }

222    /// Takes ownership of the host data and copies it to device data asynchronously.
223    ///
224    /// # Safety
225    ///
226    /// 1. Since `src` is owned by this funcion, it is safe to copy data. Any actions executed
227    ///    after this will take place after the data has been successfully copied.
228    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
229    pub fn htod_copy<T: Unpin + DeviceRepr>(
230        self: &Arc<Self>,
231        src: Vec<T>,
232    ) -> Result<CudaSlice<T>, result::DriverError> {
233        let mut dst = unsafe { self.alloc(src.len()) }?;
234        self.htod_copy_into(src, &mut dst)?;
235        Ok(dst)
236    }

238    /// Takes ownership of the host data and copies it to device data asynchronously.
239    ///
240    /// # Safety
241    ///
242    /// 1. Since `src` is owned by this funcion, it is safe to copy data. Any actions executed
243    ///    after this will take place after the data has been successfully copied.
244    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
245    pub fn htod_copy_into<T: DeviceRepr + Unpin>(
246        self: &Arc<Self>,
247        src: Vec<T>,
248        dst: &mut CudaSlice<T>,
249    ) -> Result<(), result::DriverError> {
250        assert_eq!(src.len(), dst.len());
251        dst.host_buf = Some(Pin::new(src));
252        self.bind_to_thread()?;
253        if self.is_async {
254            unsafe {
255                result::memcpy_htod_async(
256                    dst.cu_device_ptr,
257                    dst.host_buf.as_ref().unwrap(),
258                    self.stream,
259                )
260            }?
261        } else {
262            unsafe { result::memcpy_htod_sync(dst.cu_device_ptr, dst.host_buf.as_ref().unwrap()) }?
263        }
264        Ok(())
265    }

267    /// Allocates new device memory and synchronously copies data from `src` into the new allocation.
268    ///
269    /// If you want an asynchronous copy, see [CudaDevice::htod_copy()].
270    ///
271    /// # Safety
272    ///
273    /// 1. Since this function doesn't own `src` it is executed synchronously.

275    pub fn htod_sync_copy<T: DeviceRepr>(
276        self: &Arc<Self>,
277        src: &[T],
278    ) -> Result<CudaSlice<T>, result::DriverError> {
279        let mut dst = unsafe { self.alloc(src.len()) }?;

284    /// Synchronously copies data from `src` into the new allocation.
285    ///
286    /// If you want an asynchronous copy, see [CudaDevice::htod_copy()].
287    ///
288    /// # Panics
289    ///
290    /// If the lengths of slices are not equal, this method panics.
291    ///
292    /// # Safety
293    /// 1. Since this function doesn't own `src` it is executed synchronously.
294    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
295    pub fn htod_sync_copy_into<T: DeviceRepr, Dst: DevicePtrMut<T>>(
296        self: &Arc<Self>,
297        src: &[T],
298        dst: &mut Dst,
299    ) -> Result<(), result::DriverError> {
300        assert_eq!(src.len(), dst.len());
301        self.bind_to_thread()?;
302        if self.is_async {
303            unsafe { result::memcpy_htod_async(*dst.device_ptr_mut(), src, self.stream) }?;
304        } else {
305            unsafe { result::memcpy_htod_sync(*dst.device_ptr_mut(), src) }?;
306        }
307        self.synchronize()
308    }

310    /// Synchronously copies device memory into host memory.
311    /// Unlike [`CudaDevice::dtoh_sync_copy_into`] this returns a [`Vec<T>`].
312    ///
313    /// # Safety
314    /// 1. Since this function doesn't own `dst` (after returning) it is executed synchronously.
315    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
316    #[allow(clippy::uninit_vec)]
317    pub fn dtoh_sync_copy<T: DeviceRepr, Src: DevicePtr<T>>(
318        self: &Arc<Self>,
319        src: &Src,
320    ) -> Result<Vec<T>, result::DriverError> {
321        let mut dst = Vec::with_capacity(src.len());
322        unsafe { dst.set_len(src.len()) };
323        self.dtoh_sync_copy_into(src, &mut dst)?;
324        Ok(dst)
325    }

327    /// Synchronously copies device memory into host memory
328    ///
329    /// Use [`CudaDevice::dtoh_sync_copy`] if you need [`Vec<T>`] and can't provide
330    /// a correctly sized slice.
331    ///
332    /// # Panics
333    ///
334    /// If the lengths of slices are not equal, this method panics.
335    ///
336    /// # Safety
337    /// 1. Since this function doesn't own `dst` it is executed synchronously.
338    /// 2. Self is [`Arc<Self>`], and this method increments the rc for self
339    pub fn dtoh_sync_copy_into<T: DeviceRepr, Src: DevicePtr<T>>(
340        self: &Arc<Self>,
341        src: &Src,
342        dst: &mut [T],
343    ) -> Result<(), result::DriverError> {
344        assert_eq!(src.len(), dst.len());
345        self.bind_to_thread()?;
346        if self.is_async {
347            unsafe { result::memcpy_dtoh_async(dst, *src.device_ptr(), self.stream) }?;
348        } else {
349            unsafe { result::memcpy_dtoh_sync(dst, *src.device_ptr()) }?;
350        }
351        self.synchronize()
352    }

354    /// Synchronously de-allocates `src` and converts it into it's host value.
355    /// You can just [drop] the slice if you don't need the host data.
356    ///
357    /// # Safety
358    /// 1. Self is [`Arc<Self>`], and this method increments the rc for self
359    pub fn sync_reclaim<T: Clone + Default + DeviceRepr + Unpin>(
360        self: &Arc<Self>,
361        mut src: CudaSlice<T>,
362    ) -> Result<Vec<T>, result::DriverError> {
363        let buf = src.host_buf.take();
364        let mut buf = buf.unwrap_or_else(|| {
365            let mut b = Vec::with_capacity(src.len);
366            b.resize(src.len, Default::default());
367            Pin::new(b)
368        });
369        self.dtoh_sync_copy_into(&src, &mut buf)?;
370        Ok(Pin::into_inner(buf))
371    }
372
373    /// Synchronizes the stream.
374    pub fn synchronize(self: &Arc<Self>) -> Result<(), result::DriverError> {
375        self.bind_to_thread()?;
376        unsafe { result::stream::synchronize(self.stream) }
377    }

380/// Marker trait to indicate that the type is valid
381/// when all of its bits are set to 0.
382///
383/// # Safety
384/// Not all types are valid when all bits are set to 0.
385/// Be very sure when implementing this trait!
386pub unsafe trait ValidAsZeroBits {}

400unsafe impl ValidAsZeroBits for f32 {}

403unsafe impl ValidAsZeroBits for half::f16 {}
404#[cfg(feature = "f16")]
405unsafe impl ValidAsZeroBits for half::bf16 {}
406unsafe impl<T: ValidAsZeroBits, const M: usize> ValidAsZeroBits for [T; M] {}
407/// Implement `ValidAsZeroBits` for tuples if all elements are `ValidAsZeroBits`,
408///
409/// # Note
410/// This will also implement `ValidAsZeroBits` for a tuple with one element
411macro_rules! impl_tuples {
412    ($t:tt) => {
413        impl_tuples!(@ $t);
414    };
415    // the $l is in front of the reptition to prevent parsing ambiguities
416    ($l:tt $(,$t:tt)+) => {
417        impl_tuples!($($t),+);
418        impl_tuples!(@ $l $(,$t)+);
419    };
420    (@ $($t:tt),+) => {
421        unsafe impl<$($t: ValidAsZeroBits,)+> ValidAsZeroBits for ($($t,)+) {}
422    };
423}
