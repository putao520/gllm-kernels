9/// A memory pool that allocates buffers in a range of sizes and reuses them to minimize allocations.
10///
11/// - Only one slice is supported per page, due to the limitations in WGPU where each buffer should only bound with
12///   either read only or read_write slices but not a mix of both.
13/// - The pool uses a ring buffer to efficiently manage and reuse pages.
14pub struct ExclusiveMemoryPool {
15    pages: HashMap<StorageId, MemoryPage>,
16    slices: HashMap<SliceId, Slice>,
17    ring_buffer: Vec<StorageId>,
18    index: usize,
19    max_page_size: u64,
20    alignment: u64,
21    dealloc_period: u64,
22    last_dealloc: u64,
23}
24
25struct MemoryPage {
26    slice_id: SliceId,
27    dealloc_mark: bool,
28}
29
30impl ExclusiveMemoryPool {
31    pub(crate) fn new(page_size: u64, alignment: u64, dealloc_period: u64) -> Self {
32        // Pages should be allocated to be aligned.
33        assert_eq!(page_size % alignment, 0);
34        Self {
35            pages: HashMap::new(),
36            slices: HashMap::new(),
37            ring_buffer: Vec::new(),
38            index: 0,
39            max_page_size: page_size,
40            alignment,
41            dealloc_period,
42            last_dealloc: 0,
43        }
44    }
45
46    /// Finds a free page that can contain the given size
47    /// Returns a slice on that page if successful.
48    fn get_free_page(&mut self, locked: Option<&MemoryLock>) -> Option<SliceId> {
49        for _ in 0..self.ring_buffer.len() {
50            let storage_id = &self.ring_buffer[self.index];
51            if let Some(locked) = locked.as_ref() {
52                if locked.is_locked(storage_id) {
53                    continue;
54                }
55            }
56
57            let page = self.pages.get_mut(storage_id).unwrap();
58            page.dealloc_mark = false;
59            let slice = self.slices.get(&page.slice_id).unwrap();
60            self.index = (self.index + 1) % self.ring_buffer.len();
61            if slice.handle.is_free() {
62                return Some(page.slice_id);
63            }
64        }
65
66        None
67    }
68}
69
70impl MemoryPool for ExclusiveMemoryPool {
71    /// Returns the resource from the storage, for the specified handle.
72    fn get(&self, binding: &SliceBinding) -> Option<&StorageHandle> {
73        self.slices.get(binding.id()).map(|s| &s.storage)
74    }
75
76    /// Reserves memory of specified size using the reserve algorithm, and return
77    /// a handle to the reserved memory.
78    ///
79    /// Also clean ups, merging free slices together if permitted by the merging strategy
80    fn reserve<Storage: ComputeStorage>(
81        &mut self,
82        storage: &mut Storage,
83        size: u64,
84        exclude: Option<&MemoryLock>,
85    ) -> SliceHandle {
86        let page = self.get_free_page(exclude);
87        let slice_id = if let Some(page) = page {
88            page
89        } else {
90            *self.alloc(storage, self.max_page_size).id()
91        };
92
93        let padding = calculate_padding(size, self.alignment);
94        let slice = self.slices.get_mut(&slice_id).unwrap();
95        // Return a smaller part of the slice. By construction, we only ever
96        // get a page with a size > size, so this is ok to do.
97        slice.storage.utilization = StorageUtilization { offset: 0, size };
98        slice.padding = padding;
99        slice.handle.clone()
100    }
101
102    fn alloc<Storage: ComputeStorage>(&mut self, storage: &mut Storage, size: u64) -> SliceHandle {
103        let storage = storage.alloc(size);
104        self.ring_buffer.push(storage.id);
105
106        let handle = SliceHandle::new();
107        let padding = calculate_padding(size, self.alignment);
108        let slice = Slice::new(storage.clone(), handle, padding);
109
110        let handle_slice = slice.handle.clone();
111        let slice_id = *slice.handle.id();
112        self.pages.insert(
113            storage.id,
114            MemoryPage {
115                slice_id,
116                dealloc_mark: false,
117            },
118        );
119        self.slices.insert(slice_id, slice);
120        handle_slice
121    }
122
123    fn get_memory_usage(&self) -> MemoryUsage {
124        let used_slices: Vec<_> = self
125            .slices
126            .values()
127            .filter(|slice| !slice.is_free())
128            .collect();
129
130        MemoryUsage {
131            number_allocs: used_slices.len() as u64,
132            bytes_in_use: used_slices.iter().map(|s| s.storage.size()).sum(),
133            bytes_padding: used_slices.iter().map(|s| s.padding).sum(),
134            bytes_reserved: self.pages.len() as u64 * self.max_page_size,
135        }
136    }
137
138    fn max_alloc_size(&self) -> u64 {
139        self.max_page_size
140    }
141
142    fn cleanup<Storage: ComputeStorage>(&mut self, storage: &mut Storage, alloc_nr: u64) {
143        let elapsed = alloc_nr - self.last_dealloc;
144
145        if elapsed < self.dealloc_period {
146            return;
147        }
148
149        self.last_dealloc = alloc_nr;
150
151        let deallocations: HashSet<_> = self
152            .pages
153            .iter_mut()
154            .filter_map(|(storage_id, page)| {
155                let slice = self.slices.get(&page.slice_id).unwrap();
156
157                if slice.is_free() {
158                    // If not marked yet the memory might just have been freed.
159                    if !page.dealloc_mark {
160                        page.dealloc_mark = true;
161                        None
162                    } else {
163                        Some(*storage_id)
164                    }
165                } else {
166                    None
167                }
168            })
169            .collect();
170
171        // Perform any deallocations if necessary.
172        if !deallocations.is_empty() {
173            for storage_id in deallocations.iter() {
174                let slice_id = self.pages[storage_id].slice_id;
175                self.pages.remove(storage_id);
176                self.slices.remove(&slice_id);
177                storage.dealloc(*storage_id);
178            }
179
180            self.index = 0;
181            self.ring_buffer
182                .retain(|storage| !deallocations.contains(storage));
183        }
184    }
185}
