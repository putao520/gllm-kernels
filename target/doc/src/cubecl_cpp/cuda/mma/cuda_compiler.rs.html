1use crate::{
2    cuda::{arch::CudaArchitecture, CudaDialect},

10const WMMA_NAMESPACE: &str = "nvcuda::wmma";
11const WMMA_MINIMUM_VERSION: u32 = 70;
12
13#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
14pub struct CudaWmmaCompiler {}
15
16impl WmmaCompiler<CudaDialect<Self>> for CudaWmmaCompiler {
17    type Architecture = CudaArchitecture;
18
19    fn wmma_includes(f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
20        f.write_str("#include <mma.h>\n")
21    }
22
23    fn deftypes(_f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        Ok(())
25    }
26
27    fn local_variables(_f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
28        Ok(())
29    }
30
31    fn compile_fragment_ident(
32        ident: &FragmentIdent<CudaDialect<Self>>,
33        f: &mut std::fmt::Formatter<'_>,
34    ) -> std::fmt::Result {
35        wmma_api_base::compile_fragment_ident(WMMA_NAMESPACE, ident, f)
36    }
37
38    fn compile_fragment_layout(
39        layout: &FragmentLayout<CudaDialect<Self>>,
40        f: &mut std::fmt::Formatter<'_>,
41    ) -> std::fmt::Result {
42        wmma_api_base::compile_fragment_layout(WMMA_NAMESPACE, layout, f)
43    }
44
45    fn compile_fragment(
46        fragment: &Fragment<CudaDialect<Self>>,
47        f: &mut std::fmt::Formatter<'_>,
48    ) -> std::fmt::Result {
49        wmma_api_base::compile_fragment(WMMA_NAMESPACE, fragment, f)
50    }
51
52    fn compile_instruction(
53        instruction: &WmmaInstruction<CudaDialect<Self>>,
54        f: &mut std::fmt::Formatter<'_>,
55    ) -> std::fmt::Result {
56        wmma_api_base::compile_instruction(WMMA_NAMESPACE, instruction, f)
57    }
58
59    fn supported_wmma_combinations(arch: &Self::Architecture) -> SupportedWmmaCombinations {
60        let mut result: SupportedWmmaCombinations = vec![];
61        if arch.version >= WMMA_MINIMUM_VERSION {
62            let tdims = vec![(16, 16, 16), (32, 8, 16), (8, 32, 16)];
63            // Types fully supported.
64            let types = vec![
65                (
66                    gpu::Elem::Float(gpu::FloatKind::F16), // a
67                    gpu::Elem::Float(gpu::FloatKind::F16), // b
68                    gpu::Elem::Float(gpu::FloatKind::F16), // c
69                ),
70                (
71                    gpu::Elem::Float(gpu::FloatKind::F16),
72                    gpu::Elem::Float(gpu::FloatKind::F16),
73                    gpu::Elem::Float(gpu::FloatKind::F32),
74                ),
75                (
76                    gpu::Elem::Float(gpu::FloatKind::BF16),
77                    gpu::Elem::Float(gpu::FloatKind::BF16),
78                    gpu::Elem::Float(gpu::FloatKind::F32),
79                ),
80            ];
81            let combinations: SupportedWmmaCombinations = types
82                .into_iter()
83                .map(|(a, b, c)| {
84                    let dimensions = tdims.clone();
85                    (a, b, c, dimensions)
86                })
87                .collect();
88            result.extend(combinations);
89            result.push((
90                gpu::Elem::Float(gpu::FloatKind::TF32),
91                gpu::Elem::Float(gpu::FloatKind::TF32),
92                gpu::Elem::Float(gpu::FloatKind::F32),
93                vec![(16, 16, 8)],
94            ));
95        }
96        result
97    }
98}
